-- Invisible drag handle for main frame
local dragHandle = Instance.new("Frame", frame)

dragHandle.Position = UDim2.new(0, 0, 0, 0)
dragHandle.BackgroundTransparency = 1
dragHandle.Active = true

local function updateDrag(input)
    if globals.isDragging2 then
        local delta = input.Position - globals.dragStart
        frame.Position = UDim2.new(globals.startPos.X.Scale, globals.startPos.X.Offset + delta.X, globals.startPos.Y.Scale, globals.startPos.Y.Offset + delta.Y)
    end
end

dragHandle.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        globals.isDragging2 = true
        globals.dragStart = input.Position
        globals.startPos = frame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                globals.isDragging2 = false
            end
        end)
    end
end)

dragHandle.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        updateDrag(input)
    end
end)

globals.UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        updateDrag(input)
    end
end)

globals.UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        globals.isDragging2 = false
    end
end)

-- Function to update right panel position based on frame position
local function updateRightPanelPosition()
    local screenSize, frameAbsPos, frameAbsSize, frameCenterX, screenCenterX, panelWidth, padding = 
        workspace.CurrentCamera.ViewportSize, frame.AbsolutePosition, frame.AbsoluteSize, 
        frame.AbsolutePosition.X + (frame.AbsoluteSize.X / 2), workspace.CurrentCamera.ViewportSize.X / 2, 320, 10
    
    if frameCenterX < screenCenterX then
        rightPanel.Position = UDim2.new(0, frameAbsSize.X + padding, 0, 0)
    else
        rightPanel.Position = UDim2.new(0, -(panelWidth + padding), 0, 0)
    end
end

-- Function to show right panel with proper positioning
local function showRightPanel()
    rightPanel.Visible = true
    updateRightPanelPosition()
end

-- Function to hide right panel
local function hideRightPanel()
    rightPanel.Visible = false
    updateRightPanelPosition()
end

globals.connection = game:GetService("RunService").Heartbeat:Connect(function()
    if frame.Position ~= globals.lastPosition then
        globals.lastPosition = frame.Position
        updateRightPanelPosition()
    end
end)

-- Also update when screen size changes
workspace.CurrentCamera:GetPropertyChangedSignal("ViewportSize"):Connect(updateRightPanelPosition)

-- Add smooth scrolling behavior
local function smoothScroll(frame)
    local uis = game:GetService("UserInputService")
    local function onScroll(_, delta)
        frame.CanvasPosition = Vector2.new(
            frame.CanvasPosition.X,
            math.clamp(frame.CanvasPosition.Y - (delta * 80),
                0,
                frame.AbsoluteCanvasSize.Y - frame.AbsoluteSize.Y
            )
        )
    end
    frame.MouseWheelForward:Connect(function() onScroll(nil, 1) end)
    frame.MouseWheelBackward:Connect(function() onScroll(nil, -1) end)
end

-- Fix sustain issues by ensuring proper note-off timing
function MidiProcessor:fixSustain()
    -- Don't modify the notes, just ensure they're properly sorted
    -- The MIDI parser should already have correct note-on/note-off timing
    table.sort(self.notes, function(a, b) return a[1] < b[1] end)
    
    -- Validate note pairs but don't modify timing
    local activeNotes = {}  -- Track active notes: key -> {time, count}
    local validatedNotes = {}
    
    for _, note in ipairs(self.notes) do
        local time = note[1]
        local noteData = note[2]
        
        if string.find(noteData, "tempo") then
            -- Keep tempo events as-is
            table.insert(validatedNotes, note)
        elseif string.find(noteData, "~") then
            -- Note-off event
            local noteKey = string.sub(noteData, 2) -- Remove ~ prefix
            
            if activeNotes[noteKey] and activeNotes[noteKey].count > 0 then
                -- Valid note-off, decrease count
                activeNotes[noteKey].count = activeNotes[noteKey].count - 1
                if activeNotes[noteKey].count == 0 then
                    activeNotes[noteKey] = nil
                end
                table.insert(validatedNotes, note)
            else
                -- Orphaned note-off - skip it to prevent invalid sustain
                -- This helps fix notes being held longer than expected
            end
        else
            -- Note-on event(s)
            if #noteData == 1 then
                -- Single note
                if not activeNotes[noteData] then
                    activeNotes[noteData] = {time = time, count = 1}
                else
                    activeNotes[noteData].count = activeNotes[noteData].count + 1
                end
                table.insert(validatedNotes, note)
            else
                -- Chord - handle each note individually
                for i = 1, #noteData do
                    local noteKey = string.sub(noteData, i, i)
                    if not activeNotes[noteKey] then
                        activeNotes[noteKey] = {time = time, count = 1}
                    else
                        activeNotes[noteKey].count = activeNotes[noteKey].count + 1
                    end
                end
                table.insert(validatedNotes, note)
            end
        end
    end
    
    self.notes = validatedNotes
end

function MidiProcessor:checkStartSequence()
    for i = 1, #self.startSequence do
        if #self.startSequence[i] == self.startCounter[i] then
            return true
        end
    end
    return false
end

function MidiProcessor:readLength()
    local length = 0
    local contFlag = true
    
    while contFlag and self.itr <= #self.bytes do
        local byte = self.bytes[self.itr]
        if bit32.rshift(bit32.band(byte, 0x80), 7) == 1 then
            length = bit32.lshift(length, 7) + bit32.band(byte, 0x7F)
        else
            contFlag = false
            length = bit32.lshift(length, 7) + bit32.band(byte, 0x7F)
        end
        self.itr = self.itr + 1
    end
    return length
end

function MidiProcessor:getInt(i)
    local k = 0
    for n = 0, i - 1 do
        if self.itr + n <= #self.bytes then
            k = bit32.lshift(k, 8) + self.bytes[self.itr + n]
        end
    end
    self.itr = self.itr + i
    return k
end

function MidiProcessor:readMThd()
    self.headerLength = self:getInt(4)
    self.format = self:getInt(2)
    self.tracks = self:getInt(2)
    local div = self:getInt(2)
    self.divisionType = bit32.rshift(bit32.band(div, 0x8000), 15)
    self.division = bit32.band(div, 0x7FFF)
end

function MidiProcessor:readMTrk()
    local length = self:getInt(4)
    self:readMidiTrackEvent(length)
end

function MidiProcessor:readText(length)
    local s = ""
    local start = self.itr
    while self.itr < length + start and self.itr <= #self.bytes do
        s = s .. string.char(self.bytes[self.itr])
        self.itr = self.itr + 1
    end
    return s
end

function MidiProcessor:readMidiMetaEvent(deltaT)
    if self.itr > #self.bytes then return false end
    
    local eventType = self.bytes[self.itr]
    self.itr = self.itr + 1
    local length = self:readLength()
    
    if eventType == 0x2F then
        self.itr = self.itr + 2
        return false
    elseif eventType == 0x01 or eventType == 0x02 or eventType == 0x03 or eventType == 0x04 or 
           eventType == 0x05 or eventType == 0x06 or eventType == 0x07 or eventType == 0x08 or 
           eventType == 0x09 or eventType == 0x0A or eventType == 0x0C then
        self:readText(length)
    elseif eventType == 0x51 then
        local tempo = math.floor(60000000 / self:getInt(3) + 0.5)
        self.tempo = tempo
        table.insert(self.notes, {self.deltaTime / self.division, "tempo=" .. tempo})
    else
        self.itr = self.itr + length
    end
    return true
end

function MidiProcessor:readMidiTrackEvent(length)
    self.deltaTime = 0
    local start = self.itr
    local continueFlag = true
    
    while length > self.itr - start and continueFlag and self.itr <= #self.bytes do
        local deltaT = self:readLength()
        self.deltaTime = self.deltaTime + deltaT
        
        if self.itr <= #self.bytes then
            if self.bytes[self.itr] == 0xFF then
                self.itr = self.itr + 1
                continueFlag = self:readMidiMetaEvent(deltaT)
            elseif self.bytes[self.itr] >= 0xF0 and self.bytes[self.itr] <= 0xF7 then
                self.runningStatusSet = false
                self.runningStatus = -1
            else
                self:readVoiceEvent(deltaT)
            end
        end
    end
    self.itr = start + length
end

function MidiProcessor:readVoiceEvent(deltaT)
    if self.itr > #self.bytes then return end
    
    local eventType
    if self.bytes[self.itr] < 0x80 and self.runningStatusSet then
        eventType = self.runningStatus
    else
        eventType = self.bytes[self.itr]
        if eventType >= 0x80 and eventType <= 0xF7 then
            self.runningStatus = eventType
            self.runningStatusSet = true
        end
        self.itr = self.itr + 1
    end
    
    if self.itr + 1 > #self.bytes then return end
    
    if bit32.rshift(eventType, 4) == 0x9 then
        local key = self.bytes[self.itr]
        self.itr = self.itr + 1
        local velocity = self.bytes[self.itr]
        self.itr = self.itr + 1
        
        local map = key - 36
        while map >= #self.virtualPianoScale do
            map = map - 12
        end
        while map < 0 do
            map = map + 12
        end
        map = map + 1
        
        if velocity == 0 then
            table.insert(self.notes, {self.deltaTime / self.division, "~" .. self.virtualPianoScale[map]})
        else
            table.insert(self.notes, {self.deltaTime / self.division, self.virtualPianoScale[map]})
            self.keyPressCount = self.keyPressCount + 1
        end
        
    elseif bit32.rshift(eventType, 4) == 0x8 then
        local key = self.bytes[self.itr]
        self.itr = self.itr + 1
        local velocity = self.bytes[self.itr]
        self.itr = self.itr + 1
        
        local map = key - 36
        while map >= #self.virtualPianoScale do
            map = map - 12
        end
        while map < 0 do
            map = map + 12
        end
        map = map + 1
        
        table.insert(self.notes, {self.deltaTime / self.division, "~" .. self.virtualPianoScale[map]})
        
    elseif not (bit32.rshift(eventType, 4) == 0x8 or bit32.rshift(eventType, 4) == 0x9 or 
                bit32.rshift(eventType, 4) == 0xA or bit32.rshift(eventType, 4) == 0xB or 
                bit32.rshift(eventType, 4) == 0xD or bit32.rshift(eventType, 4) == 0xE) then
        self.itr = self.itr + 1
    else
        self.itr = self.itr + 2
    end
end

function MidiProcessor:readEvents()
    while self.itr + 1 <= #self.bytes do
        for i = 1, #self.startCounter do
            self.startCounter[i] = 0
        end
        
        while self.itr + 1 <= #self.bytes and not self:checkStartSequence() do
            for i = 1, #self.startSequence do
                if self.itr <= #self.bytes and self.bytes[self.itr] == self.startSequence[i][self.startCounter[i] + 1] then
                    self.startCounter[i] = self.startCounter[i] + 1
                else
                    self.startCounter[i] = 0
                end
            end
            
            self.itr = self.itr + 1
            
            if self.startCounter[1] == 4 then
                self:readMThd()
            elseif self.startCounter[2] == 4 then
                self:readMTrk()
            end
        end
    end
end

function MidiProcessor:cleanNotes()
    -- Sort notes by time
    table.sort(self.notes, function(a, b) return a[1] < b[1] end)
    
    -- Remove exact duplicates only
    local i = 1
    while i < #self.notes do
        local aTime = self.notes[i][1]
        local bTime = self.notes[i + 1][1]
        local aNotes = self.notes[i][2]
        local bNotes = self.notes[i + 1][2]
        
        -- Only remove exact duplicates (same time and same note)
        if aTime == bTime and aNotes == bNotes then
            table.remove(self.notes, i + 1)
        else
            i = i + 1
        end
    end
end

-- Detect and group chords based on timing windows
function MidiProcessor:detectChords(chordWindow)
    chordWindow = chordWindow or 0.05  -- Default 50ms window for chord detection
    
    local groupedNotes = {}
    local currentGroup = {}
    local groupStartTime = nil
    
    for _, note in ipairs(self.notes) do
        local time = note[1]
        local noteData = note[2]
        
        -- Skip tempo and release events for chord detection
        if not string.find(noteData, "tempo") and not string.find(noteData, "~") then
            if #currentGroup == 0 then
                -- Start new group
                groupStartTime = time
                table.insert(currentGroup, note)
            elseif time - groupStartTime <= chordWindow then
                -- Add to current group
                table.insert(currentGroup, note)
            else
                -- Process current group and start new one
                if #currentGroup == 1 then
                    -- Single note
                    table.insert(groupedNotes, currentGroup[1])
                else
                    -- Chord - combine notes but keep original timing
                    local chordKeys = ""
                    for _, chordNote in ipairs(currentGroup) do
                        chordKeys = chordKeys .. chordNote[2]
                    end
                    -- Remove duplicates in chord
                    local uniqueKeys = self:removeDuplicateChars(chordKeys)
                    table.insert(groupedNotes, {groupStartTime, uniqueKeys})
                end
                
                -- Start new group
                currentGroup = {note}
                groupStartTime = time
            end
        else
            -- Add tempo and release events as-is
            table.insert(groupedNotes, note)
        end
    end
    
    -- Process final group
    if #currentGroup > 0 then
        if #currentGroup == 1 then
            table.insert(groupedNotes, currentGroup[1])
        else
            local chordKeys = ""
            for _, chordNote in ipairs(currentGroup) do
                chordKeys = chordKeys .. chordNote[2]
            end
            local uniqueKeys = self:removeDuplicateChars(chordKeys)
            table.insert(groupedNotes, {groupStartTime, uniqueKeys})
        end
    end
    
    self.notes = groupedNotes
end

-- Helper function to remove duplicate characters in a string
function MidiProcessor:removeDuplicateChars(str)
    local seen = {}
    local result = ""
    
    for i = 1, #str do
        local char = string.sub(str, i, i)
        if not seen[char] then
            seen[char] = true
            result = result .. char
        end
    end
    
    return result
end

function MidiProcessor:formatOutput()
    local output = ""
    for _, note in ipairs(self.notes) do
        local time = note[1]
        local keys = note[2]
        output = output .. string.format('{ time = %.6f, keys = "%s" },\n', time, keys)
    end
    return output
end

-- File handling functions
local function readMidiFile(filePath)
    local success, result = pcall(function()
        return readfile(filePath)
    end)
    
    if success then
        local bytes = {}
        for i = 1, #result do
            table.insert(bytes, string.byte(result, i))
        end
        return bytes
    else
        warn("Failed to read file: " .. filePath)
        return nil
    end
end

local function getFolderCategories()
    local categories = {}
    local success, entries = pcall(function()
        return listfiles("ChordRushMidi")
    end)
    if success then
        for _, path in ipairs(entries) do
            path = path:gsub("\\", "/")
            local name = path:match("ChordRushMidi/([^/\\]+)[/\\]?$")
            if name and not name:find("%.") then
                local isFolder = pcall(function()
                    local inside = listfiles("ChordRushMidi/" .. name)
                    return typeof(inside) == "table"
                end)
                table.insert(categories, name)
            end
        end
    end
    return categories
end

local function getMidiFilesFromCategory(category)
    local midiFiles = {}
    local folderPath = "ChordRushMidi/" .. category
    
    local success, files = pcall(function()
        return listfiles(folderPath)
    end)
    
    if success then
        for _, file in ipairs(files) do
            if string.lower(string.sub(file, -4)) == ".mid" or string.lower(string.sub(file, -4)) == ".smf" then
                table.insert(midiFiles, file)
            end
        end
    end
    
    return midiFiles
end

-- New function to get MIDI files directly from ChordRushMidi root

    local success, files = pcall(function()
        return listfiles("ChordRushMidi")

    if success then
        for _, file in ipairs(files) do

                table.insert(rootMidiFiles, file)
            end
        end
    end
    
    return rootMidiFiles
end

-- Universal HTTP request fallback
local function getHttp()
    return syn and syn.request
        or http and http.request
        or http_request
        or request
        or (fluxus and fluxus.request)
        or (isexecutorclosure and isexecutorclosure().request)
        or nil
end

-- Encode unsafe characters for GitHub URLs
local function urlEncode(str)
	return str:gsub("([^%w%-%_%.%~])", function(c)
		return string.format("%%%02X", string.byte(c))
	end)
end

-- Modified loader for each category
function loadCategoryByCode(code)
	local cat = githubConfig.PRESET_CATEGORIES[code + 1]
	if not cat or not cat.returnFile then return end

	-- Lua file is safe to load with HttpGet
	local success, result = pcall(function()
		return game:HttpGet(githubConfig.GITHUB_RETURN_URL .. cat.returnFile)
	end)

	if success then
		local songList = loadstring(result)()
		globals.CachedGitHubSongs[cat.name] = {}

		for _, songName in ipairs(songList) do
			local songUrl = githubConfig.GITHUB_PRESET_URL .. urlEncode(cat.name) .. "/" .. urlEncode(songName)

			local httpRequest = getHttp()
			if not httpRequest then
				warn("❌ HTTP not supported on this executor.")
				return
			end

			local ok, response = pcall(function()
				return httpRequest({
					Url = songUrl,
					Method = "GET"
				})
			end)

			if ok and response and response.StatusCode == 200 and response.Body then
				local fileData = response.Body
				local bytes = {}
				for i = 1, #fileData do
					bytes[i] = string.byte(fileData, i)
				end

				local processor = MidiProcessor.new()
				local processed = processor:processMidiBytes(bytes)

				local formattedNotes = {}
				for _, note in ipairs(processed.notes) do
					table.insert(formattedNotes, { time = note[1], keys = note[2] })
				end

				globals.CachedGitHubSongs[cat.name][songName] = {
					notes = formattedNotes,
					bpm = processor.tempo or 120
				}

				-- UI update
				if songInfoLabel and currentCategory == cat.name then
					songInfoLabel.Text = "Loading " .. cat.name .. " Preset Songs..."
					refreshRightSongList()
				end
			else
				warn("❌ Failed to load")
			end
		end

		globals.categoriesLoaded += 1
		if globals.categoriesLoaded == 7 then
			globals.allsongspreloaded = true
			print("✅ All preset songs are fully preloaded.")
		end
	end
end

function preloadAllPresetSongs()
	for code = 1, 7 do
		task.spawn(function()
			loadCategoryByCode(code)
		end)
	end
end

local function fetchPresetList(category)
    if category == "All Songs" then
        -- Combine all category return files
        local allSongs = {}
        for _, cat in ipairs(githubConfig.PRESET_CATEGORIES) do
            if cat.returnFile then
                local success, result = pcall(function()
                    return game:HttpGet(githubConfig.GITHUB_RETURN_URL .. cat.returnFile)

                    local songs = loadstring(result)()
                    for _, song in ipairs(songs) do
                        table.insert(allSongs, {name = song, category = cat.name})
                    end
                end
            end
        end
        return allSongs
    else
        -- Fetch specific category return file
        local returnFile = nil
        for _, cat in ipairs(githubConfig.PRESET_CATEGORIES) do

                returnFile = cat.returnFile
                break
            end
        end

                return game:HttpGet(githubConfig.GITHUB_RETURN_URL .. returnFile)
            end)
            if success then
                local songs = loadstring(result)()

                for _, song in ipairs(songs) do
                    table.insert(songList, {name = song, category = category})
                end

            end
        end
    end
    return {}

local function loadPresetSong(songName, category)
    if globals.CachedGitHubSongs[category] and globals.CachedGitHubSongs[category][songName] then
        local songData = globals.CachedGitHubSongs[category][songName]
        return songData.notes, songData.bpm
    end

    -- fallback: if not in cache (shouldn’t happen if preload worked)
    local binPath = githubConfig.GITHUB_PRESET_URL .. category .. "/" .. songName
    local success, result = pcall(function()
        return game:HttpGet(binPath)
    end)

    if success then
        local bytes = {}
        for i = 1, #result do
            table.insert(bytes, string.byte(result, i))
        end

        local processor = MidiProcessor.new()
        local processed = processor:processMidiBytes(bytes)

        local formattedNotes = {}
        for _, note in ipairs(processed.notes) do
            table.insert(formattedNotes, { time = note[1], keys = note[2] })
        end

        return formattedNotes, processor.tempo or 120
    end

    return nil, nil
end

-- 🎵 Tempo Management Functions
local function extractTempoMarkers(notesArray)
    local markers = {}
    for _, note in ipairs(notesArray) do
        if note.keys:match("^tempo=(%d+)$") then
            local bpm = tonumber(note.keys:match("tempo=(%d+)"))
            if bpm then
                table.insert(markers, {time = note.time, bpm = bpm})
            end
        end
    end
    table.sort(markers, function(a, b) return a.time < b.time end)
    return markers
end

local function getCurrentTempoBPM(songTime)
    if not usetempo or #tempoMarkers == 0 then
        return manualBpm
    end
    
    -- Find the most recent tempo marker that's <= current song time
    local activeBpm = manualBpm -- Default fallback
    for _, marker in ipairs(tempoMarkers) do
        if songTime >= marker.time then
            activeBpm = marker.bpm
        else
            break -- Since markers are sorted by time
        end
    end
    
    return activeBpm
end

-- 🧠 Utility Function
local VirtualInputManager = game:GetService("VirtualInputManager")

local heldKeys = {} -- Keep track of held keys

local function safeKeyPress(keyCode)
    if keyCode and Enum.KeyCode then
        local success, result = pcall(function()
            VirtualInputManager:SendKeyEvent(true, keyCode, false, game)
        end)
        if not success then warn("❌ Failed to press key:", keyCode, result) end
    end
end

local function safeKeyRelease(keyCode)
    if keyCode and Enum.KeyCode then
        local success, result = pcall(function()
            VirtualInputManager:SendKeyEvent(false, keyCode, false, game)
        end)
        if not success then warn("❌ Failed to release key:", keyCode, result) end
    end
end

local function pressKey(char)
    local keyCode = AsciiKeys[char:lower()]
    local symbolCode = SymbolKeyCodes[char]
    if symbolCode then
        safeKeyPress(shiftKey)
        safeKeyPress(symbolCode)
        safeKeyRelease(shiftKey)
        heldKeys[char] = symbolCode
    elseif keyCode then
        if char ~= char:lower() then
            safeKeyPress(shiftKey)
            safeKeyPress(keyCode)
            safeKeyRelease(shiftKey)
        else
            safeKeyPress(keyCode)
        end
        heldKeys[char] = keyCode
    end
end

local function releaseKey(char)
    if heldKeys[char] then
        safeKeyRelease(heldKeys[char])
        heldKeys[char] = nil
    end
end

local function releaseAllKeys()
    for char, keyCode in pairs(heldKeys) do
        safeKeyRelease(keyCode)
    end
    heldKeys = {}
    safeKeyRelease(shiftKey)
end

local function parseNotes(txt)
    local t = {}
    for line in txt:gmatch("[^\r\n]+") do
        local time, keys = line:match('{%s*time%s*=%s*([%d%.]+)%s*,%s*keys%s*=%s*"([^"]-)"%s*}')
        if time and keys then
            table.insert(t, { time = tonumber(time), keys = keys })
        end
    end
    table.sort(t, function(a, b) return a.time < b.time end)
    if #t > 0 then
        local offset = t[1].time
        local adjusted = {}
        for _, v in ipairs(t) do
            table.insert(adjusted, { time = v.time - offset, keys = v.keys })
        end
        return adjusted
    end
    return t
end

-- 📊 Enhanced Duration Calculation with BPM Changes
local function calculateSongDuration(notesArray, initialBpm, useTempoMarkers)
    if #notesArray == 0 then return 0 end
    
    local totalRealTime = 0
    local currentBpm = initialBpm
    local lastBeatTime = 0
    
    -- Get all tempo markers first
    local tempoMarkersForCalc = {}
    if useTempoMarkers then
        for _, note in ipairs(notesArray) do
            if note.keys:match("^tempo=(%d+)$") then
                local bpm = tonumber(note.keys:match("tempo=(%d+)"))
                if bpm then
                    table.insert(tempoMarkersForCalc, {time = note.time, bpm = bpm})
                end
            end
        end
        table.sort(tempoMarkersForCalc, function(a, b) return a.time < b.time end)
    end
    
    -- Find the last note with actual content (not just tempo markers)
    local lastNoteTime = 0
    for _, note in ipairs(notesArray) do
        if not note.keys:match("^tempo=(%d+)$") then
            lastNoteTime = math.max(lastNoteTime, note.time)
        end
    end
    
    if lastNoteTime == 0 then return 0 end
    
    -- Calculate duration by processing each tempo segment
    local segmentStart = 0
    local segmentBpm = initialBpm
    
    if useTempoMarkers and #tempoMarkersForCalc > 0 then
        for _, marker in ipairs(tempoMarkersForCalc) do
            if marker.time <= lastNoteTime then
                -- Calculate time for this segment
                local segmentBeats = marker.time - segmentStart
                if segmentBeats > 0 then
                    local beatDuration = 60 / segmentBpm
                    totalRealTime = totalRealTime + (segmentBeats * beatDuration)
                end
                
                -- Move to next segment
                segmentStart = marker.time
                segmentBpm = marker.bpm
            end
        end
    end
    
    -- Calculate remaining time from last tempo change to end
    local remainingBeats = lastNoteTime - segmentStart
    if remainingBeats > 0 then
        local beatDuration = 60 / segmentBpm
        totalRealTime = totalRealTime + (remainingBeats * beatDuration)
    end
    
    return totalRealTime
end

local function formatTime(seconds)
    local minutes = math.floor(seconds / 60)
    local secs = math.floor(seconds % 60)
    return string.format("%d:%02d", minutes, secs)
end

local function calculateRealTimeFromBeatPosition(beatPosition, tempoMarkers, initialBpm, useTempoMarkers)
    if not useTempoMarkers or #tempoMarkers == 0 then
        return beatPosition * (60 / initialBpm)
    end

    local totalRealTime = 0
    local currentBpm = initialBpm
    local lastBeatTime = 0

for _, marker in ipairs(tempoMarkers) do
    if marker.time <= beatPosition then
        -- Calculate time for this segment
        local segmentBeats = marker.time - lastBeatTime
        if segmentBeats > 0 then
            local beatDuration = 60 / currentBpm
            totalRealTime = totalRealTime + (segmentBeats * beatDuration)
        end

        -- Move to next segment
        lastBeatTime = marker.time
        currentBpm = marker.bpm
    else
        break
    end
end


    -- Calculate remaining time from last tempo change to target position
    local remainingBeats = beatPosition - lastBeatTime
    if remainingBeats > 0 then
        local beatDuration = 60 / currentBpm
        totalRealTime = totalRealTime + (remainingBeats * beatDuration)
    end

    return totalRealTime
end

local function calculateBeatPositionFromRealTime(realTime, tempoMarkers, initialBpm, useTempoMarkers)
    if not useTempoMarkers or #tempoMarkers == 0 then
        return realTime / (60 / initialBpm)
    end

    local totalBeats = 0
    local currentBpm = initialBpm
    local lastBeatTime = 0
    local remainingRealTime = realTime

    for _, marker in ipairs(tempoMarkers) do
        -- Calculate time for this segment
        local segmentBeats = marker.time - lastBeatTime
        if segmentBeats > 0 then
            local beatDuration = 60 / currentBpm
            local segmentRealTime = segmentBeats * beatDuration
            
            if remainingRealTime <= segmentRealTime then
                -- Target time is within this segment
                return totalBeats + (remainingRealTime / beatDuration)
            end
            
            remainingRealTime = remainingRealTime - segmentRealTime
            totalBeats = totalBeats + segmentBeats
        end

        -- Move to next segment
        lastBeatTime = marker.time
        currentBpm = marker.bpm
    end

    -- Calculate remaining beats from last tempo change
    if remainingRealTime > 0 then
        local beatDuration = 60 / currentBpm
        totalBeats = totalBeats + (remainingRealTime / beatDuration)
    end

    return totalBeats
end

-- Update seekToTime function
local function seekToTime(targetTime)
    if globals.currentSongDuration == 0 then return end
    local now = tick()
    if now - globals.lastSeekTime < globals.seekCooldown then return end
    globals.lastSeekTime = now
    targetTime = math.max(0, math.min(targetTime, globals.currentSongDuration))
    globals.currentSongTime = targetTime
    songBeatPosition = calculateBeatPositionFromRealTime(targetTime, tempoMarkers, 
        usetempo and (#tempoMarkers > 0 and tempoMarkers[1].bpm or currentBpm) or currentBpm, usetempo)
    if usetempo then
        local expectedBpm = getCurrentTempoBPM(songBeatPosition)
        if expectedBpm ~= currentBpm then
            currentBpm = expectedBpm
            updateBPM(expectedBpm, false)
        end
    end
    releaseAllKeys()
    currentNoteIndex = 1
    for i = 1, #notes do
        if notes[i].time < songBeatPosition then
            currentNoteIndex = i + 1
        else
            break
        end
    end
    if currentNoteIndex > #notes then
        currentNoteIndex = #notes + 1
    end
    globals.seekRequested = true
    updateProgressDisplay()
end

-- Function to handle slider interaction
local function handleSliderInteraction(container, handle, highlight, progress, isFloat)
    local function updateSliderPosition(inputPosition)
        local containerPosition = container.AbsolutePosition
        local containerSize = container.AbsoluteSize
        local relativeX = inputPosition.X - containerPosition.X
        local percentage = math.max(0, math.min(1, relativeX / containerSize.X))
        
        -- Update handle position
        handle.Position = UDim2.new(percentage, -handle.Size.X.Offset / 2, 0.5, -handle.Size.Y.Offset / 2)
        
        -- Update progress bar
        progress.Size = UDim2.new(percentage, 0, 1, 0)
        
        -- Calculate target time and seek
        local targetTime = percentage * globals.currentSongDuration
        seekToTime(targetTime)
    end
    
    local function onInputBegan(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            if isFloat then
                globals.isFloatDragging = true
            else
                globals.isDragging = true
            end
            highlight.Visible = true
            updateSliderPosition(input.Position)
        end
    end
    
    local function onInputChanged(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            if (isFloat and globals.isFloatDragging) or (not isFloat and globals.isDragging) then
                updateSliderPosition(input.Position)
            end
        end
    end
    
    local function onInputEnded(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            if isFloat then
                globals.isFloatDragging = false
            else
                globals.isDragging = false
            end
            highlight.Visible = false
        end
    end
    
    -- Connect events to container for better mobile compatibility
    container.InputBegan:Connect(onInputBegan)
    container.InputChanged:Connect(onInputChanged)
    container.InputEnded:Connect(onInputEnded)
    
    -- Connect events to handle as well
    handle.InputBegan:Connect(onInputBegan)
    handle.InputChanged:Connect(onInputChanged)
    handle.InputEnded:Connect(onInputEnded)
    
    -- Handle mouse hover effects
    handle.MouseEnter:Connect(function()
        if not globals.isDragging and not globals.isFloatDragging then
            highlight.Visible = true
        end
    end)
    
    handle.MouseLeave:Connect(function()
        if not globals.isDragging and not globals.isFloatDragging then
            highlight.Visible = false
        end
    end)
end

-- ▶️ Enhanced Playback Function with Real-time BPM Support
local function playNotes(initialBpm)
    local lastUpdateTime = tick()
    local wasPaused = false
    local pausedTime = 0

    -- Extract tempo markers
    tempoMarkers = extractTempoMarkers(notes)

    -- Set initial BPM with nil safety check
    if usetempo then
        currentBpm = getCurrentTempoBPM(songBeatPosition or 0)
    else
        currentBpm = manualBpm
    end

    playing = true
    -- Only reset these if we're starting fresh (not seeking)
    if (globals.currentSongTime == 0 or globals.currentSongTime == nil) and (songBeatPosition == 0 or songBeatPosition == nil) then
        globals.currentSongTime = 0
        songBeatPosition = 0
        currentNoteIndex = 1
    end
    
    globals.currentSongDuration = calculateSongDuration(notes, currentBpm, usetempo)

    startProgressUpdater() -- Start progress updates
    updateBPM(currentBpm, false) -- Update BPM display
    updateProgressDisplay() -- Initial progress display

    while playing and currentNoteIndex <= #notes do
        if stopRequested then break end

        if paused then
            if not wasPaused then
                releaseAllKeys()
                wasPaused = true
                pausedTime = tick()
            end
            task.wait(0.05)
        else
            if wasPaused then
                local pauseDuration = tick() - pausedTime
                lastUpdateTime = tick()
                wasPaused = false
            end

            -- Check if seeking was requested and reset timing
            if globals.seekRequested then
                lastUpdateTime = tick()
                globals.seekRequested = false
            end

            local now = tick()
            local deltaTime = now - lastUpdateTime
            lastUpdateTime = now

            -- Update currentSongTime directly instead of converting from beat position
            globals.currentSongTime = globals.currentSongTime + deltaTime
            
            -- Calculate beat position from the current song time
            songBeatPosition = calculateBeatPositionFromRealTime(globals.currentSongTime, tempoMarkers, 
                usetempo and (#tempoMarkers > 0 and tempoMarkers[1].bpm or currentBpm) or currentBpm, usetempo)

            -- Update BPM if in tempo mode
            if usetempo then
                local expectedBpm = getCurrentTempoBPM(songBeatPosition or 0)
                if expectedBpm ~= currentBpm then
                    currentBpm = expectedBpm
                    updateBPM(expectedBpm, false)
                end
            else
                if currentBpm ~= manualBpm then
                    currentBpm = manualBpm
                    globals.currentSongDuration = calculateSongDuration(notes, currentBpm, usetempo)
                    updateBPM(currentBpm, false)
                end
            end

            -- Process notes at current beat position
            while currentNoteIndex <= #notes and songBeatPosition >= notes[currentNoteIndex].time do
                local keyString = notes[currentNoteIndex].keys
                if usetempo and keyString:match("^tempo=(%d+)$") then
                    local newBpm = tonumber(keyString:match("tempo=(%d+)"))
                    if newBpm and newBpm ~= currentBpm then
                        currentBpm = newBpm
                        updateBPM(newBpm, false)
                    end
                elseif keyString:sub(1, 1) == "~" then
                    -- Release keys
                    for i = 2, #keyString do
                        releaseKey(keyString:sub(i, i))
                    end
                else
                    -- Press keys
                    pressKey(keyString)
                end
                currentNoteIndex = currentNoteIndex + 1
            end

            task.wait(0.001)
        end

        if currentNoteIndex > #notes then
            if loopMode then
                stopRequested = false
                currentNoteIndex = 1
                songBeatPosition = 0
                lastUpdateTime = tick()
                if usetempo then
                    currentBpm = getCurrentTempoBPM(0)
                else
                    currentBpm = manualBpm
                end
                wasPaused = false
                globals.currentSongTime = 0
                releaseAllKeys()
                globals.currentSongDuration = calculateSongDuration(notes, currentBpm, usetempo)
                updateBPM(currentBpm, false)
                updateProgressDisplay()
                print("🔄 Looping song...")
                task.wait(1)
                lastUpdateTime = tick()
            else
                break
            end
        end
    end

    print("🎵 Song finished, releasing all keys")
    releaseAllKeys()
    playing, stopRequested, paused = false, false, false
    stopProgressUpdater()

    -- Reset progress display
    progressBar.Size = UDim2.new(0, 0, 1, 0)
    progressHandle.Position = UDim2.new(0, -progressHandle.Size.X.Offset / 2, 0.5, -progressHandle.Size.Y.Offset / 2)
    timeLabel.Text = "0:00 / " .. formatTime(globals.currentSongDuration)
    floatProgressBar.Size = UDim2.new(0, 0, 1, 0)
    floatProgressHandle.Position = UDim2.new(0, -floatProgressHandle.Size.X.Offset / 2, 0.5, -floatProgressHandle.Size.Y.Offset / 2)
    floatTimeLabel.Text = "0:00 / " .. formatTime(globals.currentSongDuration)
    tempoMarkers = {}  -- Clear tempo markers
    globals.currentSongTime = 0
    currentNoteIndex = 1
end

local function populateCategoryFrame(categoryName, songs)
    if not globals.categoryFrames[categoryName] then
        globals.categoryFrames[categoryName] = {}
        globals.categoryFrames[categoryName].frame, globals.categoryFrames[categoryName].layout = createCategoryFrame(categoryName)
    end

    local frame = globals.categoryFrames[categoryName].frame
    
    -- Clear existing buttons
    for _, child in ipairs(frame:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end

    -- Sort songs alphabetically
    local sortedNames = {}
    for name in pairs(songs) do
        table.insert(sortedNames, name)
    end
    table.sort(sortedNames, function(a, b)
        return a:lower() < b:lower()
    end)

    -- Create song buttons
    for i, songName in ipairs(sortedNames) do
        createSongButton(frame, songName, songs[songName], i)
    end

    -- Store song data for this category
    globals.categoryData[categoryName] = songs
    
    return #sortedNames
end

local function loadPresetCategory(categoryName)
    if categoryName == "All Songs" then
        local allSongs = {}
        for _, cat in ipairs(githubConfig.PRESET_CATEGORIES) do
            if cat.returnFile then
                local success, result = pcall(function()
                    return game:HttpGet(githubConfig.GITHUB_RETURN_URL .. cat.returnFile)
                end)
                if success then
                    local songs = loadstring(result)()
                    for _, song in ipairs(songs) do
                        allSongs[song] = {
                            url = "preset",
                            category = cat.name,
                            originalName = song
                        }
                    end
                end
            end
        end
        
        local songCount = populateCategoryFrame("Preset: " .. categoryName, allSongs)
        switchToCategory("Preset: " .. categoryName)
        return songCount > 0
    else
        local returnFile = nil
        for _, cat in ipairs(githubConfig.PRESET_CATEGORIES) do
            if cat.name == categoryName then
                returnFile = cat.returnFile
                break
            end
        end
        
        if returnFile then
            local success, result = pcall(function()
                return game:HttpGet(githubConfig.GITHUB_RETURN_URL .. returnFile)
            end)
            if success then
                local songList = loadstring(result)()
                local songs = {}
                for _, song in ipairs(songList) do
                    songs[song] = {
                        url = "preset",
                        category = categoryName,
                        originalName = song
                    }
                end
                
                local songCount = populateCategoryFrame("Preset: " .. categoryName, songs)
                switchToCategory("Preset: " .. categoryName)
                return songCount > 0
            end
        end
    end
    return false
end

-- Enhanced loadSongs function to handle dynamic categories
local function loadSongsFromCategory(category)
    -- Check if category is already loaded
    if globals.categoryData[category] then
        print("✅ Using cached data for category: " .. category)
        switchToCategory(category)
        return true
    end

    local songs, midiFiles, totalFiles = {}, {}, 0
    
    if category == "All Songs" then
        local allFiles = getAllMidiFiles()
        for _, fileData in ipairs(allFiles) do
            table.insert(midiFiles, fileData.filePath)
        end
        totalFiles = #allFiles
        print("🌐 Loading all songs from " .. #getFolderCategories() .. " categories...")
    else
        midiFiles = getMidiFilesFromCategory(category)
        totalFiles = #midiFiles
    end

    songInfoLabel.Text = "Loading " .. category .. "... (0/" .. totalFiles .. ")"
    
    for i, filePath in ipairs(midiFiles) do
        local fileName = string.match(filePath, "([^/\\]+)%.mid%.smf$")
          or string.match(filePath, "([^/\\]+)%.smf$")
          or string.match(filePath, "([^/\\]+)%.mid$")
          or string.match(filePath, "([^/\\]+)%.midi$")

        if fileName then
            local displayName = fileName
            if category == "All Songs" then
                local fileCategory = string.match(filePath, "ChordRushMidi[/\\]([^/\\]+)[/\\]")
                if fileCategory then
                    displayName = fileName
                end
            end
            
            local bytes = readMidiFile(filePath)
            if bytes then
                local processor = MidiProcessor.new()
                local result = processor:processMidiBytes(bytes)
                
                local formattedNotes = {}
                for _, note in ipairs(result.notes) do
                    table.insert(formattedNotes, { time = note[1], keys = note[2] })
                end
                
                globals.MidiSongOutput[displayName] = formattedNotes
                
                songs[displayName] = {
                    url = "local",
                    bpm = processor.tempo or 120,
                    notes = formattedNotes,
                    originalName = fileName,
                    category = category == "All Songs" and string.match(filePath, "ChordRushMidi[/\\]([^/\\]+)[/\\]") or category
                }
                
                if i % 5 == 0 or i == totalFiles then
                    songInfoLabel.Text = "Loading " .. category .. "... (" .. i .. "/" .. totalFiles .. ")"
                    task.wait()
                end
            end
        end
    end
    
    local songCount = populateCategoryFrame(category, songs)
    switchToCategory(category)
    
    print("✅ Loaded " .. totalFiles .. " MIDI files from " .. category .. (category == "All Songs" and " (all categories)" or " category"))
    return true
end

local function loadSongs(category)
    return loadSongsFromCategory(category)

-- 🔁 Refresh songs in the list
function refreshRightSongList()
    -- Clear existing buttons
    for _, child in ipairs(rightSongList:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end

    -- Ensure songsTbl and filteredSongs are defined
    globals.filteredSongs = filteredSongs or {}
    songsTbl = songsTbl or {}

    -- Choose the song list to show
    local songsToShow = (next(globals.filteredSongs) ~= nil and filteredSongs) or songsTbl

    local sortedNames = {}
    for name in pairs(songsToShow) do
        table.insert(sortedNames, name)
    end
    table.sort(sortedNames, function(a, b)
        return a:lower() < b:lower()
    end)
    
    local count = 0
    for _, name in ipairs(sortedNames) do
        local data = songsToShow[name]

        count += 1

        local btn = Instance.new("TextButton", rightSongList)
        btn.Size = UDim2.new(1, -10, 0, 40)
        btn.Position = UDim2.new(0, 6, 0, (count-1) * 44)
        btn.Text = name
        btn.TextSize = 14
        btn.Font = Enum.Font.Gotham
        btn.BackgroundColor3 = colors.accent
        btn.TextColor3 = colors.text
        btn.BorderSizePixel = 0
        btn.ClipsDescendants = true
        
        local btnCorner = Instance.new("UICorner", btn)
        btnCorner.CornerRadius = UDim.new(0, 8)
        
        -- Add ripple effect
        local ripple = Instance.new("Frame", btn)
        ripple.BackgroundColor3 = Color3.new(1, 1, 1)
        ripple.BackgroundTransparency = 0.8
        ripple.BorderSizePixel = 0
        
        -- Double-click variables
        local lastClickTime = 0
        local doubleClickThreshold = 0.5 -- 500ms for double-click detection
        
        -- Add hover effect
        btn.MouseEnter:Connect(function()
            game:GetService("TweenService"):Create(btn, TweenInfo.new(0.2), {
                BackgroundColor3 = Color3.fromRGB(
                    math.min(colors.accent.R + 0.1, 1),
                    math.min(colors.accent.G + 0.1, 1),
                    math.min(colors.accent.B + 0.1, 1)
                )
            }):Play()
        end)
        
        btn.MouseLeave:Connect(function()
            game:GetService("TweenService"):Create(btn, TweenInfo.new(0.2), {
                BackgroundColor3 = colors.accent
            }):Play()
        end)
        
        btn.MouseButton1Down:Connect(function(x, y)
            -- Create ripple effect
            ripple.Position = UDim2.new(0, x - btn.AbsolutePosition.X, 0, y - btn.AbsolutePosition.Y)
            ripple.Size = UDim2.new(0, 0, 0, 0)
            
            local goal = {}
            goal.Size = UDim2.new(2, 0, 2, 0)
            goal.Position = UDim2.new(-0.5, x - btn.AbsolutePosition.X, -0.5, y - btn.AbsolutePosition.Y)
            
            local tween = game:GetService("TweenService"):Create(ripple, TweenInfo.new(0.5), goal)
            tween:Play()
            tween.Completed:Connect(function()
                ripple.Size = UDim2.new(0, 0, 0, 0)
            end)
        end)

        -- Double-click to select song
        btn.MouseButton1Click:Connect(function()
            local currentTime = tick()
            local timeSinceLastClick = currentTime - lastClickTime
            
            if timeSinceLastClick <= doubleClickThreshold and timeSinceLastClick > 0.05 then
                -- Double-click detected - select the song
                stopRequested, playing, paused = false, false, false
                releaseAllKeys()
                globals.currentSongTime = 0
                songBeatPosition = 0
                currentNoteIndex = 1
                releaseAllKeys()
                updateProgressDisplay() -- Update display when stopping
                songName = name
                title.Text = "🎵 " .. name
                
                local songData = data
                if songData then
                    if songData.url == "preset" then
                        -- Load preset song from GitHub
                        local notes, tempo = loadPresetSong(songData.originalName, songData.category)
                        if notes then
                            songData.notes = notes
                            songData.bpm = tempo
                            print("✅ Loaded preset song: " .. name .. " (" .. #notes .. " notes)")
                        else
                            warn("❌ Failed to load preset song")
                            return
                        end
                    end
                    
                    if songData.notes then
                        notes = songData.notes
                        bpmInput.Text = tostring(songData.bpm or 120)
                        
                        -- Enhanced info display for All Songs category and presets
                        local categoryInfo = currentCategory
                        if currentCategory:find("Preset:") then
                            categoryInfo = currentCategory
                        elseif currentCategory == "All Songs" and songData.category then
                            categoryInfo = "All Songs (from " .. songData.category .. ")"
                        end
                        
                        -- Update tempo markers for the new song
                        tempoMarkers = extractTempoMarkers(notes)
                        if #tempoMarkers > 0 then
                            print("🎵 Found " .. #tempoMarkers .. " tempo markers in song")
                        end
                        
                        -- Update song info display
                        songInfoLabel.Text = "Now in: " .. categoryInfo .. " • Total Songs: " .. count
                        
                        -- Reset progress display
                        globals.currentSongDuration = calculateSongDuration(notes, tonumber(bpmInput.Text) or 120, usetempo)
                        progressBar.Size = UDim2.new(0, 0, 1, 0)
                        timeLabel.Text = "0:00 / " .. formatTime(globals.currentSongDuration)
                        
                        floatProgressBar.Size = UDim2.new(0, 0, 1, 0)
                        floatTimeLabel.Text = "0:00 / " .. formatTime(globals.currentSongDuration)
                        
                        print("✅ Loaded song: " .. name .. " (" .. #notes .. " notes)")
                    else
                        warn("⚠️ Song data missing for " .. name)
                    end
                end
                
                -- Reset click timer after successful double-click
                lastClickTime = 0
            else
                -- First click or too much time passed - just update the timer
                lastClickTime = currentTime
                -- Optional: Provide visual feedback for first click
                print("Single click detected on: " .. name .. " (double-click to select)")
            end
        end)

        -- Add category indicator for preset songs
        if data.url == "preset" then
            local categoryIndicator = Instance.new("TextLabel", btn)
            categoryIndicator.Size = UDim2.new(1, -10, 0, 16)
            categoryIndicator.Position = UDim2.new(0, 4, 1, -15)
            categoryIndicator.BackgroundTransparency = 1
            categoryIndicator.TextColor3 = colors.textSecondary
            categoryIndicator.TextSize = 10
            categoryIndicator.Font = Enum.Font.Gotham
            categoryIndicator.Text = "🌟 " .. data.category
            categoryIndicator.TextXAlignment = Enum.TextXAlignment.Left
        end
    end

    -- Update song info label
    local categoryInfo = currentCategory
    if currentCategory == "All Songs" then
        local totalCategories = #getFolderCategories()
        categoryInfo = "All Songs (" .. totalCategories .. " categories)"
    elseif currentCategory:find("Preset:") then
        categoryInfo = currentCategory
    end
    
    songInfoLabel.Text = "Now in: " .. categoryInfo .. " • Total Songs: " .. count

    -- Update canvas size
    if rightSongLayout then
        rightSongList.CanvasSize = UDim2.new(0, 0, 0, count * 44 + 10)
    end
end

-- 🔍 Search function
local searchFrame, searchLayout = createSearchResultsFrame()

local function performSearch(query)
    globals.searchResults = {}
    query = (query or ""):lower()

    -- If search box is cleared
    if query == "" then
        globals.isSearchActive = false
        searchFrame.Visible = false

        if globals.currentCategoryFrame then
            globals.currentCategoryFrame.Visible = true
        end

        -- Restore song info label
        local categoryInfo = currentCategory or "Custom"

        if categoryInfo == "All Songs" then
            local totalCategories = #getFolderCategories()
            categoryInfo = "All Songs (" .. totalCategories .. " categories)"
        end

        local songsInCategory = globals.categoryData[currentCategory] or {}
        local count = 0
        for _ in pairs(songsInCategory) do
            count += 1
        end

        songInfoLabel.Text = "Now in: " .. categoryInfo .. " • Total Songs: " .. count
        return
    end

    -- Search across all loaded categories
    for categoryName, songs in pairs(globals.categoryData) do
        for songName, songData in pairs(songs) do
            if songName:lower():find(query, 1, true) then
                globals.searchResults[songName] = songData
            end
        end
    end

    -- Clear previous search results
    for _, child in ipairs(searchFrame:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end

    -- Sort and display new results
    local sortedResults = {}
    for name in pairs(globals.searchResults) do
        table.insert(sortedResults, name)
    end
    table.sort(sortedResults, function(a, b)
        return a:lower() < b:lower()
    end)

    for i, songName in ipairs(sortedResults) do
        createSongButton(searchFrame, songName, globals.searchResults[songName], i)
    end

    -- Update UI for search
    globals.isSearchActive = true
    if globals.currentCategoryFrame then
        globals.currentCategoryFrame.Visible = false
    end
    searchFrame.Visible = true
    songInfoLabel.Text = "Search Results: " .. #sortedResults .. " songs found for \"" .. query .. "\""
end

-- Bind search to input box
searchInput:GetPropertyChangedSignal("Text"):Connect(function()
    performSearch(searchInput.Text)
end)

local c, t, p = 0, 0, Vector2.new()

local function h(s)
    if typeof(s) == "Vector3" then
        s = Vector2.new(s.X, s.Y)
    end

    local k = tick()
    if k - t <= 0.5 and (s - p).Magnitude <= 50 then 
        c = c + 1 
    else 
        c = 1 
    end
    
    t, p = k, s
    
    if c >= 3 and recordingMode and not frame.Visible then 
        floatFrame.Visible = true
        c = 0 
    end 
end

local u, l = game:GetService("UserInputService"), game:GetService("Players").LocalPlayer

pcall(function()
    u.InputBegan:Connect(function(i)
        local s
        if i.UserInputType == Enum.UserInputType.MouseButton1 then 
            s = u:GetMouseLocation()
        elseif i.UserInputType == Enum.UserInputType.Touch then 
            s = i.Position 
        end
        
        if s then 
            h(s) 
        end 
    end)
end)

if l then 
    local m = l:GetMouse()
    if m then 
        pcall(function()
            m.Button1Down:Connect(function()
                h(Vector2.new(m.X, m.Y))
            end)
        end)
    end 
end
