                        dragInput = input
                end

        end)

        local function animateButtonFlight(oldButton, newButton)
                local startIndex = table.find(buttonOrder, oldButton)
                local endIndex = table.find(buttonOrder, newButton)
                if not startIndex or not endIndex then return end

                local step = (startIndex < endIndex) and 1 or -1
                local totalSteps = math.abs(endIndex - startIndex) + 1
                local perButtonTime = uiPageTweenTime / totalSteps

                -- 1) ▄berflug: nacheinander hell machen
                for i = startIndex, endIndex, step do

                        local btn = buttons[btnName]

                for i = startIndex, endIndex, step do

                                task.wait(perButtonTime * 0.1) -- Verz÷gerung zwischen den Buttons
                                btn.BackgroundColor3 = defaultColor
                        end
                end
        end

        -- Page-Switch-Funktion
        local function switchPage(newActive)
                local page = pagesList[newActive]
                if not page then return end

                pages:JumpTo(page)
                animateButtonFlight(currentActive, newActive)
                currentActive = newActive
        end

        -- Buttons verbinden
        for name, btn in pairs(buttons) do
                btn.MouseButton1Click:Connect(function()
                        switchPage(name)
                end)
        end

        local hoverTweenInfo = TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
        local clickTweenInfo = TweenInfo.new(0.1, Enum.EasingStyle.Back, Enum.EasingDirection.Out)

        local hoverScale = 1.05 -- leichte Vergr÷▀erung beim Hover
        local clickScale = 0.9  -- Verkleinerung beim Klick

        -- Hilfsfunktion um UDim2-Skalierung zu berechnen und Position anzupassen
        local function scaleUDim2Centered(size, position, scale)
                local newSize = UDim2.new(size.X.Scale * scale, size.X.Offset * scale, size.Y.Scale * scale, size.Y.Offset * scale)
                local deltaX = (newSize.X.Offset - size.X.Offset) / 2
                local deltaY = (newSize.Y.Offset - size.Y.Offset) / 2
                local newPos = UDim2.new(position.X.Scale, position.X.Offset - deltaX, position.Y.Scale, position.Y.Offset - deltaY)
                return newSize, newPos
        end

        -- Alle Pages durchgehen
        for _, page in pairs(main:GetChildren()) do
                if page:IsA("Frame") then
                        for i = 1, 10 do
                                local btn = page:FindFirstChild(tostring(i))
                                if btn and btn:IsA("ImageButton") then
                                        local originalSize = btn.Size
                                        local originalPos = btn.Position
                                        local isHovered = false

                                        -- Hover Effekte
                                        btn.MouseEnter:Connect(function()
                                                isHovered = true
                                                local newSize, newPos = scaleUDim2Centered(originalSize, originalPos, hoverScale)
                                                TweenService:Create(btn, hoverTweenInfo, {Size = newSize, Position = newPos}):Play()
                                        end)
                                        btn.MouseLeave:Connect(function()
                                                isHovered = false
                                                TweenService:Create(btn, hoverTweenInfo, {Size = originalSize, Position = originalPos}):Play()
                                        end)

                                        -- Click Effekt
                                        btn.MouseButton1Click:Connect(function()
                                                local clickSize, clickPos = scaleUDim2Centered(originalSize, originalPos, clickScale)
                                                local tween = TweenService:Create(btn, clickTweenInfo, {Size = clickSize, Position = clickPos})
                                                tween:Play()
                                                tween.Completed:Wait()
                                                -- nach Click zur³ck zur Hover- oder Original-Size
                                                local targetScale = isHovered and hoverScale or 1
                                                local targetSize, targetPos = scaleUDim2Centered(originalSize, originalPos, targetScale)
                                                TweenService:Create(btn, hoverTweenInfo, {Size = targetSize, Position = targetPos}):Play()
                                        end)
                                end
                        end
                end
        end

end
