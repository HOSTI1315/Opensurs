-- Kapao Hub Loader System
local KapaoLoader = {}

function KapaoLoader:showNotification(title, text, duration)
    game.StarterGui:SetCore("SendNotification", {
        Title = title,
        Text = text,
        Duration = duration or 5
    })
end

function KapaoLoader:getGameInfo()
    local universeId = game.GameId
    local placeId = game.PlaceId
    local gameName = "Unknown Game"
    
    -- ลองดึงชื่อเกม
    local success, productInfo = pcall(function()
        return game:GetService("MarketplaceService"):GetProductInfo(placeId)
    end)
    
    if success and productInfo then
        gameName = productInfo.Name
    end
    
    return {
        universeId = universeId,
        placeId = placeId,
        name = gameName
    }
end

function KapaoLoader:findSupportedGame()
    local gameInfo = self:getGameInfo()
    local gameData = UniverseDatabase[gameInfo.universeId]
    
    if gameData then
        return gameData, gameInfo
    end
    
    return nil, gameInfo
end

function KapaoLoader:loadScript(scriptUrl)
    local success, error = pcall(function()
        loadstring(game:HttpGet(scriptUrl))()
    end)
    
    if not success then
        self:showNotification(
            "❌ Kapao Hub Error",
            "Failed to load script:\n" .. tostring(error),
            8
        )
        return false
    end
    
    return true
end

function KapaoLoader:showSupportedGames()
    local gamesList = "🎮 Supported Games:\n\n"
    local count = 0
    
    for universeId, gameData in pairs(UniverseDatabase) do
        count = count + 1
        gamesList = gamesList .. "• " .. gameData.name .. "\n"
    end
    
    gamesList = gamesList .. "\nTotal: " .. count .. " games"
    
    self:showNotification(
        "☕ Kapao Hub",
        gamesList,
        12
    )
end

function KapaoLoader:reportUnsupportedGame(gameInfo)
    print("=== UNSUPPORTED GAME REPORT ===")
    print("Game Name:", gameInfo.name)
    print("UniverseId:", gameInfo.universeId)
    print("PlaceId:", gameInfo.placeId)
    print("Timestamp:", os.date("%Y-%m-%d %H:%M:%S"))
    print("===============================")
    
    -- แสดงข้อมูลให้ผู้ใช้เห็น
    self:showNotification(
        "❌ Game Not Supported",
        "Game: " .. gameInfo.name .. "\nUniverseId: " .. gameInfo.universeId .. "\nPlaceId: " .. gameInfo.placeId,
        10
    )
end

function KapaoLoader:init()
    -- แสดง loading notification
    self:showNotification(
        "☕ Kapao Hub",
        "Initializing Universal Loader...",
        3
    )
    
    wait(1)
    
    -- หาเกมที่รองรับ
    local gameData, gameInfo = self:findSupportedGame()
    
    if gameData then
        -- เจอเกมที่รองรับ
        self:showNotification(
            "🎮 Kapao Hub",
            "Loading " .. gameData.name,
            3
        )
        
        wait(2)
        
        -- โหลด script
        local loadSuccess = self:loadScript(gameData.script)
        
        if loadSuccess then
            self:showNotification(
                "✅ Kapao Hub",
                gameData.name .. " loaded successfully!",
                4
            )
        end
    else
        -- ไม่เจอเกมที่รองรับ
        self:reportUnsupportedGame(gameInfo)
        wait(2)
        self:showSupportedGames()
    end
end
