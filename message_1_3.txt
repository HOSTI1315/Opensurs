local OriginalLighting = {}
local SavedAtmospheres = {}
local SavedBlooms = {}

local function SaveOriginalLighting()
    OriginalLighting.ClockTime = Lighting.ClockTime
    OriginalLighting.Ambient = Lighting.Ambient
    OriginalLighting.Brightness = Lighting.Brightness
    OriginalLighting.ColorShift_Bottom = Lighting.ColorShift_Bottom
    OriginalLighting.ColorShift_Top = Lighting.ColorShift_Top
    OriginalLighting.EnvironmentDiffuseScale = Lighting.EnvironmentDiffuseScale
    OriginalLighting.EnvironmentSpecularScale = Lighting.EnvironmentSpecularScale
    OriginalLighting.GlobalShadows = Lighting.GlobalShadows
    OriginalLighting.OutdoorAmbient = Lighting.OutdoorAmbient
    OriginalLighting.ShadowSoftness = Lighting.ShadowSoftness
    OriginalLighting.ExposureCompensation = Lighting.ExposureCompensation
    
    SavedAtmospheres = {}
    SavedBlooms = {}
    for _, child in ipairs(Lighting:GetChildren()) do
        if child:IsA('Atmosphere') then
            local clone = child:Clone()
            clone.Parent = nil
            table.insert(SavedAtmospheres, {original = child, clone = clone})
        elseif child:IsA('BloomEffect') then
            local clone = child:Clone()
            clone.Parent = nil
            table.insert(SavedBlooms, {original = child, clone = clone})
        end
    end
end

local function RestoreLighting()
    pcall(function()
        local akihaAtmos = Lighting:FindFirstChild("AkihaAtmosphere")
        local akihaBloom = Lighting:FindFirstChild("AkihaBloom")
        if akihaAtmos then akihaAtmos:Destroy() end
        if akihaBloom then akihaBloom:Destroy() end
    end)
    
    for prop, val in pairs(OriginalLighting) do
        pcall(function() Lighting[prop] = val end)
    end
    
    for _, child in ipairs(Lighting:GetChildren()) do
        if child:IsA('Atmosphere') or child:IsA('BloomEffect') then
            child.Enabled = true
        end
    end
end

local JumpDebounce = false
local IsGliding = false
local CurrentBodyGyro = nil
local JumpConnection = nil
local OriginalGravity = workspace.Gravity

local function CleanupJumpSystem()
    if CurrentBodyGyro then
        pcall(function() CurrentBodyGyro:Destroy() end)
        CurrentBodyGyro = nil
    end
    IsGliding = false
    JumpDebounce = false
end

local function SetupCustomJumpSystem(char)
    if not CustomJumpEnabled then return end
    if not char then return end
    
    local hum = char:FindFirstChildOfClass("Humanoid")
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hum or not hrp then return end
    
    OriginalGravity = workspace.Gravity
    workspace.Gravity = 80
    hum.JumpPower = 25
    hum.JumpHeight = 7.2
    
    if JumpConnection then
        pcall(function() JumpConnection:Disconnect() end)
    end
    
    JumpConnection = UserInputService.JumpRequest:Connect(function()
        if not CustomJumpEnabled then return end
        if JumpDebounce then return end
        
        JumpDebounce = true
        IsGliding = true
        
        local bodyGyro = Instance.new('BodyGyro', hrp)
        bodyGyro.Name = 'AkihaJumpGyro'
        bodyGyro.MaxTorque = Vector3.new(1, 1, 1) * 10^15
        bodyGyro.D = 4500
        bodyGyro.P = 15500
        CurrentBodyGyro = bodyGyro
        
        hum:ChangeState(Enum.HumanoidStateType.Flying)
        bodyGyro.CFrame = hrp.CFrame * CFrame.Angles(math.rad(-45), 0, 0)
        
        local wasAutoRotate = hum.AutoRotate
        hum.AutoRotate = false
        
        local Anim = Instance.new("Animation")
        Anim.AnimationId = "rbxassetid://121032789756540"
        local animTrack
        pcall(function()
            animTrack = hum:LoadAnimation(Anim)
            animTrack:Play(0.3)
            animTrack.Priority = Enum.AnimationPriority.Action
            animTrack:AdjustSpeed(1)
            animTrack.TimePosition = 5
        end)
        
        task.spawn(function()
            local startTime = tick()
            while CurrentBodyGyro and CurrentBodyGyro.Parent == hrp and (tick() - startTime) < 8 do
                bodyGyro.D = 1500 - math.sin((tick() - startTime) * 10) * 1500
                task.wait(0.01)
            end
        end)
        
        task.delay(0.2, function()
            JumpDebounce = false
            if animTrack then
                animTrack:AdjustSpeed(0.2)
            end
            
            repeat task.wait() until not char or not char.Parent or hum.FloorMaterial ~= Enum.Material.Air or not CustomJumpEnabled
            
            CleanupJumpSystem()
            hum.AutoRotate = wasAutoRotate
            
            if animTrack then
                animTrack:Stop()
            end
            
            local landAnim = Instance.new("Animation")
            landAnim.AnimationId = "rbxassetid://18462892217"
            pcall(function()
                local landTrack = hum:LoadAnimation(landAnim)
                landTrack:Play(0.1)
                landTrack.Priority = Enum.AnimationPriority.Action
                landTrack:AdjustSpeed(1.4)
                task.wait(0.4)
                landTrack:Stop(0.3)
            end)
        end)
    end)
    
    return JumpConnection
end

local function DisableCustomJump()
    if JumpConnection then
        pcall(function() JumpConnection:Disconnect() end)
        JumpConnection = nil
    end
    CleanupJumpSystem()
    workspace.Gravity = OriginalGravity or 196.2
end

local function FindNearestEnemy()
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end
    
    local nearestEnemy = nil
    local nearestDist = math.huge
    
    for _, part in ipairs(workspace.Live:GetChildren()) do
        if part:IsA('Model') and part ~= char and part:FindFirstChild('HumanoidRootPart') then
            local enemyHum = part:FindFirstChildOfClass("Humanoid")
            if enemyHum and enemyHum.Health > 0 then
                local dist = (part.HumanoidRootPart.Position - hrp.Position).Magnitude
                if dist < nearestDist and dist < 20 then
                    nearestDist = dist
                    nearestEnemy = part
                end
            end
        end
    end
    
    return nearestEnemy
end

local function DoHitEffect(doLightning, dir, amount)
    DoCustomShake()
    
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    
    local target = EnemyTarget or FindNearestEnemy()
    
    if doLightning and target and target:FindFirstChild("HumanoidRootPart") then
        CreateLightningEffect(hrp.Position, target.HumanoidRootPart.Position + Vector3.new(0, dir or 15, 0))
    end
    
    local p = Instance.new('Part')

        task.spawn(function()
            for i = 1, 55 do
                if not target or not target:FindFirstChild("HumanoidRootPart") then break end
                p.CFrame = target.HumanoidRootPart.CFrame * CFrame.new(0, 0, -2 + i / 3)

    game:GetService("Debris"):AddItem(p, 1)

RunService.Heartbeat:Connect(function()
    if not AutoBlockEnabled then
        if AutoBlockIsBlocking then
            StopAutoBlock()
            AutoBlockIsBlocking = false
        end
        return
    end
    
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    
    local myPosition = char.HumanoidRootPart.Position
    local closeRangeTarget = nil
    local closestCloseRange = math.huge
    local longRangeTarget = nil
    local closestLongRange = math.huge
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local targetChar = player.Character
            if targetChar and targetChar:FindFirstChild("HumanoidRootPart") then
                local distance = (myPosition - targetChar.HumanoidRootPart.Position).Magnitude
                
                if distance < AutoBlockCloseRange and distance < closestCloseRange then
                    closestCloseRange = distance
                    closeRangeTarget = player
                end
                
                if distance < AutoBlockLongRange and distance < closestLongRange then
                    closestLongRange = distance
                    longRangeTarget = player
                end
            end
        end
    end
    
    local shouldBlock = false
    
    if longRangeTarget then
        local targetHum = longRangeTarget.Character:FindFirstChildOfClass("Humanoid")
        if targetHum and IsPlayingEnemyAnimation(targetHum, LongRangeAnimations) then
            shouldBlock = true
        end
    end
    
    if not shouldBlock and closeRangeTarget then
        local targetHum = closeRangeTarget.Character:FindFirstChildOfClass("Humanoid")
        if targetHum and IsPlayingEnemyAnimation(targetHum, CloseRangeAnimations) then
            shouldBlock = true
        end
    end
    
    if shouldBlock then
        if not AutoBlockIsBlocking then
            AutoBlockIsBlocking = true
            StartAutoBlock()
        end
    elseif AutoBlockIsBlocking then
        AutoBlockIsBlocking = false
        StopAutoBlock()
        if M1AfterBlockEnabled then
            StartM1Attack()
            task.delay(0.11, StopM1Attack)
        end
    end
end)

local EnemyTarget = nil

local function SetupHitDetection(char)
    if not char then return end
    
    local connection
    connection = char:GetAttributeChangedSignal('LastDamageDone'):Connect(function()
        if not AkihaEnabled then return end
        
        LastHitTime = os.clock()
        
        for _, part in ipairs(workspace.Live:GetChildren()) do
            if part:IsA('Model') and part:FindFirstChild('HumanoidRootPart') then
                pcall(function()
                    local You = string.sub(tostring(char:GetAttribute('LastDamageDone') or ""), 1, 13)
                    local Them = string.sub(tostring(part:GetAttribute('LastDamage') or ""), 1, 13)
                    
                    if You == Them then
                        EnemyTarget = part
                        local enemyHum = part:FindFirstChildOfClass("Humanoid")
                        if enemyHum then
                            local newHealth = enemyHum.Health
                            
                            if EnemyHealth and newHealth < EnemyHealth then
                                local damageDone = EnemyHealth - newHealth
                                ComboDamage = ComboDamage + damageDone
                                ComboHits = ComboHits + 1
                                UpdateComboCounter(ComboHits, ComboDamage)
                                
                                DoCustomShake()
                                
                                local enemyHrp = part:FindFirstChild("HumanoidRootPart")
                                local myHrp = char:FindFirstChild("HumanoidRootPart")
                                if enemyHrp and myHrp then
                                    CreateBloodParticle(enemyHrp)
                                    CreateLightningEffect(myHrp.Position, enemyHrp.Position)
                                end
                            end
                            
                            if newHealth <= 0 and not BloodHeatActive then
                                ActivateBloodHeat()
                                
                                if AutoRoastEnabled then
                                    local roastMsg = RoastMessages[math.random(1, #RoastMessages)]
                                    pcall(function()
                                        game:GetService("ReplicatedStorage"):WaitForChild("DefaultChatSystemChatEvents"):WaitForChild("SayMessageRequest"):FireServer(roastMsg, "All")
                                    end)
                                end
                            end
                            
                            EnemyHealth = newHealth
                        end
                    end
                end)
            end
        end
        
        for _, part in ipairs(workspace.Live:GetChildren()) do
            if part:IsA('Model') and part:FindFirstChild('HumanoidRootPart') then
                pcall(function()
                    local You = string.sub(tostring(char:GetAttribute('LastDamageDone') or ""), 1, 13)
                    local Them = string.sub(tostring(part:GetAttribute('LastDamage') or ""), 1, 13)
                    if You == Them then
                        local enemyHum = part:FindFirstChildOfClass("Humanoid")
                        if enemyHum then
                            OldEnemyHealth = enemyHum.Health
                            EnemyHealth = enemyHum.Health
                        end
                    end
                end)
            end
        end
    end)
    
    return connection
end

local function SetupAkihaFeatures(char)

    local hrp = char:WaitForChild("HumanoidRootPart")
    
    table.insert(AkihaConnections, SetupHitDetection(char))
    
    if CustomJumpEnabled then

        if jumpConn then
            table.insert(AkihaConnections, jumpConn)
        end
        
        table.insert(AkihaConnections, hum:GetPropertyChangedSignal('JumpPower'):Connect(function()
            if CustomJumpEnabled and hum.JumpPower ~= 25 then
                hum.JumpPower = 25
            end
        end))
        
        table.insert(AkihaConnections, hum:GetPropertyChangedSignal('PlatformStand'):Connect(function()
            if hum.PlatformStand == true then
                hum.PlatformStand = false
            end
        end))
    end
    
    table.insert(AkihaConnections, hum.AnimationPlayed:Connect(function(animTrack)
        if not CustomAnimationsEnabled then return end
        local anim = animTrack.Animation
        if not anim then return end
        
        if M1AnimIds[anim.AnimationId] then
            HandleM1Animation(animTrack, char)
        end
        
        for _, moveAnimId in pairs(MoveAnimIds) do
            if anim.AnimationId == moveAnimId then
                HandleMoveAnimation(animTrack, char)
                break
            end
        end
    end))
end


local AnimatorConnections = {}

local function DisconnectAnimatorListeners()
    for _, conn in ipairs(AnimatorConnections) do
        if conn then pcall(function() conn:Disconnect() end) end
    end
    AnimatorConnections = {}
end

local function AttachAnimatorListeners(animator)
    if not animator then return end
    DisconnectAnimatorListeners()
    
    table.insert(AnimatorConnections, animator.AnimationPlayed:Connect(function(animTrack)
        if not AutoVoidEnabled or IsVoiding then return end
        local animation = animTrack.Animation
        if not animation then return end
        for _, animData in ipairs(VoidAnimations) do
            local animId = animation.AnimationId:gsub("rbxassetid://", "")
            if animId == animData.id then
                IsVoiding = true
                local char = LocalPlayer.Character
                local hum = char and char:FindFirstChildOfClass("Humanoid")
                local rootPart = char and char:FindFirstChild("HumanoidRootPart")
                if not char or not char.Parent or not hum or hum.Health <= 0 or not rootPart then
                    IsVoiding = false
                    return
                end
                local oldCFrame = rootPart.CFrame
                task.wait(animData.timewait)
                char = LocalPlayer.Character
                hum = char and char:FindFirstChildOfClass("Humanoid")
                rootPart = char and char:FindFirstChild("HumanoidRootPart")
                if char and char.Parent and AutoVoidEnabled and hum and hum.Health > 0 and rootPart then
                    rootPart.CFrame = CFrame.new(TeleportPosition)
                    animTrack.Stopped:Wait()
                    if TpBackEnabled then
                        char = LocalPlayer.Character
                        hum = char and char:FindFirstChildOfClass("Humanoid")
                        rootPart = char and char:FindFirstChild("HumanoidRootPart")
                        if char and char.Parent and AutoVoidEnabled and hum and hum.Health > 0 and rootPart then
                            rootPart.CFrame = oldCFrame
                        end
                    end
                end
                IsVoiding = false
                break
            end
        end
    end))
    
    table.insert(AnimatorConnections, animator.AnimationPlayed:Connect(function(animTrack)
        if AutoWhirlwindDunk and animTrack.Animation then
            local animId = animTrack.Animation.AnimationId
            if animId == WhirlwindAnimId or animId == "rbxassetid://12296113986" then
                task.wait(1.2)
                local char = LocalPlayer.Character
                local rootPart = char and char:FindFirstChild("HumanoidRootPart")
                if rootPart then
                    local pos = rootPart.Position
                    rootPart.CFrame = CFrame.new(pos.X, pos.Y + 100, pos.Z)
                end
            end
        end
    end))
end

local function SendQInput()
    VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Q, false, game)
    task.wait(0.1)
    VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Q, false, game)
end

local function HandleWallComboAnimation(animTrack, character)
    if AutoWallCombo and animTrack.Animation then
        local rootPart = WallComboAnimIds[animTrack.Animation.AnimationId] and character:FindFirstChild("HumanoidRootPart")
        if rootPart then
            local savedCFrame = rootPart.CFrame
            local startTime = tick()
            local connection
            connection = RunService.Heartbeat:Connect(function()
                if AutoWallCombo and tick() - startTime < 0.3 then
                    if rootPart and rootPart.Parent then
                        rootPart.CFrame = savedCFrame * CFrame.Angles(math.rad(-25), 0, 0)
                    end
                else
                    if rootPart and rootPart.Parent then
                        rootPart.CFrame = savedCFrame
                    end
                    connection:Disconnect()
                end
            end)
        end
    end
end

local function SetupWallCombo(character)
    if WallComboConnection1 then WallComboConnection1:Disconnect() end
    if WallComboConnection2 then WallComboConnection2:Disconnect() end
    WallComboConnection1 = character.DescendantAdded:Connect(function(descendant)
        if descendant:IsA("ObjectValue") and descendant.Name:lower() == "wallcombo" and AutoWallCombo then
            local startTime = tick()
            repeat
                SendQInput()
                task.wait()
            until not descendant.Parent or descendant.Parent ~= character or tick() - startTime >= (descendant:GetAttribute("DeleteMe") or 0.6)
        end
    end)
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    if humanoid then
        WallComboConnection2 = humanoid.AnimationPlayed:Connect(function(animTrack)
            HandleWallComboAnimation(animTrack, character)
        end)
    end
end

local function CreateUltEsp(player, character)
    if not character then return end
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local head = character:FindFirstChild("Head")
    if not humanoid or not head then return end
    if UltEspData[player] then UltEspData[player][1]() end
    local espText = Drawing.new("Text")
    espText.Visible = false
    espText.Center = true
    espText.Outline = true
    espText.Font = Drawing.Fonts.UI
    espText.Size = 14
    espText.Color = Color3.fromRGB(255, 255, 0)
    local function GetUltText()
        local liveChar = workspace.Live:FindFirstChild(player.Name)
        if liveChar then
            local ultBar = liveChar:GetAttribute("UltimateBar") or 0
            return string.format("[ULT: %.0f%%]", ultBar * 100)
        end
        return "[ULT: 0%]"
    end

    local function Cleanup()
        espText:Remove()
        for _, conn in ipairs(connections) do conn:Disconnect() end
        UltEspData[player] = nil
    end

        if not parent then Cleanup() end
    end))
    table.insert(connections, humanoid.Died:Connect(Cleanup))
    table.insert(connections, RunService.RenderStepped:Connect(function()
        if EspUltBar and head and head.Parent then
            local screenPos, onScreen = workspace.CurrentCamera:WorldToViewportPoint(head.Position)
            if onScreen then
                espText.Position = Vector2.new(screenPos.X, screenPos.Y - 42)
                espText.Text = GetUltText()
                espText.Visible = true
            else
                espText.Visible = false
            end
        else
            espText.Visible = false
        end
    end))
    UltEspData[player] = {Cleanup}
end

local function createESP(player)
    if player == LocalPlayer then return end
    local char = player.Character
    if not char then return end
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "ESP"
    billboard.AlwaysOnTop = true
    billboard.Size = UDim2.new(0, 200, 0, 50)
    billboard.StudsOffset = Vector3.new(0, 3, 0)
    billboard.Adornee = char:FindFirstChild("Head")
    billboard.Parent = char:FindFirstChild("Head")
    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size = UDim2.new(1, 0, 0.5, 0)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text = player.Name
    nameLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    nameLabel.TextSize = 14
    nameLabel.Parent = billboard
    local healthLabel = Instance.new("TextLabel")
    healthLabel.Size = UDim2.new(1, 0, 0.5, 0)
    healthLabel.Position = UDim2.new(0, 0, 0.5, 0)
    healthLabel.BackgroundTransparency = 1
    healthLabel.Text = ""
    healthLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
    healthLabel.TextSize = 14
    healthLabel.Parent = billboard
    return {billboard = billboard, healthLabel = healthLabel, chams = {}}
end

local function removeESP(player)
    if espInstances[player] then
        if espInstances[player].billboard then espInstances[player].billboard:Destroy() end
        for _, cham in pairs(espInstances[player].chams or {}) do cham:Destroy() end
        espInstances[player] = nil
    end
end

local function updateESP()
    for player, data in pairs(espInstances) do
        local char = player.Character
        if char and data.healthLabel then
            local humanoid = char:FindFirstChild("Humanoid")
            if humanoid then data.healthLabel.Text = tostring(math.floor(humanoid.Health)) end
        end
    end
end

local function updateRainbowChams()
    local hue = (tick() * rainbowCycleSpeed) % 1
    local color = Color3.fromHSV(hue, 1, 1)
    for _, data in pairs(espInstances) do
        for _, cham in pairs(data.chams or {}) do
            cham.FillColor = color
            cham.OutlineColor = color
        end
    end
end

            EspUltBar = state
            if state then
                for _, player in pairs(Players:GetPlayers()) do
                    SetupUltEspForPlayer(player)
                end
            else
                ClearAllUltEsp()
            end
        end
    })
    
    EspSection:Space()
    
    EspSection:Toggle({
        Title = "ESP Death Counter",
        Desc = "Show player deaths",
        Value = false,
        Callback = function(state)
            EspDeathCounter = state
            if state then
                for _, player in pairs(Players:GetPlayers()) do
                    SetupDeathEspForPlayer(player)
                end
            else
                ClearAllDeathEsp()
            end
        end
    })
    
    EspTab:Space()
    
    local ChamsSection = EspTab:Section({
        Title = "Chams & Highlights",
        Box = true,
        BoxBorder = true,
        Opened = true,
    })
    
    ChamsSection:Toggle({
        Title = "ESP (Username + Health)",
        Value = false,
        Callback = function(state)
            espEnabled = state
            if state then
                for _, player in pairs(Players:GetPlayers()) do
                    if player ~= LocalPlayer then
                        espInstances[player] = createESP(player)
                    end
                end
                Players.PlayerAdded:Connect(function(player)
                    if espEnabled and player ~= LocalPlayer then
                        player.CharacterAdded:Connect(function()
                            espInstances[player] = createESP(player)
                        end)
                    end
                end)
                RunService.RenderStepped:Connect(updateESP)
            else
                for player in pairs(espInstances) do
                    removeESP(player)
                end
            end
        end
    })
    
    ChamsSection:Space()
    
    ChamsSection:Toggle({
        Title = "Rainbow Chams",
        Value = false,
        Callback = function(state)
            chamsEnabled = state
            if state then
                for player in pairs(espInstances) do
                    local char = player.Character
                    if char then
                        espInstances[player].chams = {}
                        for _, part in pairs(char:GetChildren()) do
                            if part:IsA("BasePart") then
                                local highlight = Instance.new("Highlight")
                                highlight.FillTransparency = 0.5
                                highlight.OutlineTransparency = 0
                                highlight.Parent = part
                                table.insert(espInstances[player].chams, highlight)
                            end
                        end
                    end
                end
            else
                for player in pairs(espInstances) do
                    for _, cham in pairs(espInstances[player].chams or {}) do
                        cham:Destroy()
                    end
                    espInstances[player].chams = {}
                end
            end
        end
    })
    
    ChamsSection:Space()
    
    ChamsSection:Slider({
        Title = "Rainbow Speed",
        IsTooltip = true,
        Step = 0.1,
        Value = {
            Min = 0.1,
            Max = 5,
            Default = 1,
        },
        Callback = function(value)
            rainbowCycleSpeed = value
        end
    })
