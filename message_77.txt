local function createEsp(plr)
if not plr.Character or ESPHighlights[plr] then return end
local highlight=Instance.new("Highlight")
highlight.Name="ESP"
highlight.Adornee=plr.Character
highlight.FillTransparency=0.5
highlight.OutlineTransparency=0
highlight.FillColor=Color3.fromRGB(255,0,0)
highlight.OutlineColor=Color3.fromRGB(255,255,255)
highlight.DepthMode=Enum.HighlightDepthMode.AlwaysOnTop
highlight.Parent=plr.Character
ESPHighlights[plr]=highlight
end

local function updateEsp(plr)
if not plr.Character then return end
local role="Innocent"
if plr.Backpack:FindFirstChild("Knife") or (plr.Character and plr.Character:FindFirstChild("Knife")) then role="Murderer"
elseif plr.Backpack:FindFirstChild("Gun") or (plr.Character and plr.Character:FindFirstChild("Gun")) then role="Sheriff" end
local color=role=="Murderer" and Color3.fromRGB(255,0,0) or role=="Sheriff" and Color3.fromRGB(0,100,255) or Color3.fromRGB(0,255,0)
if ESPHighlights[plr] then
ESPHighlights[plr].FillColor=color
else
createEsp(plr)
end
end

local function removeEsp(plr)
if ESPHighlights[plr] then
ESPHighlights[plr]:Destroy()
ESPHighlights[plr]=nil
end
end

local function espLoop()
for _,plr in pairs(Players:GetPlayers()) do
if plr~=LocalPlayer then
updateEsp(plr)
end
end
end

RunService.Heartbeat:Connect(espLoop)

for _,plr in pairs(Players:GetPlayers()) do
if plr~=LocalPlayer then
plr.CharacterAdded:Connect(function() updateEsp(plr) end)
end
end

Players.PlayerAdded:Connect(function(plr)
plr.CharacterAdded:Connect(function() updateEsp(plr) end)
end)
Players.PlayerRemoving:Connect(removeEsp)
