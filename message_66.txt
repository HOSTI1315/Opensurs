local dragging = false
local dragInput, dragStart, startPos

local function update(input)
    local delta = input.Position - dragStart
    MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

TopBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

TopBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        update(input)
    end
end)

local function safeDisconnect(connection)
    if connection and typeof(connection) == "RBXScriptConnection" then
        connection:Disconnect()
        return nil
    end
    return connection
end

local function getFishingRodFromBackpack()
    local backpack = player:WaitForChild("Backpack")
    
    for _, tool in ipairs(backpack:GetChildren()) do
        if tool:IsA("Tool") and (tool.Name:lower():find("rod") or tool.Name:lower():find("fishing")) then
            return tool
        end
    end
    
    return nil
end

local function getFishingRod()
    local character = player.Character
    if not character then return nil end

    for _, tool in ipairs(character:GetChildren()) do
        if tool:IsA("Tool") and (tool.Name:lower():find("rod") or tool.Name:lower():find("fishing")) then
            return tool
        end
    end

    return nil
end

local function equipFishingRod()
    local rod = getFishingRod()
    if rod then
        return rod
    end
    
    rod = getFishingRodFromBackpack()
    if rod then
        local character = player.Character
        if character and character:FindFirstChild("Humanoid") then
            character.Humanoid:EquipTool(rod)
            task.wait(0.5)
            return getFishingRod()
        end
    end
    
    return nil
end

local function performAutoCast()
    local currentTime = tick()
    if currentTime - State.lastCastTime < State.castCooldown then
        return
    end

    if State.isAutoCasting or State.isCasted then return end

    local rod = getFishingRod()
    if not rod then return end

    local hasCooldown = rod:GetAttribute("castCooldown")
    local alreadyCasted = rod:GetAttribute("casted")

    if hasCooldown or alreadyCasted then
        return
    end

    State.isAutoCasting = true
    State.lastCastTime = currentTime

    local castSuccess = false

    pcall(function()
        rod:Activate()
        task.wait(Config.CastTiming)
        rod:Deactivate()
        castSuccess = true
    end)

    if not castSuccess and CastEvent then
        pcall(function()
            if CastEvent:IsA("RemoteEvent") then
                CastEvent:FireServer()
                castSuccess = true
            end
        end)
    end

    task.wait(0.1)
    State.isAutoCasting = false
end

local function autoCastLoop()
    if State.castLoopRunning then return end
    State.castLoopRunning = true

    task.spawn(function()
        while Config.FishingActive do
            local rod = getFishingRod()

            if rod and not State.isCasted and not State.isAutoCasting then
                local hasCooldown = rod:GetAttribute("castCooldown")
                local alreadyCasted = rod:GetAttribute("casted")

                if not hasCooldown and not alreadyCasted then
                    performAutoCast()
                end
            end

            task.wait(Config.RecastDelay)
        end
        State.castLoopRunning = false
    end)
end

local function autoClickMinigame()
    State.clickConnection = safeDisconnect(State.clickConnection)

    local function performClick(rod)
        pcall(function()
            rod:Activate()
            rod:Deactivate()
        end)

        if MinigameClickEvent then
            pcall(function()
                MinigameClickEvent:FireServer()
            end)
        end
    end

    State.clickConnection = RunService.RenderStepped:Connect(function()
        if not State.isMinigameActive or not Config.FishingActive then
            State.clickConnection = safeDisconnect(State.clickConnection)
            return
        end

        local rod = getFishingRod()

        if not rod or not rod:GetAttribute("minigame") then
            State.clickConnection = safeDisconnect(State.clickConnection)
            State.isMinigameActive = false
            return
        end

        for i = 1, 15 do
            performClick(rod)
        end
    end)
end

local function setupRodMonitoring(rod)
    if not rod then return end

    State.currentRod = rod

    for i, conn in ipairs(State.connections) do
        safeDisconnect(conn)
    end
    State.connections = {}

    local minigameConnection = rod:GetAttributeChangedSignal("minigame"):Connect(function()
        local minigameState = rod:GetAttribute("minigame")

        if minigameState and not State.isMinigameActive then
            State.isMinigameActive = true
            StatusText.Text = "MINIGAME!"
            StatusIndicator.BackgroundColor3 = Color3.fromRGB(255, 255, 50)
            if Config.FishingActive then
                task.spawn(autoClickMinigame)
            end
        elseif not minigameState and State.isMinigameActive then
            State.isMinigameActive = false
            State.fishCaught = State.fishCaught + 1
            FishCountText.Text = string.format("Fish: %d | Time: %ds", State.fishCaught, math.floor(tick() - State.startTime))
            StatusText.Text = "FISHING"
            StatusIndicator.BackgroundColor3 = Color3.fromRGB(50, 255, 50)
            State.clickConnection = safeDisconnect(State.clickConnection)
        end
    end)
    table.insert(State.connections, minigameConnection)

    local castConnection = rod:GetAttributeChangedSignal("casted"):Connect(function()
        local wasCasted = State.isCasted
        State.isCasted = rod:GetAttribute("casted") or false

        if not State.isCasted and wasCasted then
            if Config.FishingActive then
                task.wait(Config.RecastDelay)
                performAutoCast()
            end
        end
    end)
    table.insert(State.connections, castConnection)

    local unequipConnection = rod.AncestryChanged:Connect(function()
        if not rod:IsDescendantOf(player.Character) then
            for i, conn in ipairs(State.connections) do
                safeDisconnect(conn)
            end
            State.connections = {}
            State.clickConnection = safeDisconnect(State.clickConnection)
            State.isMinigameActive = false
            State.isCasted = false
            State.currentRod = nil
            State.castLoopRunning = false
        end
    end)
    table.insert(State.connections, unequipConnection)
end

local function onCharacterAdded(character)
    for i, conn in ipairs(State.connections) do
        safeDisconnect(conn)
    end
    State.connections = {}
    State.clickConnection = safeDisconnect(State.clickConnection)

    local childAddedConnection = character.ChildAdded:Connect(function(child)
        if child:IsA("Tool") and (child.Name:lower():find("rod") or child.Name:lower():find("fishing")) then
            task.wait(0.2)
            setupRodMonitoring(child)

            if Config.FishingActive then
                task.wait(0.3)
                autoCastLoop()
            end
        end
    end)
    table.insert(State.connections, childAddedConnection)

    task.wait(0.2)
    local rod = getFishingRod()
    if rod then
        setupRodMonitoring(rod)
        if Config.FishingActive then
            autoCastLoop()
        end
    end
end

if player.Character then
    onCharacterAdded(player.Character)
end

player.CharacterAdded:Connect(onCharacterAdded)

StartButton.MouseButton1Click:Connect(function()
    Config.FishingActive = not Config.FishingActive

    if Config.FishingActive then
        State.startTime = tick()
        State.fishCaught = 0
        FishCountText.Text = "Fish: 0 | Time: 0s"

        local rod = equipFishingRod()
        if rod then
            setupRodMonitoring(rod)
            task.wait(0.5)
            autoCastLoop()
        else
            Config.FishingActive = false
            StatusText.Text = "NO ROD FOUND"
            StatusIndicator.BackgroundColor3 = Color3.fromRGB(255, 150, 50)
            task.wait(2)
            StatusText.Text = "IDLE"
            StatusIndicator.BackgroundColor3 = Color3.fromRGB(255, 50, 50)
        end
    else
        State.castLoopRunning = false
        State.clickConnection = safeDisconnect(State.clickConnection)
    end

    updateUI()

    local original = StartButton.BackgroundColor3
    StartButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    task.wait(0.08)
    StartButton.BackgroundColor3 = original
end)

local function addHoverEffect(button, normalColor)
    button.MouseEnter:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.18), {BackgroundColor3 = button.BackgroundColor3:Lerp(Color3.fromRGB(255, 255, 255), 0.18)}):Play()
    end)

    button.MouseLeave:Connect(function()
        local targetColor = normalColor
        if button == StartButton then
            targetColor = Config.FishingActive and Color3.fromRGB(255, 50, 50) or Color3.fromRGB(0, 170, 255)
        end
        TweenService:Create(button, TweenInfo.new(0.18), {BackgroundColor3 = targetColor}):Play()
    end)
end
