local Aiming = false
local nearestTarget = nil
local maxDistance = 100
local checkDelay = 0.1
local lastCheckTime = 0

-- Aim directly at the target with a small upward offset
local function AimAt(part)
	if part then
		local cam = workspace.CurrentCamera
		local camPos = cam.CFrame.Position

		local targetPos = part.Position + Vector3.new(0, 1.5, 0) -- Slight height only
		cam.CFrame = CFrame.lookAt(camPos, targetPos)
	end
end

-- Finds the nearest "Normal" model's PrimaryPart
local function getNearestTarget()
	local closest = nil
	local shortest = maxDistance

	for _, model in ipairs(workspace:GetDescendants()) do
		if model:IsA("Model") and model.Name == "Normal" then
			local part = model.PrimaryPart or model:FindFirstChildWhichIsA("BasePart")
			if part then
				local dist = (part.Position - workspace.CurrentCamera.CFrame.Position).Magnitude
				if dist < shortest then
					shortest = dist
					closest = part
				end
			end
		end
	end

	return closest
end

	Aiming = Value

game:GetService("RunService").RenderStepped:Connect(function()
	if Aiming then
		local now = tick()

		if not nearestTarget 
			or not nearestTarget:IsDescendantOf(workspace)
			or (nearestTarget.Position - workspace.CurrentCamera.CFrame.Position).Magnitude > maxDistance then

			if now - lastCheckTime >= checkDelay then
				lastCheckTime = now
				nearestTarget = getNearestTarget()
			end
		end

		if nearestTarget then
			AimAt(nearestTarget)
		end
	end
end)

local movementConnection = nil  -- Used to store the movement loop connection

-- Function to start CFrame movement
function startCFrameSpeed()
    if movementConnection == nil then
        movementConnection = RunService.Stepped:Connect(function()
            LocalPlayer.Character.HumanoidRootPart.CFrame = LocalPlayer.Character.HumanoidRootPart.CFrame + LocalPlayer.Character.Humanoid.MoveDirection * Multiplier
        end)
    end
end

-- Function to stop CFrame movement
function stopCFrameSpeed()
    if movementConnection then
        movementConnection:Disconnect()
        movementConnection = nil
    end
end

local function createTutorialGui()
    -- Create the GUI
    local Turtorial = Instance.new("ScreenGui")
    local main = Instance.new("Frame")
    local JoinButton = Instance.new("TextButton")
    local UICorner_Join = Instance.new("UICorner")
    local shadow = Instance.new("Frame")

    Turtorial.Name = "Turtorial"
    Turtorial.Parent = player:WaitForChild("PlayerGui")
    Turtorial.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

    main.Name = "main"
    main.Parent = Turtorial
    main.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    main.AnchorPoint = Vector2.new(0, 0)

    -- Function to position the GUI just below the top-left corner of the screen
    local function setPositionBelowTopLeft()
        local screenSize = game:GetService("Workspace").CurrentCamera.ViewportSize
        main.Position = UDim2.new(0, 10, 0, 10)  -- Just below the top-left with a slight offset
        main.Size = UDim2.new(0.15, 0, 0.10, 0)  -- Standard size for both devices
    end

    setPositionBelowTopLeft()

    main.Active = true
    main.Draggable = true

    -- Shadow effect for main frame
    shadow.Name = "shadow"
    shadow.Parent = main
    shadow.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    shadow.BackgroundTransparency = 0.6
    shadow.Position = UDim2.new(0, 5, 0, 5)
    shadow.Size = UDim2.new(1, 0, 1, 0)
    shadow.ZIndex = 0

    JoinButton.Name = "JoinButton"
    JoinButton.Parent = main
    JoinButton.BackgroundColor3 = Color3.fromRGB(40, 120, 255)

    -- Adjust button size based on device type
    if game:GetService("UserInputService").TouchEnabled then
        JoinButton.Size = UDim2.new(0.9, 0, 0.4, 0)
    else
        JoinButton.Size = UDim2.new(0.9, 0, 0.4, 0)
    end

    JoinButton.Position = UDim2.new(0.05, 0, 0.5, 0)
    JoinButton.Font = Enum.Font.Highway
    JoinButton.Text = "Toggle Gui"
    JoinButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    JoinButton.TextSize = 25
    JoinButton.TextStrokeTransparency = 0.7
    JoinButton.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)

    UICorner_Join.CornerRadius = UDim.new(0.15, 0)
    UICorner_Join.Parent = JoinButton

    -- Toggle the visibility of ScreenGuis
    JoinButton.MouseButton1Down:Connect(function()
        local CoreGui = game:GetService("CoreGui")
        local function containsNumber(name)
            return name:match("%d") ~= nil
        end

        local function toggleScreenGuisWithNumbers()
            for _, gui in ipairs(CoreGui:GetChildren()) do
                if gui:IsA("ScreenGui") and containsNumber(gui.Name) and gui.Name ~= "Turtorial" then
                    gui.Enabled = not gui.Enabled
                    print((gui.Enabled and "Enabled" or "Disabled") .. " ScreenGui:", gui.Name)
                end
            end
        end

        toggleScreenGuisWithNumbers()

        CoreGui.ChildAdded:Connect(function(child)
            if child:IsA("ScreenGui") then
                task.wait(0.1)
                if containsNumber(child.Name) and child.Name ~= "Turtorial" then
                    child.Enabled = false
                    print("Disabled new ScreenGui:", child.Name)
                end
            end
        end)
    end)
end

-- Connect to CharacterAdded to ensure GUI is created again after respawn
player.CharacterAdded:Connect(function()
    -- Wait for the PlayerGui to load in case it's reset
    player:WaitForChild("PlayerGui")
    -- Create the GUI again after respawn
    createTutorialGui()
end)

-- Initial creation of the GUI when the player first joins
createTutorialGui()
