local function showSystemNotification(message)
    local notifyGui = Instance.new("ScreenGui")
    notifyGui.Name = "SystemNotify"
    notifyGui.Parent = PlayerGui
    
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, 300, 0, 50)
    frame.Position = UDim2.new(0.5, -150, 0.1, -100)

    stroke.Parent = frame

    label.BackgroundTransparency = 1
    label.Text = message
    label.TextColor3 = Color3.new(1,1,1)

    label.TextSize = 16
    label.Parent = frame
    
    TweenService:Create(frame, TweenInfo.new(0.5, Enum.EasingStyle.Back), {Position = UDim2.new(0.5, -150, 0.1, 20)}):Play()
    task.wait(2.5)
    local out = TweenService:Create(frame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.In), {Position = UDim2.new(0.5, -150, 0.1, -100)})
    out:Play()
    out.Completed:Wait()
    notifyGui:Destroy()
end

    -- === CORE LOGIC ===

    local function applyHighlight(obj, typeName)
        if not obj or not obj.Parent then return end
        if activeHighlights[obj] then return end 

        local colors = HIGHLIGHT_COLORS[typeName]
        if not colors then return end

        local esp = Instance.new("Highlight")
        esp.Name = typeName .. "Esp"
        esp.Adornee = obj
        esp.FillColor = colors[1]
        esp.OutlineColor = colors[2]
        esp.FillTransparency = 0.5 
        esp.OutlineTransparency = 0.1
        esp.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
        esp.Parent = obj
        
        activeHighlights[obj] = esp
        
        obj.Destroying:Connect(function()
            activeHighlights[obj] = nil
        end)
    end

    local function checkObject(obj)
        if not obj then return end
        
        -- Instant Prompt Setup
        if obj:IsA("ProximityPrompt") and states.instantPrompt then
            obj.HoldDuration = 0
        end
        
        -- Priority Checks
        if states.espDoors and obj.Name == "Door" and obj:IsA("BasePart") and obj.Parent.Name == "Door" then 
            applyHighlight(obj, "door")
        elseif states.espDupe and obj.Name == "Door" and obj:IsA("BasePart") and obj.Parent.Name == "DoorFake" then
            applyHighlight(obj, "dupe")
        elseif states.antiDupe and obj.Name == "Hidden" and obj.Parent.Name == "DoorFake" then
            obj:Destroy()
            
        elseif states.chestHighlight and obj.Name == "ChestBox" then applyHighlight(obj, "chest")
        elseif states.leverHighlight and obj.Name == "LeverForGate" then applyHighlight(obj, "lever")
        elseif states.smoothieHighlight and obj.Name == "Smoothie" then applyHighlight(obj, "smoothie")
        elseif states.keyHighlight and obj.Name == "KeyObtain" then applyHighlight(obj, "key")
        elseif states.wardrobeEsp and obj.Name == "Wardrobe" then applyHighlight(obj, "wardrobe")
        elseif states.liveHintBookEsp and obj.Name == "LiveHintBook" then applyHighlight(obj, "liveHintBook")
        elseif states.figureEsp and obj.Name == "FigureRig" then applyHighlight(obj, "figure")
        elseif states.liveBreakerEsp and obj.Name == "LiveBreakerPolePickup" then applyHighlight(obj, "lever")
        
        elseif states.itemsEsp and obj:IsA("Model") then
            local name = obj.Name
            if name ~= "Battery" and name ~= "Bandage" and 
               name ~= "Wardrobe" and name ~= "LeverForGate" and 
               name ~= "Door" and name ~= "DoorFake" and 
               name ~= "DrawerContainer" and name ~= "GiggleCeiling" then
                
                local hasHandle = obj:FindFirstChild("Handle")
                local hasPrompt = obj:FindFirstChildWhichIsA("ProximityPrompt", true)
                
                if hasHandle and hasPrompt then
                    applyHighlight(obj, "item")
                end
            end
        end

        if states.rushAlert and obj.Name == "RushMoving" then applyHighlight(obj, "rush")
        elseif states.ambushAlert and obj.Name == "AmbushMoving" then applyHighlight(obj, "rush") end
        
        if states.noPaintRoomDoor and obj.Name == "MovingDoor" then obj:Destroy() end
        
        if states.antiLag then
            if obj:IsA("Light") or obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") then obj:Destroy() end
        end
        
        if states.superFps then
            if obj:FindFirstAncestor("Eyes") then return end
            if obj:IsA("BasePart") then
                obj.Material = Enum.Material.SmoothPlastic
                obj.Reflectance = 0
                obj.CastShadow = false
            elseif obj:IsA("Texture") or obj:IsA("Decal") then 
                obj:Destroy() 
            end
        end
    end

    local mainConnection = Workspace.DescendantAdded:Connect(checkObject)

    local function scanExisting()
        for _, obj in ipairs(Workspace:GetDescendants()) do 
            checkObject(obj) 
        end
    end

    local function giggleLogic()
        local function removeHitbox(model)
            for _, part in ipairs(model:GetDescendants()) do
                if part.Name == "Hitbox" then part:Destroy() end
            end
        end
        for _, model in ipairs(Workspace:GetDescendants()) do
            if model.Name == "GiggleCeiling" then removeHitbox(model) end
        end
        Workspace.DescendantAdded:Connect(function(descendant)
            if descendant.Name == "GiggleCeiling" then removeHitbox(descendant)
            elseif descendant.Name == "Hitbox" then
                local p = descendant.Parent
                while p do
                    if p.Name == "GiggleCeiling" then descendant:Destroy(); break end
                    p = p.Parent
                end
            end
        end)
    end

    local function createSlider(parent, text, min, max, default, stateKey)
        local container = Instance.new("Frame")
        container.Size = UDim2.new(0.98, 0, 0, 40)
        container.BackgroundColor3 = Color3.fromRGB(40, 40, 45)
        container.Parent = parent
        Instance.new("UICorner", container).CornerRadius = UDim.new(0, 8)
        Instance.new("UIStroke", container).Color = Color3.fromRGB(70, 70, 80)
        local label = Instance.new("TextLabel")
        label.Size = UDim2.new(1, 0, 0, 20)
        label.BackgroundTransparency = 1
        label.Text = text .. ": " .. default
        label.TextColor3 = Color3.fromRGB(200, 200, 200)
        label.Font = Enum.Font.GothamSemibold
        label.TextSize = 12
        label.Parent = container
        local sliderBg = Instance.new("TextButton")
        sliderBg.Text = ""
        sliderBg.Size = UDim2.new(0.9, 0, 0, 6)
        sliderBg.Position = UDim2.new(0.05, 0, 0.7, 0)
        sliderBg.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
        sliderBg.AutoButtonColor = false
        sliderBg.Parent = container
        Instance.new("UICorner", sliderBg).CornerRadius = UDim.new(1, 0)
        local fill = Instance.new("Frame")
        fill.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
        fill.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
        fill.BorderSizePixel = 0
        fill.Parent = sliderBg
        Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)
        states[stateKey] = default
        local dragging = false
        local function update(input)
            local pos = input.Position.X
            local rect = sliderBg.AbsolutePosition.X
            local width = sliderBg.AbsoluteSize.X
            local percent = math.clamp((pos - rect) / width, 0, 1)
            local value = math.floor(min + (max - min) * percent)
            states[stateKey] = value
            label.Text = text .. ": " .. value
            TweenService:Create(fill, TweenInfo.new(0.1), {Size = UDim2.new(percent, 0, 1, 0)}):Play()
        end
        sliderBg.MouseButton1Down:Connect(function() dragging = true end)
        UserInputService.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end end)
        UserInputService.InputChanged:Connect(function(input) if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then update(input) end end)
        sliderBg.MouseButton1Down:Connect(function() local input = {Position = Vector3.new(mouse.X, mouse.Y, 0), UserInputType = Enum.UserInputType.MouseMovement}; update(input) end)
    end

        end
        
        if states.speedEnabled and player.Character then
            local humanoid = player.Character:FindFirstChild("Humanoid")
            local rootPart = player.Character:FindFirstChild("HumanoidRootPart")
            
            if humanoid and rootPart and humanoid.MoveDirection.Magnitude > 0 then
                local newVelocity = humanoid.MoveDirection * states.speedValue
                rootPart.AssemblyLinearVelocity = Vector3.new(newVelocity.X, rootPart.AssemblyLinearVelocity.Y, newVelocity.Z)
            end
        end
    end)

    ProximityPromptService.PromptButtonHoldBegan:Connect(function(prompt)
        if states.instantPrompt and prompt.Enabled then
            fireproximityprompt(prompt)
        end
    end)

    ProximityPromptService.PromptShown:Connect(function(prompt)
        if states.instantPrompt then prompt.HoldDuration = 0 end
        if states.autoInteract then
            local excluded = false
            local current = prompt.Parent
            for i = 1, 3 do if current then if current.Name == "Wardrobe" or current.Name == "Bed" then excluded = true; break end current = current.Parent end end
            if not excluded then fireproximityprompt(prompt) end
        end
    end)

    RunService.RenderStepped:Connect(function()
        if states.fullbright then
            Lighting.Brightness = 2
            Lighting.ClockTime = 14
            Lighting.FogEnd = 100000
            Lighting.GlobalShadows = false
            Lighting.Ambient = Color3.new(1, 1, 1)
            Lighting.OutdoorAmbient = Color3.new(1, 1, 1)
        end
        if states.instantPrompt then
            for _, prompt in ipairs(ProximityPromptService:GetProximityPrompts()) do
                if prompt.HoldDuration ~= 0 then prompt.HoldDuration = 0 end
            end
        end
    end)
    
    print("HubHubHub FINAL VERSION Loaded")
    task.wait(1)
    showNotification("Tip: Press F2 to Toggle Menu")
end

-- === KEY SYSTEM GUI CREATION ===
local function CreateKeySystem()
    local KeyGui = Instance.new("ScreenGui")
    KeyGui.Name = "HubHubHubKeySystem"
    KeyGui.Parent = PlayerGui
    KeyGui.ResetOnSpawn = false

    local KeyFrame = Instance.new("Frame")
    KeyFrame.Size = UDim2.new(0, 400, 0, 250)
    KeyFrame.Position = UDim2.new(0.5, -200, 0.5, -125)
    KeyFrame.BackgroundColor3 = Color3.fromRGB(20, 20, 25)
    KeyFrame.BorderSizePixel = 0
    KeyFrame.Parent = KeyGui

    Instance.new("UICorner", KeyFrame).CornerRadius = UDim.new(0, 12)
    local stroke = Instance.new("UIStroke")
    stroke.Color = Color3.fromRGB(60, 60, 70)
    stroke.Thickness = 2
    stroke.Parent = KeyFrame

    local Title = Instance.new("TextLabel")
    Title.Size = UDim2.new(1, 0, 0, 50)
    Title.BackgroundTransparency = 1
    Title.Text = "HubHubHub Key System"
    Title.TextColor3 = Color3.fromRGB(255, 255, 255)
    Title.Font = Enum.Font.GothamBlack
    Title.TextSize = 24
    Title.Parent = KeyFrame

    local KeyInput = Instance.new("TextBox")
    KeyInput.Size = UDim2.new(0.8, 0, 0, 45)
    KeyInput.Position = UDim2.new(0.1, 0, 0.35, 0)
    KeyInput.BackgroundColor3 = Color3.fromRGB(35, 35, 40)
    KeyInput.TextColor3 = Color3.fromRGB(255, 255, 255)
    KeyInput.PlaceholderText = "Enter Key..."
    KeyInput.Font = Enum.Font.GothamBold
    KeyInput.TextSize = 16
    KeyInput.Text = ""
    KeyInput.Parent = KeyFrame
    Instance.new("UICorner", KeyInput).CornerRadius = UDim.new(0, 8)

    -- Buttons
    local function createBtn(text, pos, callback)
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(0.35, 0, 0, 40)
        btn.Position = pos
        btn.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
        btn.Text = text
        btn.TextColor3 = Color3.fromRGB(255, 255, 255)
        btn.Font = Enum.Font.GothamBold
        btn.TextSize = 14
        btn.Parent = KeyFrame
        Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 8)
        
        btn.MouseButton1Click:Connect(function()
            TweenService:Create(btn, TweenInfo.new(0.1), {BackgroundColor3 = Color3.fromRGB(80, 80, 90)}):Play()
            task.wait(0.1)
            TweenService:Create(btn, TweenInfo.new(0.1), {BackgroundColor3 = Color3.fromRGB(45, 45, 50)}):Play()
            callback()
        end)
        return btn
    end

    createBtn("Get Key", UDim2.new(0.1, 0, 0.7, 0), function()
        setclipboard(KEY_LINK)
        showSystemNotification("Link Copied to Clipboard!")
    end)

    createBtn("Activate Key", UDim2.new(0.55, 0, 0.7, 0), function()
        if KeyInput.Text == KEY_STRING then
            KeyGui:Destroy()
            showSystemNotification("Key Accepted! Loading...")
            task.wait(1)
            LoadMainScript()
        else
            showSystemNotification("Invalid Key!")
            KeyInput.Text = ""
        end
    end)
end

-- Start Key System
CreateKeySystem()
