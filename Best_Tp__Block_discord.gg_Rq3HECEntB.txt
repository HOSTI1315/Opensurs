-- Helper function to create crash node
local function createCrashNode(position)
    local node = Instance.new("Part")
    node.Name = "Crash_Node"
    node.Shape = Enum.PartType.Ball
    node.Size = Vector3.new(1, 1, 1)
    node.Anchored = true
    node.CanCollide = false
    node.Material = Enum.Material.Neon
    node.Color = Color3.fromRGB(255, 255, 255)
    node.Position = position
    node.Parent = Workspace
    
    local selectionBox = Instance.new("SelectionBox")
    selectionBox.Adornee = node
    selectionBox.Color3 = Color3.fromRGB(255, 255, 255)
    selectionBox.LineThickness = 0.03
    selectionBox.Parent = node
    
    return node
end

-- Create initial crash node
local CrashNode = createCrashNode(DEFAULT_POSITION)
StatusLabel.Text = "> LOADED"

-- Overwrite button click (sets node to player position)
OverwriteButton.MouseButton1Click:Connect(function()
    local character = LocalPlayer.Character
    if not character then return end
    
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    
    if CrashNode then
        CrashNode:Destroy()
    end
    
    CrashNode = createCrashNode(hrp.Position)
    StatusLabel.Text = "> UPDATED"
end)

-- Restore button click (resets node to default position)
RestoreButton.MouseButton1Click:Connect(function()
    if CrashNode then
        CrashNode:Destroy()
    end
    
    CrashNode = createCrashNode(DEFAULT_POSITION)
    StatusLabel.Text = "> RESTORED"
end)

-- Teleport button click (equips flying carpet)
TeleportButton.MouseButton1Click:Connect(function()
    local character = LocalPlayer.Character
    if not character then return end
    
    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid then return end
    
    local backpack = LocalPlayer:FindFirstChild("Backpack")
    if not backpack then return end
    
    local flyingCarpet = backpack:FindFirstChild("Flying Carpet")
    if flyingCarpet then
        humanoid:EquipTool(flyingCarpet)
    end
end)
