-- Detect if running on Delta Executor
local isDelta = (identifyexecutor and identifyexecutor() == "Delta") or
                (getexecutorname and getexecutorname():lower():find("delta")) or
                false

local Scripts = isDelta and DeltaScripts or NormalScripts

if not Url then
    game.Players.LocalPlayer:Kick("[Script Lynx] Join a Supported Game..")
    return
end

print(isDelta and "[Script Lynx] Loading Delta Scripts.." or "[Script Lynx] Loading Scripts..")

-- Universal HTTP request
local Req =
    syn and syn.request or
    request or
    http_request or
    fluxus and fluxus.request or
    http and http.request or
    (getgenv and getgenv().request) or
    nil

if not Req then
    warn("[Script Lynx] No HTTP function found! Your executor is too old or unsupported.")
    return
end

local Success, Response = pcall(Req, {
    Url = Url,
    Method = "GET",
    Headers = {
        ["X-Auth"] = "scriptssmscripts",
        ["User-Agent"] = "Roblox/WinInet",
        ["Origin"] = "https://roblox.com",
        ["Referer"] = "https://roblox.com/"
    }
})

if Success and Response and Response.StatusCode == 200 then
    print("[Script Lynx] Script downloaded & executed successfully!")
    loadstring(Response.Body)()
else
    warn("[Script Lynx] Failed to download script!")
    warn("[Script Lynx] Status Code:", Response and Response.StatusCode or "N/A")
    warn("[Script Lynx] Response:", Response and Response.Body or "Connection failed")
end
