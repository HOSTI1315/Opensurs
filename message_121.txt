-- Helper: Virtual Click
local function virtualClick(x,y)
    if not x or not y then return end
    pcall(function()
        VIM:SendMouseButtonEvent(x,y,0,true,game,0)
        task.wait(0.02)
        VIM:SendMouseButtonEvent(x,y,0,false,game,0)
    end)
end

-- Helper: Center of GuiObject
local function getCenterPosition(button)
    if not button then return 0,0 end
    local absPos = button.AbsolutePosition
    local absSize = button.AbsoluteSize
    return absPos.X + absSize.X/2 + clickOffsets.X, absPos.Y + absSize.Y/2 + clickOffsets.Y
end

-- Start a wild battle
local function startBattle()
    task.spawn(function()
        ended = false
        Main.Battle.doWildBattle(Main.Battle, Main.DataManager.currentChunk.regionData.Grass,{})
        ended = true
    end)
end

-- Current Pokémon Energy
local function getEnergy()
    if not Main or not Main.Battle or not Main.Battle.currentBattle then return 0 end
    return Main.Battle.currentBattle.p1.active[1].energy
end

-- Check energy
local function energyCheck()
    return getEnergy() >= maxEnergyMove
end

-- Check if should catch
local function shouldCatch()
    if not Main or not Main.Battle or not Main.Battle.currentBattle then return false end
    local name = string.lower(Main.Battle.currentBattle.p1.foe.active[1].species or "")
    return KeepPokemon[name] == true
end

-- --- PP Reading ---
local function getDigitFromImage(imgLabel)
    if not imgLabel or not imgLabel.ImageRectOffset or not imgLabel.ImageRectSize then return 0 end
    local digitWidth = imgLabel.ImageRectSize.X
    if digitWidth == 0 then return 0 end
    local offsetX = imgLabel.ImageRectOffset.X
    local digit = offsetX / digitWidth
    return math.floor(digit + 0.5)
end

local function getPP(moveGui)
    if not moveGui then return 0,0 end
    local ppContainer = moveGui:FindFirstChild("PPContainer", true)
    if not ppContainer then return 0,0 end
    local ppFrame = ppContainer:FindFirstChild("Frame") or ppContainer
    if not ppFrame then return 0,0 end

    local function readDigitByName(n)
        local child = ppFrame:FindFirstChild(tostring(n))
        if not child or not child:IsA("ImageLabel") then return 0 end
        return getDigitFromImage(child)
    end

    local numFrames = 0
    for _,c in ipairs(ppFrame:GetChildren()) do
        if c:IsA("ImageLabel") or c:IsA("ImageButton") then
            numFrames = numFrames + 1
        end
    end

    local currentPP, maxPP = 0,0
    if numFrames == 7 then
        local d3 = readDigitByName(3)
        local d4 = readDigitByName(4)
        local d6 = readDigitByName(6)
        local d7 = readDigitByName(7)
        currentPP = d3 * 10 + d4
        maxPP     = d6 * 10 + d7
    elseif numFrames == 6 then
        local d3 = readDigitByName(3)
        local d5 = readDigitByName(5)
        local d6 = readDigitByName(6)
        currentPP = d3
        maxPP     = d5 * 10 + d6
    elseif numFrames == 5 then
        local d3 = readDigitByName(3)
        local d5 = readDigitByName(5)
        currentPP = d3
        maxPP     = d5
    else
        currentPP = readDigitByName(3) * 10 + readDigitByName(4)
        maxPP     = readDigitByName(6) * 10 + readDigitByName(7)
    end

    return currentPP, maxPP
end

-- --- Effectiveness Reading ---
-- Tier system:
-- 5 = never use
-- 4 = if have to
-- 3 = regular
-- 2 = prefer use
-- 1 = always use
local function getEffectivenessTier(moveGui)
    if not moveGui then return 5 end
    local eff = moveGui:FindFirstChild("Effectiveness", true)
    if eff and eff.Visible then
        local ox = eff.ImageRectOffset.X
        local oy = eff.ImageRectOffset.Y
        if math.abs(ox - 300) < 1 and math.abs(oy - 0) < 1 then
            return 3 -- regular
        elseif math.abs(ox) < 1 and math.abs(oy) < 1 then
            return 1 -- always use
        else
            return 2 -- prefer use
        end
    else
        -- Effectiveness frame not visible: treat as tier 3
        return 3
    end
end

local function getPokemonName()
    local backGui = PlayerGui:FindFirstChild("BackGui", true)
    if not backGui then return "unknown" end
    local foeHealth = backGui:FindFirstChild("FoeHealthGui", true)
    if not foeHealth then return "unknown" end
    local nameContainer = foeHealth:FindFirstChild("NameContainer", true)
    if not nameContainer then return "unknown" end
    local frame = nameContainer:FindFirstChild("Frame", true)
    if not frame then return "unknown" end

    local name = ""
    for _,v in ipairs(frame:GetChildren()) do
        if v:IsA("ImageLabel") and v.ImageRectOffset then
            local k = keyForOffset(v.ImageRectOffset.X, v.ImageRectOffset.Y)
            name = name .. (LetterMap[k] or "?")
        end
    end
    if name == "" then return "unknown" end
    return name
end

-- Encounters
local totalEncounters = 0
local desiredEncounters = 0
local LastBattleDetected = false
local CurPokemonLbl = Tab:CreateLabel("Current Pokemon: ???")
local TotalEncLbl = Tab:CreateLabel("Total Encounters: 0")
local DesiredEncLbl = Tab:CreateLabel("Desired Encounters: 0")

-- Eject with key
uis.InputBegan:Connect(function(key,proc)
    if key.KeyCode == Enum.KeyCode.E and not proc then
        eject = true
        BotToggle:Set(false)
    end
end)

-- --- MAIN BOT LOOP ---
task.spawn(function()
    while true do
        task.wait(0.1)
        if not BotToggle.CurrentValue then continue end

        startBattle()

        -- Wait for BattleGui
        local BattleGui
        repeat
            BattleGui = PlayerGui:FindFirstChild("MainGui", true)
            if BattleGui then
                BattleGui = BattleGui:FindFirstChild("BattleGui")
            end
            task.wait()
        until BattleGui

        -- Count encounters only once per appearance
        if not LastBattleDetected then
            totalEncounters = totalEncounters + 1
            LastBattleDetected = true
            TotalEncLbl:Set("Total Encounters: " .. totalEncounters)
        end

        -- read current pokemon name (lowercase)
        local curPokemonRaw = string.lower(getPokemonName())
        CurPokemonLbl:Set("Current Pokemon: " .. curPokemonRaw)

        -- Wait 1 second if keep pokemon
        if KeepPokemon[curPokemonRaw] then
            task.wait(1)
            continue
        end

        -- Assign fight/rest/run buttons
        restFight = nil
        clickFight = nil
        clickRun = nil
        for _,v in ipairs(BattleGui:GetChildren()) do
            if v.ClassName == "ImageLabel" then
                if v.ImageColor3 == Color3.fromRGB(218,83,255) then
                    restFight = v:FindFirstChild("Button")
                elseif v.ImageColor3 == Color3.fromRGB(255,102,102) then
                    clickFight = v:FindFirstChild("Button")
                end
            elseif v.Name == "Run" then
                clickRun = v:FindFirstChild("Button") or v
            end
        end

        -- Click Fight until moves appear
        repeat
            if clickFight then
                local x,y = getCenterPosition(clickFight)
                virtualClick(x,y)
            end
            task.wait(0.15)
        until BattleGui:FindFirstChild("Move1",true)

        -- Assign move buttons
        clickMove1 = BattleGui:FindFirstChild("Move1",true) and BattleGui:FindFirstChild("Move1",true):FindFirstChild("Button")
        clickMove2 = BattleGui:FindFirstChild("Move2",true) and BattleGui:FindFirstChild("Move2",true):FindFirstChild("Button")
        clickMove3 = BattleGui:FindFirstChild("Move3",true) and BattleGui:FindFirstChild("Move3",true):FindFirstChild("Button")
        clickMove4 = BattleGui:FindFirstChild("Move4",true) and BattleGui:FindFirstChild("Move4",true):FindFirstChild("Button")

        -- Wait short moment to ensure PP frames are created
        task.wait(0.05)

        -- Main battle loop
        while not ended and not eject do
            task.wait(0.08)

            pcall(function()
                Main.Battle.currentBattle.fastForward = true
            end)

            -- Update PP
            local m1c,m1m = getPP(BattleGui:FindFirstChild("Move1",true))
            local m2c,m2m = getPP(BattleGui:FindFirstChild("Move2",true))
            local m3c,m3m = getPP(BattleGui:FindFirstChild("Move3",true))
            local m4c,m4m = getPP(BattleGui:FindFirstChild("Move4",true))
            PPLbl1:Set(string.format("Move 1 - %d/%d", m1c, m1m))
            PPLbl2:Set(string.format("Move 2 - %d/%d", m2c, m2m))
            PPLbl3:Set(string.format("Move 3 - %d/%d", m3c, m3m))
            PPLbl4:Set(string.format("Move 4 - %d/%d", m4c, m4m))

            -- If fight button still there and move buttons not visible -> press fight
            if clickFight and clickFight.Parent and clickFight.Parent.Visible and (not clickMove1 or not clickMove1.Parent.Visible) then
                local x,y = getCenterPosition(clickFight)
                virtualClick(x,y)
                task.wait(0.12)
                continue
            end

            -- Choose move based on effectiveness then PP
            local moveCandidates = {}
            for _,m in ipairs(selectedMoves) do
                local moveBtn = nil
                local moveGui = BattleGui:FindFirstChild("Move"..m,true)
                if m == 1 then moveBtn = clickMove1
                elseif m == 2 then moveBtn = clickMove2
                elseif m == 3 then moveBtn = clickMove3
                elseif m == 4 then moveBtn = clickMove4
                end
                local curPP,_ = getPP(moveGui)
                local effTier = getEffectivenessTier(moveGui)
                if curPP > 0 then
                    table.insert(moveCandidates,{btn=moveBtn, pp=curPP, eff=effTier})
                end
            end

            -- Sort candidates: effectiveness first, then PP
            table.sort(moveCandidates,function(a,b)
                if a.eff ~= b.eff then
                    return a.eff < b.eff
                else
                    return a.pp > b.pp
                end
            end)

            -- Click the chosen move 3 times
            if #moveCandidates > 0 then
                for i=1,3 do
                    local bx,by = getCenterPosition(moveCandidates[1].btn)
                    virtualClick(bx,by)
                    task.wait(0.05)
                end
            else
                -- No usable move -> attempt to run
                if clickRun then
                    local rx,ry = getCenterPosition(clickRun)
                    virtualClick(rx,ry)
                    task.wait(0.12)
                elseif restFight then
                    local rx,ry = getCenterPosition(restFight)
                    virtualClick(rx,ry)
                    task.wait(0.12)
                end
                break
            end

            -- Break if battle GUI disappears
            if not PlayerGui:FindFirstChild("MainGui", true) or not PlayerGui:FindFirstChild("MainGui", true):FindFirstChild("BattleGui", true) then
                break
            end
        end

        -- Wait for battle to end
        repeat
            task.wait(0.1)
        until not PlayerGui:FindFirstChild("MainGui", true) or not PlayerGui:FindFirstChild("MainGui", true):FindFirstChild("BattleGui", true)

        LastBattleDetected = false
    end
end)
