-- Button creator
local function createButton(text, sizeY)
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(0.95, 0, sizeY, 0)
    button.BackgroundColor3 = Color3.fromRGB(200, 60, 60)
    button.Text = text
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.Font = Enum.Font.GothamBold
    button.TextScaled = true
    button.Parent = Container
    
    local corner = Instance.new("UICorner")
    corner.Parent = button
    
    return button
end

-- Textbox creator

end

local hitboxButton = createButton("HITBOX EXPANDER: OFF", 0.12)

-- Open/close menu
OpenBtn.MouseButton1Click:Connect(function()
    MainFrame.Visible = not MainFrame.Visible
end)

-- Paintball spam
pbSpamButton.MouseButton1Click:Connect(function()
    pbSpamButton.Text = "PAINTBALL SPAM: ON"
    pbSpamButton.BackgroundColor3 = Color3.fromRGB(60, 200, 60)
    
    RunService.Heartbeat:Connect(function()
        local backpack = LocalPlayer:FindFirstChild("Backpack")
        local character = LocalPlayer.Character
        
        if backpack and character then
            -- Put all tools in backpack
            for _, tool in ipairs(character:GetChildren()) do
                if tool:IsA("Tool") then
                    tool.Parent = backpack
                end
            end
            
            -- Get paintball gun
            local paintball = backpack:FindFirstChild("Paintball Gun")
            if paintball then
                paintball.Parent = character
            end
        end
    end)
end)

-- Anti-knockback
antiKnockButton.MouseButton1Click:Connect(function()
    antiKnockButton.Text = "ANTI-KNOCKBACK: ON"
    antiKnockButton.BackgroundColor3 = Color3.fromRGB(60, 200, 60)
    
    RunService.Heartbeat:Connect(function()
        local character = LocalPlayer.Character
        if character then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                -- Reset velocity for anti-knockback
                local root = character:FindFirstChild("HumanoidRootPart")
                if root then
                    root.Velocity = Vector3.new(0, root.Velocity.Y, 0)
                    root.AssemblyLinearVelocity = Vector3.new(0, root.AssemblyLinearVelocity.Y, 0)
                end
            end
        end
    end)
end)

-- Hitbox expander
hitboxButton.MouseButton1Click:Connect(function()
    hitboxButton.Text = "HITBOX EXPANDER: ON"
    hitboxButton.BackgroundColor3 = Color3.fromRGB(60, 200, 60)
    
    local size = tonumber(hitboxInput.Text) or 10
    
    RunService.RenderStepped:Connect(function()
        for _, player in pairs(Players:GetPlayers()) do
            local character = player.Character
            if character and character ~= LocalPlayer.Character then
                local root = character:FindFirstChild("HumanoidRootPart")
                if root then
                    root.Size = Vector3.new(size, size, size)
                    root.Transparency = 0.7
                    root.CanCollide = false
                end
            end
        end
    end)
end)

-- Reset hitbox size
hitboxInput.FocusLost:Connect(function()
    if hitboxInput.Text == "" then
        hitboxInput.Text = "10"
    end
end)

-- Speed hack
speedButton.MouseButton1Click:Connect(function()
    speedButton.Text = "SPEED: ON"
    speedButton.BackgroundColor3 = Color3.fromRGB(60, 200, 60)
    
    local speed = tonumber(speedInput.Text) or 25
    
    RunService.Stepped:Connect(function()
        local character = LocalPlayer.Character
        if character then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                humanoid.WalkSpeed = speed
            end
        end
    end)
end)

-- Reset speed value
speedInput.FocusLost:Connect(function()
    if speedInput.Text == "" then
        speedInput.Text = "25"
    end
end)

-- Infinite jump
infJumpButton.MouseButton1Click:Connect(function()
    infJumpButton.Text = "INF JUMP: ON"
    infJumpButton.BackgroundColor3 = Color3.fromRGB(60, 200, 60)
    
    UserInputService.JumpRequest:Connect(function()
        local character = LocalPlayer.Character
        if character then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
            end
        end
    end)
end)

-- Auto bat toggle
autoBatButton.MouseButton1Click:Connect(function()
    if autoBatButton.Text == "AUTO BAT (X): ON" then
        autoBatButton.Text = "AUTO BAT (X): OFF"
        autoBatButton.BackgroundColor3 = Color3.fromRGB(60, 30, 30)
    else
        autoBatButton.Text = "AUTO BAT (X): ON"
        autoBatButton.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    end
end)

-- Close menu when clicking outside
UserInputService.InputBegan:Connect(function(input, processed)
    if not processed and input.UserInputType == Enum.UserInputType.MouseButton1 then
        local mousePos = UserInputService:GetMouseLocation()
        local framePos = MainFrame.AbsolutePosition
        local frameSize = MainFrame.AbsoluteSize
        
        if MainFrame.Visible then
            if mousePos.X < framePos.X or mousePos.X > framePos.X + frameSize.X or
               mousePos.Y < framePos.Y or mousePos.Y > framePos.Y + frameSize.Y then
                MainFrame.Visible = false
            end
        end
    end
end)
