local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/TheMugenKingMods/Gui-Library/refs/heads/main/Gerad's"))()

local Window = Library:CreateWindow('ArceusXArchive') -- :CreateWindow(Title)

local Section = Window:Section('Slash') -- :Section(Title)

Section:Button('Click', function() -- :Button(Title, callback)
    local Players = game:GetService("Players")
local Event = Players.LocalPlayer.Character.sl.Event
local newVector = Vector3.new(999, 0, 999)

for _, plr in ipairs(Players:GetPlayers()) do
    if plr ~= Players.LocalPlayer then
        local targetChar = plr.Character
        if targetChar then
            Event:FireServer("slash", targetChar, newVector)
        end
    end
end
end)

local Section = Window:Section('Door') -- :Section(Title)

Section:Button('Click', function() -- :Button(Title, callback)
    local d1 = workspace:FindFirstChild("door1")
if d1 then
    d1:Destroy()
end
end)

local Section = Window:Section('Reload') -- :Section(Title)

Section:Button('Click', function() -- :Button(Title, callback)
    -- Services
local Players = game:GetService("Players")

-- Remote
local reload = Players.LocalPlayer.Character.gun.Model.reload

-- Loop reload
while true do
    pcall(function()
        reload:FireServer()
    end)
    wait(0.05) -- delay 0.05 detik (bisa dibuat lebih kecil)
end
end)
