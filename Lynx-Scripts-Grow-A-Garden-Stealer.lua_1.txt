local function HopToOldServer()
    hopBtn.Text = "Searching old servers..."
    hopBtn.BackgroundColor3 = Color3.fromRGB(255, 180, 60)
    local servers = {}
    local cursor = nil
    for i = 1, 30 do
        local url = "https://games.roblox.com/v1/games/"..PlaceId.."/servers/Public?sortOrder=Asc&limit=100"
        if cursor then url = url .. "&cursor=" .. HttpService:UrlEncode(cursor) end
        local success, body = pcall(function() return game:HttpGet(url, true) end)
        if not success then task.wait(2) continue end
        local data = HttpService:JSONDecode(body)
        if data and data.data then
            for _, server in ipairs(data.data) do
                if server.id ~= JobId and server.playing <= 2 and (server.maxPlayers - server.playing) >= 2 then
                    table.insert(servers, server.id)
                end
            end
            cursor = data.nextPageCursor  -- ← Now correctly placed AFTER the loop
        end
        task.wait(1)
        if #servers >= 10 then break end
    end
    if #servers > 0 then
        local target = servers[math.random(#servers)]
        hopBtn.Text = "Joining server..."
        TeleportService:TeleportToPlaceInstance(PlaceId, target, Players.LocalPlayer)
    else
        hopBtn.Text = "No servers found – Try again"
        hopBtn.BackgroundColor3 = Color3.fromRGB(255, 80, 80)
        task.wait(3)
        hopBtn.Text = "Hop to Old Server"
        hopBtn.BackgroundColor3 = Color3.fromRGB(70, 140, 255)
    end
end -- ← THIS END WAS MISSING! ADD IT!
