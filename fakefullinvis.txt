-- Função para fazer partes caírem com física (sem Motor6D/Ragdoll)
local function makePartsFall()
	-- Salvar posições originais ANTES de cair
	originalCFrames = {}
	for _, part in pairs(character:GetDescendants()) do
		if part:IsA("BasePart") and part ~= hrp then
			originalCFrames[part] = part.CFrame -- Salvar posição ANTES
		end
	end
	
	-- Criar WeldConstraints temporários para cada parte com HRP
	local welds = {}
	for _, part in pairs(character:GetDescendants()) do
		if part:IsA("BasePart") and part ~= hrp and part.Parent == character then
			-- Remover conexões existentes
			for _, child in pairs(part:GetChildren()) do
				if child:IsA("Weld") or child:IsA("WeldConstraint") or child:IsA("Motor6D") then
					child:Destroy()
				end
			end
			
			-- Ativar física
			part.CanCollide = false
			part.Massless = false
			part.CustomPhysicalProperties = PhysicalProperties.new(0.7, 0.3, 0.5)
			
			-- Aplicar força de gravidade artificial para simular queda
			local bodyForce = Instance.new("BodyForce")
			bodyForce.Force = Vector3.new(0, -part:GetMass() * 196.2, 0) -- Gravidade aumentada
			bodyForce.Parent = part
			table.insert(welds, bodyForce)
			
			-- Adicionar velocidade inicial aleatória para movimento natural
			part.AssemblyLinearVelocity = Vector3.new(
				math.random(-5, 5),
				math.random(-10, 0),
				math.random(-5, 5)
			)
		end
	end
	
	humanoid.AutoRotate = false
	humanoid.PlatformStand = true
	
	return welds
end

-- Função para fazer partes invisíveis PERMANENTEMENTE (PARTES DO MAPA SEM COLISÃO)
local function makePartsInvisible(welds)
	-- Remover BodyForces
	for _, force in pairs(welds) do
		if force and force.Parent then
			force:Destroy()
		end
	end
	
	-- Tornar partes do WORKSPACE (MAPA) sem colisão invisíveis
	local invisibleParts = {}
	for _, part in pairs(workspace:GetDescendants()) do
		-- Ignorar partes do personagem
		if part:IsA("BasePart") and not part:IsDescendantOf(character) and part.CanCollide == false and part.Transparency < 0.95 then
			part.Transparency = 1
			table.insert(invisibleParts, part)
		end
	end
	
	-- Também processar MeshParts do mapa e remover texturas
	for _, part in pairs(workspace:GetDescendants()) do
		if part:IsA("MeshPart") and not part:IsDescendantOf(character) and part.CanCollide == false and part.Transparency < 0.95 then
			part.TextureID = ""
			part.Transparency = 1
		end
	end
	
	-- Tornar Decals e Textures invisíveis em partes do mapa sem colisão
	for _, obj in pairs(workspace:GetDescendants()) do
		if (obj:IsA("Decal") or obj:IsA("Texture")) and obj.Parent and obj.Parent:IsA("BasePart") then
			if not obj:IsDescendantOf(character) and obj.Parent.CanCollide == false and obj.Transparency < 0.95 then
				obj.Transparency = 1
			end
		end
	end
	
	print("Tornando " .. #invisibleParts .. " partes do MAPA sem colisão invisíveis")
end

local function respawnar(plr)
	local rcdEnabled, wasHidden = false, false
	if getgenv().gethidden then
		rcdEnabled = getgenv().gethidden(workspace, 'RejectCharacterDeletions') ~= Enum.RejectCharacterDeletions.Disabled
	end

	if rcdEnabled and getgenv().replicatesignal then
		getgenv().replicatesignal(plr.ConnectDiedSignalBackend)
		task.wait(Players.RespawnTime - 0.1)
		getgenv().replicatesignal(plr.Kill)
	else
		local char = plr.Character
		local hum = char:FindFirstChildWhichIsA('Humanoid')
		if hum then
			hum:ChangeState(Enum.HumanoidStateType.Dead)
		end
		char:ClearAllChildren()
		local newChar = Instance.new('Model')
		newChar.Parent = workspace
		plr.Character = newChar
		task.wait()
		plr.Character = char
		newChar:Destroy()
	end
end

local function applyDesync()
	-- Aplicar FFlags
	for name, value in pairs(FFlags) do
		pcall(function()
			if setfflag then
				setfflag(tostring(name), tostring(value))
			end
		end)
	end
	
	-- Respawnar sem mexer nas animações
	respawnar(player)
	
	print("Desync aplicado com sucesso!")
end
local function customMovement()
	-- Desconectar conexão anterior se existir
	if moveConnection then
		moveConnection:Disconnect()
		moveConnection = nil
	end
	
	if speedEnabled then
		-- Criar nova conexão que continua indefinidamente
		moveConnection = RunService.Heartbeat:Connect(function()
			if humanoid and hrp and humanoid.MoveDirection.Magnitude > 0 then
				local moveDir = humanoid.MoveDirection
				-- Speed de 28.3 usando Velocity
				local newVelocity = Vector3.new(moveDir.X * 28.3, hrp.Velocity.Y, moveDir.Z * 28.3)
				hrp.AssemblyLinearVelocity = newVelocity
			end
		end)
	end
end

-- Sistema de detecção de ProximityPrompt
local function setupProximityPromptDetection()
	local connection
	
	connection = ProximityPromptService.PromptButtonHoldEnded:Connect(function(prompt, playerWhoTriggered)
		if playerWhoTriggered == player and fullInvisEnabled and not isProcessing then
			isProcessing = true
			
			-- Atualizar referências do personagem
			character = player.Character or player.CharacterAdded:Wait()
			humanoid = character:WaitForChild("Humanoid")
			hrp = character:WaitForChild("HumanoidRootPart")
			
			-- 1. Fazer partes caírem com física (sem Motor6D/Ragdoll)
			local welds = makePartsFall()
			
			-- 2. Após 2 segundos, tornar partes sem colisão do MAPA invisíveis PERMANENTEMENTE
			task.wait(2)
			makePartsInvisible(welds)
			
			-- 3. Após 3 segundos adicionais (total 5s), restaurar humanoid
			task.wait(3)
			restoreHumanoid()
			
			isProcessing = false
		end
	end)
	
	return connection
end
