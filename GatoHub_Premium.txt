LabsUI:CreateWindow({
    Title = "GatoHub Premium 🐱",
    Theme = "Dark",
    Size = UDim2.new(0, 520, 0, 360),
    SaveConfig = true
})

-- 🧠 Aba 1: Auto Farm
local farmTab = LabsUI:AddTab("Auto Farm", "rbxassetid://3926305904")

farmTab:AddButton("Ativar Auto Farm", function()
    print("⚙️ Auto Farm ativado!")
end)

farmTab:AddButton("Desativar Auto Farm", function()
    print("🛑 Auto Farm desativado!")
end)

farmTab:AddToggle("Modo rápido", false, function(state)
    print("🚀 Modo rápido:", state)
end)

farmTab:AddSlider("Velocidade", 10, 100, 50, function(value)

end)

local configTab = LabsUI:AddTab("Configurações", "rbxassetid://3926305904")

configTab:AddDropdown("Tema", {"Dark", "Light", "Ruby", "Aqua", "Mocha", "Jester", "Gold", "Ghost"}, "Dark", function(theme)
    LabsUI:SetTheme(theme)
    print("🎨 Tema alterado para:", theme)
end)

configTab:AddButton("Salvar Configuração", function()
    LabsUI:SaveConfig()
    print("💾 Configurações salvas!")
