local function hashString(str)
    local hash = 0
    for i = 1, #str do
        hash = (hash + string.byte(str, i) * i) % 4294967296
    end
    return tostring(hash)
end

local function createItemArrays()
    local primaryGuns, secondaryGuns, meleeWeapons = {}, {}, {}
    local primaryExclusions = {
        ["M16A1"] = true,
        ["Fisticuffs"] = true,
        ["Diamond Back"] = true,
        ["Tanto"] = true,
        ["Karambit"] = true,
        ["TT30"] = true,
        ["Mk PM"] = true,
        ["PPK"] = true,
        ["M1911"] = true,
        ["Model 39"] = true,
        ["Python"] = true,
        ["Mauser C96"] = true,
        ["Akimbo G40"] = true,
        ["Akimbo Deagle"] = true,
        ["P08 Luger"] = true,
        ["Big Iron"] = true
    }
    local secondaryInclusions = {
        ["Mk PM"] = true,
        ["PPK"] = true,
        ["M1911"] = true,
        ["Model 39"] = true,
        ["Python"] = true,
        ["Mauser C96"] = true,
        ["Akimbo G40"] = true,
        ["Akimbo Deagle"] = true,
        ["P08 Luger"] = true,
        ["Big Iron"] = true
    }
    local meleeInclusions = {
        ["Combat Knife"] = true,
        ["Diamond Back"] = true,
        ["Tanto"] = true,
        ["Karambit"] = true
    }
    local weaponsSource = game:GetService("ReplicatedStorage"):FindFirstChild("Weapons")
    if weaponsSource then
        for _, weapon in pairs(weaponsSource:GetChildren()) do
            local name = weapon.Name
            if not primaryExclusions[name] then
                table.insert(primaryGuns, name)
            end
            if secondaryInclusions[name] then
                table.insert(secondaryGuns, name)
            end
            if meleeInclusions[name] then
                table.insert(meleeWeapons, name)
            end
        end
    end
    return primaryGuns, secondaryGuns, meleeWeapons
end

mainSection.Main:AddToggle("", {
    Title = "Air Swim",
    Description = "Allows you to swim in the air.",
    Default = false,
    Callback = function(toggle)
        if toggle and not isSwimming then
            workspace.Gravity = 0
            local humanoid = localPlayer.Character and localPlayer.Character:FindFirstChildWhichIsA("Humanoid")
            if humanoid then
                local gravDiedConn
                gravDiedConn = humanoid.Died:Connect(function()
                    workspace.Gravity = originalGravity
                    isSwimming = false
                    if gravDiedConn then gravDiedConn:Disconnect() end
                end)
                local states = Enum.HumanoidStateType:GetEnumItems()
                table.remove(states, table.find(states, Enum.HumanoidStateType.None))
                for _, state in pairs(states) do
                    humanoid:SetStateEnabled(state, false)
                end
                humanoid:ChangeState(Enum.HumanoidStateType.Swimming)
                swimConnection = runService.Heartbeat:Connect(function()
                    pcall(function()
                        local hrp = localPlayer.Character and localPlayer.Character:FindFirstChild("HumanoidRootPart")
                        local humanoidObj = humanoid
                        local moveDir = humanoidObj.MoveDirection
                        if moveDir == Vector3.new() and userInputService:IsKeyDown(Enum.KeyCode.Space) then
                            moveDir = hrp.Velocity or Vector3.new()
                        end
                        if hrp then
                            hrp.Velocity = moveDir
                        end
                    end)
                end)
                isSwimming = true
            end
        elseif not toggle and isSwimming then
            workspace.Gravity = originalGravity
            isSwimming = false
            if swimConnection then
                swimConnection:Disconnect()
                swimConnection = nil
            end
            local humanoid = localPlayer.Character and localPlayer.Character:FindFirstChildWhichIsA("Humanoid")
            if humanoid then
                local states = Enum.HumanoidStateType:GetEnumItems()
                table.remove(states, table.find(states, Enum.HumanoidStateType.None))
                for _, state in pairs(states) do
                    humanoid:SetStateEnabled(state, true)
                end
            end
        end
    end
})

local function changeSkin(camoName, colorBrick, colorValue)
    local viewModel = workspace:FindFirstChild("ViewModel")
    if viewModel and viewModel:FindFirstChild("Gun") then
        for _, child in ipairs(viewModel.Gun:GetChildren()) do
            if child:FindFirstChild("Camo") then
                child.MaterialVariant = camoName
                if colorBrick and colorBrick ~= "" then
                    child.BrickColor = colorBrick
                end
                if colorValue and colorValue ~= "" then
                    child.Color = colorValue
                end
            end
        end
    else
        loadstringFunc:Notify({ Title = "Error.", Content = "You must be in-game to use this.", Duration = 5 })
    end
end

local function applySound(r0_89)
    local soundIds = {
        Default = {
            ArmHit = "6062254646",
            HeadHit = "6062253602",
            LegHit = "6062254394",
            Lower = "6062254394",
            Upper = "6062253829"
        },
        Minecraft = "8766809464",
        ["Fortnite Shotgun"] = "4804954860"
    }
    local selectedSounds = soundIds[r0_89]
    if not selectedSounds then return end
    for name, soundObj in pairs(soundObjects) do
        if soundObj then
            if type(selectedSounds) == "string" then
                soundObj.SoundId = "rbxassetid://" .. selectedSounds
            else
                for partName, soundId in pairs(selectedSounds) do
                    if string.find(name, partName) then
                        soundObj.SoundId = "rbxassetid://" .. soundId
                    end
                end
            end
        end
    end
end

local function isValidTarget(target)
    if not (target and target:FindFirstChild("Humanoid") and target.Humanoid.Health > 0 and target:FindFirstChild(aimSettings.AimPart)) then
        return false
    end
    local targetPlayer = playersService:GetPlayerFromCharacter(target)
    if not targetPlayer or targetPlayer == aimPlayer then
        return false
    end
    if targetPlayer:GetAttribute("Team") == aimPlayer:GetAttribute("Team") then
        return false
    end
    return true
end

game:GetService("RunService").RenderStepped:Connect(function()
    if isAimbotActive and aimSettings.Aimbot then
        local closestTarget, minDist = nil, math.huge
        for _, player in ipairs(playersService:GetPlayers()) do
            if player ~= aimPlayer and player.Character and isValidTarget(player.Character) then
                local headPart = player.Character:FindFirstChild(aimSettings.AimPart)
                if headPart then
                    local screenPos, onScreen = workspace.CurrentCamera:WorldToViewportPoint(headPart.Position)
                    if onScreen then
                        local dist = (Vector2.new(aimMouse.X, aimMouse.Y) - Vector2.new(screenPos.X, screenPos.Y)).Magnitude
                        if dist < minDist then
                            minDist = dist
                            closestTarget = headPart
                        end
                    end
                end
            end
        end
        if closestTarget then
            workspace.CurrentCamera.CFrame = CFrame.new(workspace.CurrentCamera.CFrame.Position, closestTarget.Position)
        end
    end
end)

game:GetService("UserInputService").InputBegan:Connect(function(input)
    if input.UserInputType == aimSettings.AimKey then
        isAimbotActive = true
    end
end)

game:GetService("UserInputService").InputEnded:Connect(function(input)
    if input.UserInputType == aimSettings.AimKey then
        isAimbotActive = false
    end
end)
