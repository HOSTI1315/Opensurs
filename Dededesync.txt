local FreezeModeEnabled = false
local OriginalState = {}

local function GetCharacter(player)
    return player.Character
end

local function GetRoot(player)
    local char = GetCharacter(player)
    return char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("Torso"))
end

local function GetHumanoid(player)
    local char = GetCharacter(player)
    return char and char:FindFirstChildOfClass("Humanoid")
end

local function EnableFreezeWalk()
    local success, err = pcall(function()
        local Root = GetRoot(LocalPlayer)
        local Character = GetCharacter(LocalPlayer)
        local Humanoid = GetHumanoid(LocalPlayer)
        
        if not Root or not Character or not Humanoid then return end
        
        if Humanoid.RigType == Enum.HumanoidRigType.R6 then
            local ClonedRoot = Root:Clone()
            Root:Destroy()
            ClonedRoot.Parent = Character
            OriginalState.RigType = "R6"
        else
            Character.LowerTorso.Anchored = true
            local RootJoint = Character.LowerTorso:FindFirstChild("Root")
            if RootJoint then
                RootJoint:Destroy()
            end
            OriginalState.RigType = "R15"
        end
        
        FreezeModeEnabled = true
    end)
    
    return success
end

local function DisableFreezeWalk()
    local success, err = pcall(function()
        local Character = GetCharacter(LocalPlayer)
        if not Character then return end
        
        if OriginalState.RigType == "R15" then
            local LowerTorso = Character:FindFirstChild("LowerTorso")
            if LowerTorso then
                LowerTorso.Anchored = false
            end
        end
        
        LocalPlayer.Character = nil
        LocalPlayer.Character = Character
        
        FreezeModeEnabled = false
        OriginalState = {}
    end)
    
    return success
end

ToggleButton.MouseButton1Click:Connect(function()
    if FreezeModeEnabled then
        DisableFreezeWalk()
    else
        EnableFreezeWalk()
    end
    UpdateUI(FreezeModeEnabled)
end)

local dragging = false
local dragInput, mousePos, framePos

MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        mousePos = input.Position
        framePos = MainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - mousePos
        MainFrame.Position = UDim2.new(
            framePos.X.Scale,
            framePos.X.Offset + delta.X,
            framePos.Y.Scale,
            framePos.Y.Offset + delta.Y
        )
    end
end)

ScreenGui.Parent = PlayerGui

UpdateUI(false)
