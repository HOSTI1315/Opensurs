-- Desync Function
DesyncButton.MouseButton1Click:Connect(function()
    -- UI Feedback
    StatusLabel.Text = "Status: ON"
    StatusLabel.TextColor3 = Color3.fromRGB(220, 220, 220)
    
    TweenService:Create(StatusLabel, TweenInfo.new(0.3), {TextColor3 = Color3.fromRGB(220, 220, 220)}):Play()
    TweenService:Create(DesyncButton, TweenInfo.new(0.3), {
        TextColor3 = Color3.fromRGB(255, 255, 255),
        BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    }):Play()

    -- Core Desync Logic
    pcall(function()
        local Character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
        local Humanoid = Character:WaitForChild("Humanoid")
        local HRP = Character:WaitForChild("HumanoidRootPart")

        -- Tool activation sequence
        Humanoid:UnequipTools()
        local Tool = Backpack:FindFirstChild("Quantum Cloner")
        if Tool then
            Humanoid:EquipTool(Tool)
            Tool:Activate()
            task.wait(0.5)
            ReplicatedStorage.Packages.Net["RE/QuantumCloner/OnTeleport"]:FireServer()
        end

        -- WorldStep flag
        setfflag("WorldStepMax", "-99999999999999")

        -- First CFrame spam block
        task.spawn(function()
            pcall(function()
                setfflag("WorldStepsOffsetAdjustRate", "-1")
                task.wait(1)
                setfflag("WorldStepsOffsetAdjustRate", "60")
                
                local baseCFrame1 = HRP.CFrame * CFrame.new(-5, 0, 15)
                for i = 1, 98 do
                    HRP.CFrame = baseCFrame1
                end
            end)
            
            task.wait(1)
            setfflag("WorldStepsOffsetAdjustRate", "-9999999999")
            task.wait(1)
            setfflag("WorldStepsOffsetAdjustRate", "-9999999999")

            -- Second CFrame spam block
            pcall(function()
                local baseCFrame2 = HRP.CFrame * CFrame.new(2, 0, 6)
                for i = 1, 98 do
                    HRP.CFrame = baseCFrame2
                end
            end)
            
            task.wait(2)
            setfflag("WorldStepsOffsetAdjustRate", "-1")
        end)
    end)
    
    wait(1)
end)

-- Character respawn handling
LocalPlayer.CharacterAdded:Connect(function()

    end)
end)
