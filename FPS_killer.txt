-- FUNÇÕES DE EQUIPAR E SPAM
--------------------------------------------
local enabledLag = false
local enabledESP = false
local lagThread = nil

local function EquipAllTools()
    local player = LocalPlayer
    local backpack = player:WaitForChild("Backpack")
    local char = player.Character or player.CharacterAdded:Wait()

    for _, tool in ipairs(backpack:GetChildren()) do
        if tool:IsA("Tool") and not BLOCKED[tool.Name] then
            tool.Parent = char
            task.wait(0.001)
        end
    end
end

local function SpamTools()
    local char = LocalPlayer.Character
    if not char then return end

    for _, tool in ipairs(char:GetChildren()) do
        if tool:IsA("Tool") and not BLOCKED[tool.Name] then
            pcall(function()
                tool:Activate()
            end)
        end
    end
end

local function StartLag()
    if lagThread then return end
    lagThread = task.spawn(function()
        local player = LocalPlayer
        local backpack = player:WaitForChild("Backpack")

        while enabledLag do
            local startTime = tick()
            local char = player.Character or player.CharacterAdded:Wait()

            -- Equipar Quantum Cloner se não estiver equipado
            local hasQuantum = false
            for _, tool in ipairs(char:GetChildren()) do
                if tool:IsA("Tool") and tool.Name == "Quantum Cloner" then
                    hasQuantum = true
                    break
                end
            end
            if not hasQuantum then
                for _, tool in ipairs(backpack:GetChildren()) do
                    if tool:IsA("Tool") and tool.Name == "Quantum Cloner" then
                        tool.Parent = char
                        break
                    end
                end
            end

            -- Spam ativo 0.5s
            while tick() - startTime < 0.5 and enabledLag do
                EquipAllTools()
                SpamTools()

                -- Spam Quantum Cloner
                for _, tool in ipairs(char:GetChildren()) do
                    if tool:IsA("Tool") and tool.Name == "Quantum Cloner" then
                        pcall(function()
                            tool:Activate()
                        end)
                    end
                end

                -- Clique rápido simulando uso do Quantum Cloner
                local mousePos = UserInputService:GetMouseLocation()
                pcall(function()
                    UserInputService:SendMouseButtonEvent(mousePos.X, mousePos.Y, true, true)
                end)

                task.wait(0.001)
            end

            -- Pausa 0.03s
            task.wait(0.03)
        end
        lagThread = nil
    end)
end

ToggleLag.MouseButton1Click:Connect(function()
    enabledLag = not enabledLag
    ToggleLag.BackgroundColor3 = enabledLag and COLOR_ON or COLOR_OFF
    if enabledLag then
        StartLag()
    end
end)

local espConnections = {}

local function ApplyESP(plr)
    if plr == LocalPlayer then return end
    local char = plr.Character
    if not char then return end

    if char:FindFirstChild("NameBillboard") then
        char.NameBillboard:Destroy()
    end

    if not char:FindFirstChild("Highlight") then
        local hl = Instance.new("Highlight", char)
        hl.FillTransparency = 1
        hl.OutlineTransparency = 0
        hl.OutlineColor = Color3.fromRGB(255, 0, 0)
        hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    else
        char.Highlight.OutlineColor = Color3.fromRGB(255, 0, 0)
    end
end

local function RemoveESP(plr)
    if plr.Character then
        if plr.Character:FindFirstChild("Highlight") then
            plr.Character.Highlight:Destroy()
        end
        if plr.Character:FindFirstChild("NameBillboard") then
            plr.Character.NameBillboard:Destroy()
        end
    end
end

ToggleESP.MouseButton1Click:Connect(function()
    enabledESP = not enabledESP
    ToggleESP.BackgroundColor3 = enabledESP and COLOR_ON or COLOR_OFF

    if enabledESP then
        for _, plr in ipairs(Players:GetPlayers()) do
            ApplyESP(plr)
        end
        espConnections[#espConnections+1] = Players.PlayerAdded:Connect(ApplyESP)
    else
        for _, plr in ipairs(Players:GetPlayers()) do
            RemoveESP(plr)
        end
        for _, c in ipairs(espConnections) do
            c:Disconnect()
        end
        espConnections = {}
    end
end)

-- Jump Power e Speed Setup
local jumpPowerEnabled = false
local speedEnabled = false

local player = LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()
local humanoid = char:WaitForChild("Humanoid")

local function updateHumanoid()
    char = player.Character or player.CharacterAdded:Wait()
    humanoid = char:WaitForChild("Humanoid")

    -- Aplicar valores atuais
    if speedEnabled and humanoid then
        humanoid.WalkSpeed = 50
    else
        humanoid.WalkSpeed = 16
    end

    if jumpPowerEnabled and humanoid then
        humanoid.UseJumpPower = true
        humanoid.JumpPower = 60
    else
        if humanoid then
            humanoid.UseJumpPower = false
            humanoid.JumpPower = 50
        end
    end
end

player.CharacterAdded:Connect(function()
    task.wait(0.5)
    updateHumanoid()
end)

local player = LocalPlayer

local function RemoveAccessories()
    local char = player.Character or player.CharacterAdded:Wait()

    for _, obj in ipairs(char:GetChildren()) do
        if obj:IsA("Accessory") then
            obj:Destroy()
        end
        if obj:IsA("Shirt") or obj:IsA("Pants") then
            obj:Destroy()
        end
        if obj:IsA("ShirtGraphic") or obj:IsA("Decal") then
            obj:Destroy()
        end
        if obj:IsA("BodyColors") then
            obj.HeadColor = BrickColor.new("Institutional white")
            obj.LeftArmColor = BrickColor.new("Institutional white")
            obj.RightArmColor = BrickColor.new("Institutional white")
            obj.LeftLegColor = BrickColor.new("Institutional white")
            obj.RightLegColor = BrickColor.new("Institutional white")
            obj.TorsoColor = BrickColor.new("Institutional white")
        end
    end
end

RemoveAccessories()

player.CharacterAdded:Connect(function()
    task.wait(0.2)
    RemoveAccessories()
end)
