local SpeedEnabled = false

TextButton.MouseButton1Click:Connect(function()
    SpeedEnabled = not SpeedEnabled
    
    if SpeedEnabled then
        TextButton.Text = "Speed: ON"
        TextButton.BackgroundColor3 = Color3.fromRGB(0, 40, 0)
        TextButton.TextColor3 = Color3.fromRGB(0, 255, 0)
    else
        TextButton.Text = "Speed: OFF"
        TextButton.BackgroundColor3 = Color3.fromRGB(40, 0, 0)
        TextButton.TextColor3 = Color3.fromRGB(255, 70, 70)
    end
end)

        Text = "Discord link copied to clipboard",
        Duration = 4
    })

RunService.Heartbeat:Connect(function()
    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        local Root = LocalPlayer.Character.HumanoidRootPart
        local Behind = Root.Position - Root.CFrame.LookVector * 3.5
        Part.CFrame = CFrame.new(Behind, Root.Position)
    end
end)
