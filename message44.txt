    if Character then
        local GlitchedSlap = Character:FindFirstChild("Glitched Slap")
        if GlitchedSlap then
            local Humanoid = Character.Humanoid
            Humanoid:EquipTool(GlitchedSlap)

            local endTime = tick()
            if endTime - startTime < 0.05 then
                GlitchedSlap:Activate()
            end
        end
    end
end)
