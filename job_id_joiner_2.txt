local function parseError(errorResult)
    local errorStr = tostring(errorResult)
    
    if errorStr:find("GameFull") or errorStr:find("game full") or errorStr:find("Full") then
        return "Game Full"
    elseif errorStr:find("GameEnded") or errorStr:find("game ended") then
        return "Game Ended"
    elseif errorStr:find("Flooded") or errorStr:find("flooded") then
        return "Flooded, idk"
    elseif errorStr:find("IsTeleporting") or errorStr:find("teleporting") then
        return "Already Teleporting"
    elseif errorStr:find("Unauthorized") or errorStr:find("unauthorized") then
        return "Unauthorized, ps probably"
    elseif errorStr:find("not part of the universe") then
        return "Invalid Server"
    elseif errorStr:find("failure") or errorStr:find("Failure") then
        return "Teleport Failure"
    elseif errorStr:find("timeout") or errorStr:find("Timeout") then
        return "Timeout"
    else
        local short = errorStr:sub(1, 25)
        if #errorStr > 25 then short = short .. "..." end
        return short
    end
end

local function updateAttemptStatus(attemptNum, totalAttempts, errorText)
    local text = string.format("Attempt %d/%d", attemptNum, totalAttempts)
    if errorText and errorText ~= "" then
        text = text .. string.format('  <font color="rgb(255,100,100)">%s</font>', errorText)
    end
    attemptStatusLabel.Text = text
    attemptStatusLabel.Visible = true
end

local function showStatusMessage(text, color)
    attemptStatusLabel.Text = text
    attemptStatusLabel.TextColor3 = color or Settings.StatusNormal
    attemptStatusLabel.Visible = true
end

local function hideAttemptStatus()
    attemptStatusLabel.Visible = false
    attemptStatusLabel.Text = ""
end

local function setJoiningState(joining)
    State.isJoining = joining
    joinButton.Text = joining and "..." or "Join"
    cancelButton.Visible = joining
    if not joining then
        State.lastError = nil
    end
end

local function attemptJoin()
    if State.isJoining then return end
    
    local jobId = jobIdBox.Text
    local attempts = tonumber(attemptsBox.Text) or Settings.DefaultAttempts
    local delay = tonumber(delayBox.Text) or Settings.DefaultDelay
    
    if jobId == "" or #jobId < Settings.MinJobIdLength then
        showStatusMessage("Invalid JobId", Settings.StatusError)
        task.delay(2, function()
            if attemptStatusLabel.Text == "Invalid JobId" then
                hideAttemptStatus()
            end
        end)
        return
    end
    
    attempts = math.max(1, math.min(attempts, 10000))
    delay = math.max(0, math.min(delay, 10))
    
    State.shouldCancel = false
    State.lastError = nil
    setJoiningState(true)
    hideAttemptStatus()
    
    task.spawn(function()
        for i = 1, attempts do
            if State.shouldCancel then
                updateAttemptStatus(i, attempts, "Cancelled")
                break
            end
            
            updateAttemptStatus(i, attempts, State.lastError)
            
            local success, result = pcall(function()
                TeleportService:TeleportToPlaceInstance(game.PlaceId, jobId, player)
            end)
            
            if not success then
                local parsedError = parseError(result)
                State.lastError = parsedError
                updateAttemptStatus(i, attempts, parsedError)
                
                if parsedError == "Game Ended" then
                    updateAttemptStatus(i, attempts, "Game Ended - Stopped")
                    break
                end
            end
            
            if i < attempts and not State.shouldCancel then
                task.wait(delay)
            end
        end
        
        if not State.shouldCancel and attemptStatusLabel.Text:find("Attempt") then
            local finalText = string.format("Failed %d/%d", attempts, attempts)
            if State.lastError then
                finalText = finalText .. string.format('  <font color="rgb(255,100,100)">%s</font>', State.lastError)
            end
            attemptStatusLabel.Text = finalText
        end
        
        task.delay(5, function()
            if not State.isJoining then
                hideAttemptStatus()
            end
        end)
        
        setJoiningState(false)
    end)
end

local function cancelJoin()
    if State.isJoining then
        State.shouldCancel = true
    end
end

joinButton.Activated:Connect(attemptJoin)
cancelButton.Activated:Connect(cancelJoin)

jobIdBox.FocusLost:Connect(function(enterPressed)
    if enterPressed then
        attemptJoin()
    end
end)

TeleportService.TeleportInitFailed:Connect(function(teleportPlayer, teleportResult, errorMessage)
    if teleportPlayer == player and State.isJoining then
        local parsedError = parseError(teleportResult)
        State.lastError = parsedError
        
        if parsedError == "Game Ended" then
            State.shouldCancel = true
        end
    end
end)

local function toggleUI()
    State.uiVisible = not State.uiVisible
    mainFrame.Visible = State.uiVisible
    
    if not State.uiVisible then
        hideAttemptStatus()
        if State.isJoining then
            cancelJoin()
        end
    end
end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Settings.ToggleKey then
        toggleUI()
    end
end)
