-- Notification
local function Notify(text, duration)
    local Notif = Instance.new("Frame")
    Notif.Size = UDim2.new(0, 260, 0, 60)
    Notif.Position = UDim2.new(1, 0, 1, -80)
    Notif.BackgroundColor3 = Color3.fromRGB(15, 15, 22)
    Notif.BackgroundTransparency = 0.1
    Notif.Parent = ScreenGui

    local NotifCorner = Instance.new("UICorner")
    NotifCorner.CornerRadius = UDim.new(0, 10)
    NotifCorner.Parent = Notif

    local NotifText = Instance.new("TextLabel")
    NotifText.Text = text
    NotifText.Size = UDim2.new(1, -20, 1, 0)
    NotifText.Position = UDim2.new(0, 10, 0, 0)
    NotifText.BackgroundTransparency = 1
    NotifText.TextColor3 = Color3.new(1, 1, 1)
    NotifText.Font = Enum.Font.GothamSemibold
    NotifText.TextSize = 16
    NotifText.Parent = Notif

    TweenService:Create(Notif, TweenInfo.new(0.4, Enum.EasingStyle.Quad), {Position = UDim2.new(1, -280, 1, -80)}):Play()

    task.delay(duration or 3, function()
        TweenService:Create(Notif, TweenInfo.new(0.4), {Position = UDim2.new(1, 0, 1, -80)}):Play()
        task.delay(0.4, function() Notif:Destroy() end)
    end)
end

-- Draggable
local dragging = false
local dragStart, startPos

TitleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = Window.Position
    end
end)

TitleBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement and dragging then
        local delta = input.Position - dragStart
        Window.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)
