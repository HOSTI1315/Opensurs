activateButton.Position = UDim2.new(0, 10, 0, 30)

activateButton.BorderSizePixel = 0
activateButton.Parent = mainFrame

local buttonCorner = Instance.new("UICorner")
buttonCorner.CornerRadius = UDim.new(0, 8)
buttonCorner.Parent = activateButton

activateButton.MouseButton1Click:Connect(function()
    local humanoid = player.Character:FindFirstChild("Humanoid")
    if not humanoid then return end

    local invisibilityCloak = backpack:FindFirstChild("Invisibility Cloak")
    local quantumCloner = backpack:FindFirstChild("Quantum Cloner")

    if invisibilityCloak and quantumCloner then
    
        humanoid:UnequipTools()
        task.wait(0.05)
        humanoid:EquipTool(invisibilityCloak)
        task.wait(0.05)
        invisibilityCloak:Activate()

        activateButton.Text = "Activated!"
        activateButton.BackgroundColor3 = Color3.fromRGB(0, 200, 0)
        
        task.wait(1)

        humanoid:EquipTool(quantumCloner)
        activateButton.Text = "Switched to Cloner!"
        quantumCloner:Activate()
        
        activateButton.Text = "Cloner Activated!"
        
        task.wait(0.3)
        activateButton.Text = "Activate Combo"
        activateButton.BackgroundColor3 = Color3.fromRGB(100, 50, 200)
    else
        activateButton.Text = "Missing Tools!"
        activateButton.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
        task.wait(1)
        activateButton.Text = "Activate Combo"
        activateButton.BackgroundColor3 = Color3.fromRGB(100, 50, 200)
    end
end)

activateButton.MouseEnter:Connect(function()
    activateButton.BackgroundColor3 = Color3.fromRGB(80, 40, 180)
end)

activateButton.MouseLeave:Connect(function()
    activateButton.BackgroundColor3 = Color3.fromRGB(100, 50, 200)
end)
