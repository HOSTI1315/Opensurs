local function toggleGodMode()
    godModeEnabled = not godModeEnabled
    
    if godModeEnabled then
        toggleButton.Position = UDim2.new(1, -24, 0.5, -11)
        toggleButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
        toggleFrame.BackgroundColor3 = Color3.fromRGB(138, 43, 226)
        toggleFrame.BackgroundTransparency = 0.3
        statusLabel.Text = " "
        statusLabel.TextColor3 = Color3.fromRGB(138, 226, 43)
        
        humanoid.MaxHealth = math.huge
        humanoid.Health = math.huge
        
        humanoid.HealthChanged:Connect(function(health)
            if godModeEnabled and health < humanoid.MaxHealth then
                humanoid.Health = humanoid.MaxHealth
            end
        end)
        
        local ReplicatedStorage = game:GetService("ReplicatedStorage")
        local packages = ReplicatedStorage:FindFirstChild("Packages")
        if packages then
            local net = packages:FindFirstChild("Net")
            if net then
                local kill = net:FindFirstChild("RE/TsunamiEventService/Kill")
                if kill then kill:Destroy() end
            end
        end
        
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanTouch = false
            end
        end
    else
        toggleButton.Position = UDim2.new(0, 2, 0.5, -11)
        toggleButton.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
        toggleFrame.BackgroundColor3 = Color3.fromRGB(60, 60, 70)
        toggleFrame.BackgroundTransparency = 0
        statusLabel.Text = " "
        statusLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
        
        humanoid.MaxHealth = 100
        humanoid.Health = 100
        
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanTouch = true
            end
        end
    end
    
    local tween = TweenService:Create(toggleButton, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {
        Position = godModeEnabled and UDim2.new(1, -24, 0.5, -11) or UDim2.new(0, 2, 0.5, -11)
    })
    tween:Play()
end

local function toggleEspTimer()
    espTimerEnabled = not espTimerEnabled
    
    if espTimerEnabled then
        espToggleButton.Position = UDim2.new(1, -24, 0.5, -11)
        espToggleButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
        espToggleFrame.BackgroundColor3 = Color3.fromRGB(138, 43, 226)

        espStatusLabel.Text = " "

        screenEspGui.Name = "TimerSecretESP"
        screenEspGui.ResetOnSpawn = false
        screenEspGui.IgnoreGuiInset = true
        screenEspGui.Parent = playerGui
        
        local frame = Instance.new("Frame", screenEspGui)
        frame.Size = UDim2.new(0.52, 0, 0, 58)
        frame.Position = UDim2.new(0.24, 0, 0, 6)
        frame.BackgroundColor3 = Color3.fromRGB(15, 15, 20)
        frame.BackgroundTransparency = 0.35
        frame.BorderSizePixel = 0
        
        local corner = Instance.new("UICorner", frame)
        corner.CornerRadius = UDim.new(0, 14)
        
        local stroke = Instance.new("UIStroke", frame)
        stroke.Color = Color3.fromRGB(138, 43, 226)
        stroke.Thickness = 2.2
        stroke.Transparency = 0.35
        
        local textLabel = Instance.new("TextLabel", frame)
        textLabel.Size = UDim2.new(0.96, 0, 0.92, 0)
        textLabel.Position = UDim2.new(0.02, 0, 0.04, 0)
        textLabel.BackgroundTransparency = 1
        textLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
        textLabel.TextScaled = true
        textLabel.Font = Enum.Font.LuckiestGuy
        textLabel.TextStrokeTransparency = 0.7
        textLabel.TextStrokeColor3 = Color3.fromRGB(50, 50, 80)
        textLabel.Text = "High Tier Secret Spawning in : ???"
        
        local pulse = TweenService:Create(textLabel, TweenInfo.new(2, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {
            TextTransparency = 0.05
        })
        pulse:Play()
        
        espConnection = RunService.RenderStepped:Connect(function()
            if not espTimerEnabled then return end
            
            local map = Workspace:FindFirstChild("Map")
            if not map then 
                textLabel.Text = "High Tier Secret Spawning in : Map introuvable"
                return 
            end
            
            local sign = map:FindFirstChild("SpawnTimerSign")
            if not sign then 
                textLabel.Text = "High Tier Secret Spawning in : Sign introuvable"
                return 
            end
            
            local panel = sign:FindFirstChild("Panel")
            if not panel then 
                textLabel.Text = "High Tier Secret Spawning in : Panel introuvable"
                return 
            end
            
            local surfaceGui = panel:FindFirstChildOfClass("SurfaceGui")
            if not surfaceGui then 
                textLabel.Text = "High Tier Secret Spawning in : SurfaceGui introuvable"
                return 
            end
            
            local mainFrameGui = surfaceGui:FindFirstChild("MainFrame")
            if not mainFrameGui then 
                textLabel.Text = "High Tier Secret Spawning in : MainFrame introuvable"
                return 
            end
            
            local timerLabel = mainFrameGui:FindFirstChild("Timer")
            if timerLabel and timerLabel:IsA("TextLabel") then
                textLabel.Text = "High Tier Secret Spawning in : " .. (timerLabel.Text or "???")
            else
                textLabel.Text = "High Tier Secret Spawning in : Timer introuvable"
            end
        end)
        
    else
        espToggleButton.Position = UDim2.new(0, 2, 0.5, -11)
        espToggleButton.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
        espToggleFrame.BackgroundColor3 = Color3.fromRGB(60, 60, 70)
        espToggleFrame.BackgroundTransparency = 0
        espStatusLabel.Text = " "
        espStatusLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
        
        if espConnection then
            espConnection:Disconnect()
            espConnection = nil
        end
        
        if screenEspGui then
            screenEspGui:Destroy()
            screenEspGui = nil
        end
        
        if pulse then pulse:Cancel() end
    end
    
    local tween = TweenService:Create(espToggleButton, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {
        Position = espTimerEnabled and UDim2.new(1, -24, 0.5, -11) or UDim2.new(0, 2, 0.5, -11)
    })
    tween:Play()
end

local isOpen = false

local function toggleMenu()
    isOpen = not isOpen
    
    if isOpen then
        mainFrame.Visible = true
        local tween = TweenService:Create(mainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
            Size = UDim2.new(0, 250, 0, 180)
        })
        tween:Play()
        pulseAnimation:Cancel()
        openStroke.Transparency = 0.5
    else
        local tween = TweenService:Create(mainFrame, TweenInfo.new(0.2, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
            Size = UDim2.new(0, 0, 0, 0)
        })
        tween:Play()
        tween.Completed:Connect(function()
            mainFrame.Visible = false
        end)
        pulseAnimation:Play()
    end
end

local dragging = false
local dragInput
local dragStart
local startPos

headerBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

headerBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)
