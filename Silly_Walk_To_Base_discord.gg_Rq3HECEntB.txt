Connection = TextButton_2.MouseEnter:Connect(function()
    -- Show border on hover
    local tween = TweenService:Create(UIStroke_2, TweenInfo.new(0.22, Enum.EasingStyle.Sine), {
        Transparency = 0.55,  -- Show border
    })
    tween:Play()  -- Play animation
end)

-- Button leave effect
local Connection_2
Connection_2 = TextButton_2.MouseLeave:Connect(function()
    -- Hide border
    local tween = TweenService:Create(UIStroke_2, TweenInfo.new(0.22, Enum.EasingStyle.Sine), {
        Transparency = 1,  -- Hide border
    })
    tween:Play()  -- Play animation
end)

-- Go button click
local Connection_3
Connection_3 = TextButton_2.MouseButton1Click:Connect(function()
    -- Button press animation
    local pressTween = TweenService:Create(TextButton_2, TweenInfo.new(0.12, Enum.EasingStyle.Sine), {
        Position = UDim2.new(0.73, 0, 0.2, 0),  -- Move slightly
        Size = UDim2.new(0.24, 0, 0.6, 0),  -- Shrink slightly
    })
    
    local releaseTween = TweenService:Create(TextButton_2, TweenInfo.new(0.12, Enum.EasingStyle.Sine), {
        Position = UDim2.new(0.71, 0, 0.18, 0),  -- Return position
        Size = UDim2.new(0.25, 0, 0.65, 0),  -- Return size
    })
    
    pressTween:Play()  -- Press animation
    pressTween.Completed:Wait()  -- Wait for completion
    releaseTween:Play()  -- Release animation
    
    -- Teleport logic
    local Character_4 = LocalPlayer.Character  -- Your character
    if not Character_4 then return end  -- Stop if no character
    Character_4:WaitForChild("Humanoid")  -- Wait for humanoid
    
    -- Create pathfinding
    local Path_2 = game:GetService("PathfindingService"):CreatePath({
        AgentCanJump = true,  -- Can jump
        WaypointSpacing = 6,  -- Waypoint distance
        AgentHeight = 5,  -- Agent height
        AgentRadius = 2,  -- Agent size
    })
    
    -- Calculate path to base (24 studs behind current position)
    Path_2:ComputeAsync(Character_4:WaitForChild("HumanoidRootPart").Position, CFrame * CFrame.new(0, 0, -24).Position)
    
    -- Check if path is valid
    if Path_2.Status == Enum.PathStatus.Success then
        -- Path found - teleport would happen here
    end
end)
