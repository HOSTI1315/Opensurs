Section:NewToggle("Auto Equip Shovel + Dig", "Equips Shovel and digs for 10 seconds reset", function(Value)
    getgenv().AutoEquipShovel = Value

    if Value then
        local Players = game:GetService("Players")
        local VirtualInputManager = game:GetService("VirtualInputManager")
        local Workspace = game:GetService("Workspace")
        local player = Players.LocalPlayer
        local backpack = player:WaitForChild("Backpack")

        local raycastParams = RaycastParams.new()
        raycastParams.FilterDescendantsInstances = {player.Character}
        raycastParams.FilterType = Enum.RaycastFilterType.Blacklist
        raycastParams.IgnoreWater = true

        local lastTeleportPos = nil

        local function tapScreenMultiple(times, delayBetween)
            delayBetween = delayBetween or 0.1
            for i = 1, times do
                VirtualInputManager:SendMouseButtonEvent(0, 0, 0, true, nil, 0)
                VirtualInputManager:SendMouseButtonEvent(0, 0, 0, false, nil, 0)
                task.wait(delayBetween)
            end
        end

        local function safeNudgePlayer()
            local char = player.Character
            if not char then return end
            local hrp = char:FindFirstChild("HumanoidRootPart")
            if not hrp then return end

            local origin = hrp.Position
            local offsets = {
                Vector3.new(20, 0, 0),
                Vector3.new(-14, 0, 0),
                Vector3.new(0, 0, 16),
                Vector3.new(0, 0, -12),
                Vector3.new(9, 0, 9),
                Vector3.new(-6, 0, 6),
                Vector3.new(20, 0, -20),
                Vector3.new(-7, 0, -7),
            }

            -- Shuffle offsets randomly
            for i = #offsets, 2, -1 do
                local j = math.random(i)
                offsets[i], offsets[j] = offsets[j], offsets[i]
            end

            for _, offset in ipairs(offsets) do
                local targetPos = origin + offset

                -- Avoid recent teleport spot
                if lastTeleportPos and (targetPos - lastTeleportPos).Magnitude < 7 then
                    continue
                end

                local rayOrigin = targetPos + Vector3.new(0, 5, 0)
                local rayResult = Workspace:Raycast(rayOrigin, Vector3.new(0, -10, 0), raycastParams)

                if rayResult then
                    local y = rayResult.Position.Y
                    local finalPos = Vector3.new(targetPos.X, y + 3, targetPos.Z)

                    -- Teleport
                    hrp.CFrame = CFrame.new(finalPos)
                    lastTeleportPos = finalPos
                    return
                end
            end

            -- Fallback: tp right
            local fallback = origin + Vector3.new(10, 0, 0)
            hrp.CFrame = CFrame.new(fallback)
            lastTeleportPos = fallback
        end

        local function equipAndTap()
            local char = player.Character
            if not char then return false end

            local humanoid = char:FindFirstChildOfClass("Humanoid")
            if not humanoid then return false end

            local equippedTool = char:FindFirstChildOfClass("Tool")
            if equippedTool and equippedTool.Name:lower():find("shovel") then
                safeNudgePlayer()
                tapScreenMultiple(5, 0.1)
                return true
            end

            for _, tool in ipairs(backpack:GetChildren()) do
                if tool:IsA("Tool") and tool.Name:lower():find("shovel") then
                    humanoid:EquipTool(tool)
                    task.wait(0.15)
                    safeNudgePlayer()
                    tapScreenMultiple(5, 0.1)
                    return true
                end
            end

            return false
        end

        local function unequipShovel()
            local char = player.Character
            if not char then return end
            local humanoid = char:FindFirstChildOfClass("Humanoid")
            if not humanoid then return end
            local equippedTool = char:FindFirstChildOfClass("Tool")
            if equippedTool and equippedTool.Name:lower():find("shovel") then
                humanoid:UnequipTools()
            end
        end

        getgenv().ShovelLoop = task.spawn(function()
            while getgenv().AutoEquipShovel do
                local success = equipAndTap()
                if success then
                    local waitTime = 13
                    for i = 1, waitTime * 13 do
                        if not getgenv().AutoEquipShovel then break end
                        task.wait(0.1)
                    end
                    unequipShovel()
                else
                    task.wait(2)
                end
            end
        end)

        getgenv().ShovelCharConn = player.CharacterAdded:Connect(function(char)
            char:WaitForChild("Humanoid")
            equipAndTap()
        end)

    else
        if getgenv().ShovelLoop then
            getgenv().ShovelLoop:Cancel()
            getgenv().ShovelLoop = nil
        end
        if getgenv().ShovelCharConn then
            getgenv().ShovelCharConn:Disconnect()
            getgenv().ShovelCharConn = nil
        end
    end
end)


Section:NewToggle("Auto Strong Dig OP (Get Strong)", "Auto Digs OP Good", function(Value)
    getgenv().AutoDigOP = Value

    if Value then
        local Players = game:GetService("Players")
        local RunService = game:GetService("RunService")
        local VirtualInputManager = game:GetService("VirtualInputManager")

        local player = Players.LocalPlayer
        local gui = player:WaitForChild("PlayerGui")

        -- Disconnect any previous connection if it exists
        if getgenv().AutoDigOPConnection then
            getgenv().AutoDigOPConnection:Disconnect()
        end

        -- Monitor UI updates and reconnect
        getgenv().AutoDigOPConnection = RunService.RenderStepped:Connect(function()
            if not getgenv().AutoDigOP then return end

            local digUI = gui:FindFirstChild("Dig")
            if not digUI then return end

            local holder = digUI:FindFirstChild("Safezone") and digUI.Safezone:FindFirstChild("Holder")
            if not holder then return end

            local playerBar = holder:FindFirstChild("PlayerBar")
            local areaStrong = holder:FindFirstChild("Area_Strong")

            if not playerBar or not areaStrong then return end

            -- Check if they overlap
            local function isOverlapping(gui1, gui2)
                local abs1 = gui1.AbsolutePosition
                local size1 = gui1.AbsoluteSize
                local abs2 = gui2.AbsolutePosition
                local size2 = gui2.AbsoluteSize

                return abs1.X < abs2.X + size2.X and
                       abs1.X + size1.X > abs2.X and
                       abs1.Y < abs2.Y + size2.Y and
                       abs1.Y + size1.Y > abs2.Y
            end

            if isOverlapping(playerBar, areaStrong) then
                VirtualInputManager:SendMouseButtonEvent(0, 0, 0, true, nil, 0)
                VirtualInputManager:SendMouseButtonEvent(0, 0, 0, false, nil, 0)
            end
        end)
    else
        -- Clean disconnect when toggled off
        if getgenv().AutoDigOPConnection then
            getgenv().AutoDigOPConnection:Disconnect()
            getgenv().AutoDigOPConnection = nil
        end
    end
end)

local movementConnection = nil  -- Used to store the movement loop connection

-- Function to start CFrame movement
function startCFrameSpeed()
    if movementConnection == nil then
        movementConnection = RunService.Stepped:Connect(function()
            LocalPlayer.Character.HumanoidRootPart.CFrame = LocalPlayer.Character.HumanoidRootPart.CFrame + LocalPlayer.Character.Humanoid.MoveDirection * Multiplier
        end)
    end
end

-- Function to stop CFrame movement
function stopCFrameSpeed()
    if movementConnection then
        movementConnection:Disconnect()
        movementConnection = nil
    end
end

-- Function to get NPC names dynamically
local function getNpcNames()
    local names = {}
    for _, npc in ipairs(npcFolder:GetChildren()) do
        if npc:IsA("Model") and npc:FindFirstChild("HumanoidRootPart") then
            table.insert(names, npc.Name)
        end
    end
    return names
end

-- Function to get ambience part names dynamically
local function getParts()
    local parts = {}
    for _, obj in ipairs(ambienceFolder:GetChildren()) do
        if obj:IsA("BasePart") then
            table.insert(parts, obj.Name)
        end
    end
    return parts
end

local function createTutorialGui()
    -- Create the GUI
    local Turtorial = Instance.new("ScreenGui")
    local main = Instance.new("Frame")
    local JoinButton = Instance.new("TextButton")
    local UICorner_Join = Instance.new("UICorner")
    local shadow = Instance.new("Frame")

    Turtorial.Name = "Turtorial"
    Turtorial.Parent = player:WaitForChild("PlayerGui")
    Turtorial.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

    main.Name = "main"
    main.Parent = Turtorial
    main.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    main.AnchorPoint = Vector2.new(0, 0)

    -- Function to position the GUI just below the top-left corner of the screen
    local function setPositionBelowTopLeft()
        local screenSize = game:GetService("Workspace").CurrentCamera.ViewportSize
        main.Position = UDim2.new(0, 10, 0, 10)  -- Just below the top-left with a slight offset
        main.Size = UDim2.new(0.15, 0, 0.10, 0)  -- Standard size for both devices
    end

    setPositionBelowTopLeft()

    main.Active = true
    main.Draggable = true

    -- Shadow effect for main frame
    shadow.Name = "shadow"
    shadow.Parent = main
    shadow.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    shadow.BackgroundTransparency = 0.6
    shadow.Position = UDim2.new(0, 5, 0, 5)
    shadow.Size = UDim2.new(1, 0, 1, 0)
    shadow.ZIndex = 0

    JoinButton.Name = "JoinButton"
    JoinButton.Parent = main
    JoinButton.BackgroundColor3 = Color3.fromRGB(40, 120, 255)

    -- Adjust button size based on device type
    if game:GetService("UserInputService").TouchEnabled then
        JoinButton.Size = UDim2.new(0.9, 0, 0.4, 0)
    else
        JoinButton.Size = UDim2.new(0.9, 0, 0.4, 0)
    end

    JoinButton.Position = UDim2.new(0.05, 0, 0.5, 0)
    JoinButton.Font = Enum.Font.Highway
    JoinButton.Text = "Toggle Gui"
    JoinButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    JoinButton.TextSize = 25
    JoinButton.TextStrokeTransparency = 0.7
    JoinButton.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)

    UICorner_Join.CornerRadius = UDim.new(0.15, 0)
    UICorner_Join.Parent = JoinButton

    -- Toggle the visibility of ScreenGuis
    JoinButton.MouseButton1Down:Connect(function()
        local CoreGui = game:GetService("CoreGui")
        local function containsNumber(name)
            return name:match("%d") ~= nil
        end

        local function toggleScreenGuisWithNumbers()
            for _, gui in ipairs(CoreGui:GetChildren()) do
                if gui:IsA("ScreenGui") and containsNumber(gui.Name) and gui.Name ~= "Turtorial" then
                    gui.Enabled = not gui.Enabled
                    print((gui.Enabled and "Enabled" or "Disabled") .. " ScreenGui:", gui.Name)
                end
            end
        end

        toggleScreenGuisWithNumbers()

        CoreGui.ChildAdded:Connect(function(child)
            if child:IsA("ScreenGui") then
                task.wait(0.1)
                if containsNumber(child.Name) and child.Name ~= "Turtorial" then
                    child.Enabled = false
                    print("Disabled new ScreenGui:", child.Name)
                end
            end
        end)
    end)
end
