local dragging = false
local dragInput
local dragStart
local startPos

local function updateInput(input)
	local delta = input.Position - dragStart
	MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

TitleBar.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = MainFrame.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

TitleBar.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
		updateInput(input)
	end
end)

local Categories = {
	MAIN = {"ANTI-HIT", "GO TO BEST", "INFINITE JUMP", "SPEED BOOST"},
	HELPER = {"MAIN ESP", "SEMI-INVISIBLE", "GRAPPLE SPEED"},
	OTHER = {"AIMBOT", "LAG EVERYONE"},

}

local CurrentTab = "MAIN"

local function ClearContent()
	for _, child in ipairs(ContentFrame:GetChildren()) do
		if child:IsA("TextButton") or (child:IsA("TextLabel") and child.Name ~= "DiscordText") then
			child:Destroy()
		end
	end
end

local function CreateFeatureButton(feature, index)
	local btn = Instance.new("TextButton")
	btn.Name = feature .. "Btn"
	btn.Size = UDim2.new(1, -15, 0, 28)
	btn.Position = UDim2.new(0, 8, 0, 10 + (index - 1) * 35)
	btn.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
	btn.BorderSizePixel = 0
	btn.Text = feature
	btn.TextColor3 = Color3.fromRGB(150, 150, 150)
	btn.TextSize = 12
	btn.Font = Enum.Font.Gotham
	btn.AutoButtonColor = false
	btn.Parent = ContentFrame
	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0, 6)
	corner.Parent = btn
	local stroke = Instance.new("UIStroke")
	stroke.Color = Color3.fromRGB(30, 30, 30)
	stroke.Thickness = 1
	stroke.Parent = btn
	local enabled = false
	btn.MouseButton1Click:Connect(function()
		enabled = not enabled
		btn.TextColor3 = enabled and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(150, 150, 150)
	end)
	return btn
end

local function LoadCategory(category)
	ClearContent()
	CurrentTab = category
	local buttons = Categories[category]
	if not buttons then return end
	for i, feature in ipairs(buttons) do
		CreateFeatureButton(feature, i)
	end
end

local function HighlightTab(tabButton)
	MainTab.TextColor3 = Color3.fromRGB(150, 150, 150)
	HelperTab.TextColor3 = Color3.fromRGB(150, 150, 150)
	OtherTab.TextColor3 = Color3.fromRGB(150, 150, 150)
	tabButton.TextColor3 = Color3.fromRGB(255, 255, 255)
end

MainTab.MouseButton1Click:Connect(function()
	HighlightTab(MainTab)
	LoadCategory("MAIN")
end)

HelperTab.MouseButton1Click:Connect(function()
	HighlightTab(HelperTab)
	LoadCategory("HELPER")
end)

OtherTab.MouseButton1Click:Connect(function()
	HighlightTab(OtherTab)
	LoadCategory("OTHER")
end)

local guiOpen = false
ToggleButton.MouseButton1Click:Connect(function()
	guiOpen = not guiOpen
	MainFrame.Visible = guiOpen
end)

UserInputService.InputBegan:Connect(function(input, gameProcessed)
	if gameProcessed then return end
	if input.KeyCode == Enum.KeyCode.F then
		guiOpen = not guiOpen
		MainFrame.Visible = guiOpen
	end
end)

HighlightTab(MainTab)
LoadCategory("MAIN")
