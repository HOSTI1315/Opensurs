local function getOwnedPlot()
	for _, plot in pairs(workspace.Farm:GetChildren()) do
		local imp = plot:FindFirstChild("Important") or plot:FindFirstChild("Importanert")
		if imp then
			local data = imp:FindFirstChild("Data")
			if data and data:FindFirstChild("Owner") and data.Owner.Value == player.Name then
				return plot
			end
		end
	end
	return nil
end

local function getCFrameFromModel(model)
	if model.PrimaryPart then
		return model.PrimaryPart.CFrame
	else
		for _, part in ipairs(model:GetDescendants()) do
			if part:IsA("BasePart") then
				return part.CFrame
			end
		end
	end
	return nil
end

local function getRandomPlantCFrame()
	local plot = getOwnedPlot()
	if plot then
		local imp = plot:FindFirstChild("Important")
		local farm = imp and imp:FindFirstChild("Plants_Physical")
		if farm then
			local plants = farm:GetChildren()
			if #plants > 0 then
				local randomPlant = plants[math.random(1, #plants)]

				if randomPlant:IsA("Model") and randomPlant.PrimaryPart then
					return randomPlant.PrimaryPart.CFrame
				end

				local part = randomPlant:FindFirstChild("1")
				if part and part:IsA("BasePart") then
					return part.CFrame
				end

				for _, child in ipairs(randomPlant:GetDescendants()) do
					if child:IsA("BasePart") then
						return child.CFrame
					end
				end
			end
		end
	end
	return nil
end

local function autoBuySeedsFunction()
	local shopUI = player:WaitForChild("PlayerGui"):WaitForChild("Seed_Shop")
	local scroll = shopUI.Frame.ScrollingFrame
	for _, item in pairs(scroll:GetChildren()) do
		local stock = item:FindFirstChild("Stock")
		if stock and stock:IsA("NumberValue") and stock.Value > 0 then
			ReplicatedStorage:WaitForChild("GameEvents"):WaitForChild("BuySeedStock"):FireServer(item.Name)
		end
	end
end

local function findSeedTool()
	for _, item in ipairs(player.Character:GetChildren()) do
		if item:IsA("Tool") and item.Name:find("Seed") then
			local crop = item.Name:match("^(.-) Seed")
			return item, crop
		end
	end
	for _, item in ipairs(backpack:GetChildren()) do
		if item:IsA("Tool") and item.Name:find("Seed") then
			local crop = item.Name:match("^(.-) Seed")
			return item, crop
		end
	end
	return nil, nil
end

local function harvestRandomPlant()
	local cf = getRandomPlantCFrame()
	if cf then
		local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
		if hrp then
			hrp.CFrame = cf + CFrame.new(0, 5, 0)
			for _, prompt in ipairs((cf and (getOwnedPlot().Important.Plants_Physical:GetDescendants() or {})) ) do
				if prompt:IsA("ProximityPrompt") then
					fireproximityprompt(prompt, 1, true)
				end
			end
		end
	end
end

task.spawn(function()
	while task.wait(0.1) do
		if autoPlant then
			local plot = getOwnedPlot()
			if plot then
				local imp = plot:FindFirstChild("Important")
				local locs = imp and imp:FindFirstChild("Plant_Locations")
				local targetPos
				if locs and #locs:GetChildren() > 0 then
					local boundaries = locs:GetChildren()
					local sel = boundaries[math.random(1, #boundaries)]
					if sel:IsA("BasePart") then
						local size = sel.Size
						local center = sel.Position
						local randomX = center.X + (math.random() - 0.5) * size.X
						local randomZ = center.Z + (math.random() - 0.5) * size.Z
						targetPos = Vector3.new(randomX, center.Y, randomZ)
					elseif sel:IsA("Model") then
						local cf = getCFrameFromModel(sel)
						if cf then
							targetPos = cf.Position
						end
					end
				end
				targetPos = targetPos or (player.Character and player.Character:FindFirstChild("HumanoidRootPart") and player.Character.HumanoidRootPart.Position) or Vector3.new(0, 0, 0)
				local tool, seedType = findSeedTool()
				if tool and seedType then
					if tool.Parent == backpack then
						player.Character:WaitForChild("Humanoid"):EquipTool(tool)
						repeat task.wait() until tool.Parent == player.Character
					end
					remote:FireServer(targetPos, seedType)
					local character = player.Character or player.CharacterAdded:Wait()
					local hrp = character:WaitForChild("HumanoidRootPart")
					hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 5, 0))
				end
			end
		end
	end
end)

local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local Players = game:GetService("Players")

local localPlayer = Players.LocalPlayer
local placeId = game.PlaceId
local currentJobId = game.JobId

local function getAvailableServer()
	local cursor = nil

	while true do
		local url = "https://games.roblox.com/v1/games/" .. placeId .. "/servers/Public?sortOrder=Asc&limit=100"
		if cursor then url = url .. "&cursor=" .. cursor end

		local success, data = pcall(function()
			return HttpService:JSONDecode(game:HttpGet(url))
		end)

		if success and data and data.data then
			for _, server in ipairs(data.data) do
				if server.id ~= currentJobId and server.playing < server.maxPlayers - 1 and not server.privateServerId then
					return server
				end
			end
			cursor = data.nextPageCursor
			if not cursor then break end
		else
			break
		end
	end

	return nil
end

local function teleportLoop()
	while true do
		local server = getAvailableServer()
		if server then
			local success, err = pcall(function()
				TeleportService:TeleportToPlaceInstance(placeId, server.id, localPlayer)
			end)
			if success then
				print("Teleport success!")
				return
			else
				warn("Teleport failed, retrying... (" .. tostring(err) .. ")")
			end
		else
			warn("No valid server found, retrying...")
		end
		wait(2)
	end
end

teleportLoop()

    end
})
