-- Animation particles (simplified version - original had 47 particles)
local function createParticle(positionX, sizeX, sizeY, delay)
    task.wait(delay)
    local particle = Instance.new("Frame")
    particle.Parent = Frame
    particle.Size = UDim2.new(0, sizeX, 0, sizeY)
    particle.Position = UDim2.new(0, positionX, 0, -sizeY)
    particle.BackgroundColor3 = Color3.fromRGB(0, 200, 255)
    particle.BackgroundTransparency = 0.3
    particle.BorderSizePixel = 0
    
    local particleCorner = Instance.new("UICorner")
    particleCorner.Parent = particle
    particleCorner.CornerRadius = UDim.new(1, 0)
    
    local tween = TweenService:Create(particle, TweenInfo.new(1.8, Enum.EasingStyle.Linear), {
        Position = UDim2.new(0, positionX, 0, 1080)
    })
    tween:Play()
    
    tween.Completed:Connect(function()
        particle:Destroy()
    end)
end

-- Create animation particles
task.spawn(function()
    -- Create a few sample particles instead of 47
    local positions = {8, 71, 1553, 1690, 33, 714, 763, 1759}
    local sizes = {{6,9}, {8,12}, {6,12}, {7,11}, {6,10}, {9,6}, {9,7}, {6,8}}
    
    for i, pos in ipairs(positions) do
        if i <= #sizes then
            createParticle(pos, sizes[i][1], sizes[i][2], i * 0.08)
        end
    end
end)
