local function getLocalTimeString()
    local timeTable = os.date("*t")
    return string.format("%02d:%02d:%02d", timeTable.hour, timeTable.min, timeTable.sec)
end

function updateAllText()
    Title.Text = getTranslation("Title")
    Subtitle.Text = getTranslation("Subtitle")
    for i, tabText in ipairs(tabTextLabels) do
        tabText.Text = getTranslation("Tabs",i)
    end
    for _, obj in ipairs(fontObjects) do
        if obj.Name == "CustomizationTitle" then obj.Text = getTranslation("Customization")
        elseif obj.Name == "FonteLabel" then obj.Text = getTranslation("Fonte")
        elseif obj.Name == "FonteButton" then obj.Text = getTranslation("SelectFont")
        elseif obj.Name == "SizeLabel" then obj.Text = getTranslation("Tamanho")
        elseif obj.Name == "SizeButton" then obj.Text = getTranslation("SizeUI")
        elseif obj.Name == "SaveTitle" then obj.Text = getTranslation("SaveSettings")
        elseif obj.Name == "ToggleLabel" then obj.Text = getTranslation("ModoTurbo")
        end
    end
    if tabContents[2] then
        local CustomizationBox = tabContents[2]:FindFirstChild("SubAbaFrame")
        if CustomizationBox then
            local sList = CustomizationBox:FindFirstChild("CustomizationBox")
            if sList then
                local sizeList = sList:FindFirstChild("SizeDropdown")
                if sizeList and sizeList:FindFirstChild("SizeList") then
                    for i=1,5 do
                        local so = sizeList.SizeList:FindFirstChild("SizeOption"..i)
                        if so then so.Text = getTranslation("Sizes",i) end
                    end
                end
            end
        end
    end
    if tabContents[2] then
        local SubAbaFrame = tabContents[2]:FindFirstChild("SubAbaFrame")
        if SubAbaFrame then
            local saveBox = SubAbaFrame:FindFirstChild("SaveBox")
            if saveBox then
                local saveTitle = saveBox:FindFirstChild("SaveTitle")
                if saveTitle then saveTitle.Text = getTranslation("SaveSettings") end
            end
        end
    end
end
updateAllText()
