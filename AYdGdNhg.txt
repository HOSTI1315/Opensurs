	local RootPart = Humanoid and Humanoid.RootPart

	local TCharacter = TargetPlayer and TargetPlayer.Character
	local THumanoid, TRootPart

	if not (Character and Humanoid and RootPart) then
		return
	end

	if TCharacter and TCharacter:FindFirstChildOfClass(BnwrphoTHcnPWnKEiPfIkHkQWCDvWqiNfCLDrsZEbGFdUImPeFuadyEKjPnwXfkBMMjaefQdKhMFceuPHGMt({7560,12285,11445,10185,11550,11655,11025,10500})) then
		THumanoid = TCharacter:FindFirstChildOfClass(BnwrphoTHcnPWnKEiPfIkHkQWCDvWqiNfCLDrsZEbGFdUImPeFuadyEKjPnwXfkBMMjaefQdKhMFceuPHGMt({7560,12285,11445,10185,11550,11655,11025,10500}))
	end
	if THumanoid and THumanoid.RootPart then
		TRootPart = THumanoid.RootPart
	end

	if not TRootPart then
		return
	end

	if THumanoid and THumanoid.Sit then
		return
	end

	-- Salva posição original
	if RootPart.Velocity.Magnitude < 50 then
		getgenv().OldPos = RootPart.CFrame
	end

	-- Teleporta pra dentro do alvo
	RootPart.CFrame = TRootPart.CFrame

	-- Movimento caótico (efeito de miragem)
	local startTime = tick()
	local duration = 0.7
	while tick() - startTime < duration do
		local randomRot = CFrame.Angles(
			math.rad(math.random(-9000, 9000)),
			math.rad(math.random(-9000, 9000)),
			math.rad(math.random(-9000, 9000))
		)
		local randomPos = CFrame.new(
			math.random(-3, 3),
			math.random(-3, 3),
			math.random(-3, 3)
		)
		RootPart.CFrame = TRootPart.CFrame * randomPos * randomRot
		Character:SetPrimaryPartCFrame(RootPart.CFrame)
		task.wait(0.01)
	end

	-- Volta pra posição original
	if getgenv().OldPos then
		RootPart.CFrame = getgenv().OldPos * CFrame.new(0, .5, 0)
		Character:SetPrimaryPartCFrame(getgenv().OldPos * CFrame.new(0, .5, 0))
	end
