-- Free Axis Auto Spin Feature

local autoSpin = false
local spinSpeed = 50
local spinAngularName = "AutoSpinAngular"
local player = game:GetService("Players").LocalPlayer

-- Default axis values (Y axis only at start)
local spinAxis = Vector3.new(0, 1, 0)

-- Helper to get HumanoidRootPart safely
local function getHRP()
    local char = player.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        return char.HumanoidRootPart
    end
    return nil
end

-- Helper to apply or update BodyAngularVelocity (free axis)
local function applySpin()
    local hrp = getHRP()
    if not hrp then return end
    local av = hrp:FindFirstChild(spinAngularName)
    if not av then
        av = Instance.new("BodyAngularVelocity")
        av.Name = spinAngularName
        av.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
        av.AngularVelocity = spinAxis.Unit * spinSpeed
        av.Parent = hrp
    else
        av.AngularVelocity = spinAxis.Unit * spinSpeed
    end
end

-- Helper to remove BodyAngularVelocity
local function removeSpin()
    local hrp = getHRP()
    if hrp then
        local av = hrp:FindFirstChild(spinAngularName)
        if av then av:Destroy() end
    end
end

        end)

-- Heartbeat connection for extra safety (ensures spin is always applied)

    _G.__AutoSpinHeartbeat = game:GetService("RunService").Heartbeat:Connect(function()

    end)
end

-- Auto Guarantee System (หยุดทุกอย่างและปิด Toggle ถ้า Gold, Diamond, หรือ Void เป็น true)

local autoGuaranteeEnabled = false
local autoGuaranteeThread = nil

-- ตรวจสอบ Modifiers (Gold, Diamond, Void) ถ้ามีอันไหนเป็น true ให้หยุดและปิด Toggle
local function shouldStopByModifiers()
    local player = game:GetService("Players").LocalPlayer
    local modifiers = player:FindFirstChild("Modifiers")
    if not modifiers then return false end
    local gold = modifiers:FindFirstChild("Gold")
    local diamond = modifiers:FindFirstChild("Diamond")
    local void_ = modifiers:FindFirstChild("Void")
    -- ถ้า Gold, Diamond, หรือ Void มีค่า Value เป็น true ให้หยุด
    if (gold and gold.Value == true) or (diamond and diamond.Value == true) or (void_ and void_.Value == true) then
        return true
    end
    return false
end

local function getProgressInfo()
    local player = game:GetService("Players").LocalPlayer
    local gui = player:FindFirstChild("PlayerGui")
    if not gui then return nil, nil end
    local mainUI = gui:FindFirstChild("MainUI")
    if not mainUI then return nil, nil end
    local bottom = mainUI:FindFirstChild("Bottom")
    if not bottom then return nil, nil end
    local textLabel = bottom:FindFirstChild("TextLabel")
    if not textLabel or not textLabel:IsA("TextLabel") then return nil, nil end
    local text = textLabel.Text
    if not text then return nil, nil end
    -- Expecting format "3 / 10" or "3 / 50"
    local current, max = string.match(text, "(%d+)%s*/%s*(%d+)")
    current = tonumber(current)
    max = tonumber(max)
    return current, max
end

local function isProgressFull(current, max)
    if current and max and current >= max then
        return true
    end
    return false
end

local function stopAutoGuarantee()
    autoGuaranteeEnabled = false
    local option = Fluent and Fluent.Options and Fluent.Options.AutoGuaranteeToggle
    if option and option.SetValue then
        option:SetValue(false)
    end
end

local function teleportToPositionAndResetAndStop(targetPos)
    local Players = game:GetService("Players")
    local player = Players.LocalPlayer
    local character = player.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then return end

    local myChar = workspace:FindFirstChild(player.Name)
    if not myChar or not myChar:FindFirstChild("HumanoidRootPart") then return end

    local hrp = myChar.HumanoidRootPart
    hrp.CFrame = CFrame.new(targetPos)

    -- ก่อนจะรีเซ็ต ให้เช็ค Modifiers และ Progress อีกครั้ง
    task.wait(0.5)
    if shouldStopByModifiers() then
        return
    end
    local current, max = getProgressInfo()
    if isProgressFull(current, max) then
        return
    end
    if myChar:FindFirstChildOfClass("Humanoid") then
        myChar:FindFirstChildOfClass("Humanoid").Health = 0
    end
end

Tabs.Main:AddToggle("AutoGuaranteeToggle", {
    Title = "Auto Guarantee (Instant Move & Reset)",
    Description = "Teleport to (5, 2012, -79), wait, reset, auto stop if Gold, Diamond, or Void is true, or progress is full.",
    Default = false,
    Callback = function(value)
        autoGuaranteeEnabled = value
        if value then
            if autoGuaranteeThread and coroutine.status(autoGuaranteeThread) ~= "dead" then
                return
            end
            autoGuaranteeThread = coroutine.create(function()
                while autoGuaranteeEnabled do
                    -- ตรวจ Modifiers ก่อนเสมอ
                    if shouldStopByModifiers() then
                        stopAutoGuarantee()
                        break
                    end

                    local current, max = getProgressInfo()
                    if isProgressFull(current, max) then
                        stopAutoGuarantee()
                        break
                    end

                    teleportToPositionAndResetAndStop(Vector3.new(5, 2012, -79))

                    for i = 1, 10 do
                        if not autoGuaranteeEnabled then break end
                        if shouldStopByModifiers() then
                            stopAutoGuarantee()
                            break
                        end
                        local current, max = getProgressInfo()
                        if isProgressFull(current, max) then
                            stopAutoGuarantee()
                            break
                        end
                        task.wait(0.2)
                    end
                end
            end)
            coroutine.resume(autoGuaranteeThread)
        else
            autoGuaranteeEnabled = false
        end
    end
})
