local plants_table = {}
local selectedSeed = {}

for _, seed in ipairs(plants:GetChildren()) do
    if seed:IsA("Frame") and seed.Name:lower():find("seed") then
        local stockLabel = seed:FindFirstChild("Stock")
        local stockValue = 0
        if stockLabel then
            stockValue = tonumber(stockLabel.Text:match("%d+")) or 0
        end
        table.insert(plants_table, seed.Name .. " (Stock: " .. stockValue .. ")")
    end
end

local Dropdown = Tab2:Dropdown({
    Title = "Select Seed to buy",
    Values = plants_table,
    Value = { "None" },
    Multi = true,
    AllowNone = true,
    Callback = function(option)
        if type(option) == "table" then
            selectedSeed = {}
            for _, v in ipairs(option) do
                if v ~= "None" then
                    table.insert(selectedSeed, tostring(v))
                end
            end
        else
            selectedSeed = { tostring(option) }
        end
        print("Selected seeds:", game:GetService("HttpService"):JSONEncode(selectedSeed))
    end
})

local Button = Tab2:Button({
    Title = "Buy Selected Seed(s)",
    Desc = "Buy all currently selected seeds from the dropdown.",
    Locked = false,
    Callback = function()
        if not selectedSeed or #selectedSeed == 0 then
            WindUI:Notify({
                Title = "Error",
                Content = "No seeds selected!",
                Duration = 3,
                Icon = "x",
            })
            return
        end
        for _, seedStr in ipairs(selectedSeed) do
            local seedName = seedStr:match("^(.-) %(") or seedStr
            local seedFrame = plants:FindFirstChild(seedName)
            local stockLabel = seedFrame and seedFrame:FindFirstChild("Stock")
            local stockValue = 0
            if stockLabel then
                stockValue = tonumber(stockLabel.Text:match("%d+")) or 0
            end
            if stockValue <= 0 then
                WindUI:Notify({
                    Title = "Out of Stock",
                    Content = seedName .. " is currently out of stock!",
                    Duration = 3,
                    Icon = "octagon-alert",
                })
            else
                buyItem:FireServer(seedName, true)
                WindUI:Notify({
                    Title = "Purchased",
                    Content = seedName .. " bought successfully!",
                    Duration = 2,
                    Icon = "badge-check",
                })
            end
            task.wait(0.15)
        end
    end
})
