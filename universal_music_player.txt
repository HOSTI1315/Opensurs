    local function lerp(startVal, endVal, alpha)
        return startVal + (endVal - startVal) * alpha
    end

    local function evaluateColorSequence(colorSequence, time)
        local keypoints = colorSequence.Keypoints
        local prevKeypoint = nil
        local nextKeypoint = nil

        for i, keypoint in ipairs(keypoints) do
            if prevKeypoint then
                if keypoint.Time > prevKeypoint.Time and keypoint.Time <= time then
                    nextKeypoint = colorSequence.Keypoints[i + 1] or keypoint
                    prevKeypoint = keypoint
                end
            else
                nextKeypoint = colorSequence.Keypoints[i + 1] or keypoint
                prevKeypoint = keypoint
            end
        end

        local lerpFactor = (time - prevKeypoint.Time) / (nextKeypoint.Time - prevKeypoint.Time)

        if lerpFactor == lerpFactor then
            return prevKeypoint.Value:lerp(nextKeypoint.Value, lerpFactor)
        else
            return prevKeypoint.Value
        end
    end

    function visualizerModule.Stop(_)
        if activeVisualizer.IsActive then
            activeVisualizer.IsActive = false

            for _, connection in pairs(activeVisualizer.connections or {}) do
                connection:Disconnect()
            end

            if activeVisualizer.mainRenderConnection then
                activeVisualizer.mainRenderConnection:Disconnect()
            end

            for _, objectName in ipairs({
                'effectFolder',
                'bloomEffect',
                'sunRaysEffect',
                'sound',
            }) do
                if activeVisualizer[objectName] and activeVisualizer[objectName].Parent then
                    activeVisualizer[objectName]:Destroy()
                end
            end

            activeVisualizer = {}
        end
    end

    function visualizerModule.Start(_)
        if activeVisualizer.IsActive then
            visualizerModule:Stop()
        end

        activeVisualizer = {
            IsActive = true,
            connections = {},
            isDying = false,
            startTime = tick(),
        }

        local localPlayer = playersService.LocalPlayer
        local character = localPlayer.Character or localPlayer.CharacterAdded:Wait()
        local humanoid = character:WaitForChild('Humanoid')
        local humanoidRootPart = character:WaitForChild('HumanoidRootPart')
        local currentCamera = workspace.CurrentCamera

        activeVisualizer.defaultFieldOfView = currentCamera.FieldOfView
        activeVisualizer.sound = Instance.new('Sound')
        activeVisualizer.effectFolder = Instance.new('Folder')
        activeVisualizer.bloomEffect = Instance.new('BloomEffect')
        activeVisualizer.sunRaysEffect = Instance.new('SunRaysEffect')
        activeVisualizer.sound.SoundId = config.MUSIC_ID
        activeVisualizer.sound.Looped = true
        activeVisualizer.sound.Parent = humanoidRootPart

        activeVisualizer.sound:Play()

        activeVisualizer.effectFolder.Name = 'AudioVisualizerFolder'
        activeVisualizer.effectFolder.Parent = character
        activeVisualizer.bloomEffect.Name = 'WaveBloom'
        activeVisualizer.bloomEffect.Enabled = true
        activeVisualizer.bloomEffect.Parent = lighting
        activeVisualizer.sunRaysEffect.Name = 'WaveSunRays'
        activeVisualizer.sunRaysEffect.Enabled = true
        activeVisualizer.sunRaysEffect.Parent = lighting

        for i = 0, math.floor(360 / config.PART_SPACING) - 1 do
            local angle = i * config.PART_SPACING
            local part = Instance.new('Part')

            part.Size = Vector3.new()
            part.Anchored = true
            part.CanCollide = false
            part.Material = Enum.Material.Neon
            part.Transparency = 0.7
            part.Color = Color3.new(1, 1, 1)
            part.Name = tostring(angle)

            part:SetAttribute('RelativePos', Vector3.new(math.sin(math.rad(angle)), 0, math.cos(math.rad(angle))))

            part.Parent = activeVisualizer.effectFolder
        end

        local function startDeathSequence()
            if not activeVisualizer.isDying and activeVisualizer.IsActive then
                activeVisualizer.isDying = true

                if activeVisualizer.mainRenderConnection then
                    activeVisualizer.mainRenderConnection:Disconnect()
                    activeVisualizer.mainRenderConnection = nil
                end

                tweenService:Create(activeVisualizer.sound, TweenInfo.new(config.DEATH_FADE_DURATION), {Volume = 0}):Play()

                local deathStartTime = tick()
                local startAngle = activeVisualizer.sound.TimePosition / activeVisualizer.sound.TimeLength * 3600
                local deathConnection = nil

                deathConnection = runService.RenderStepped:Connect(function()
                    if activeVisualizer.effectFolder and activeVisualizer.effectFolder.Parent then
                        local elapsed = tick() - deathStartTime
                        local progress = math.clamp((tick() - deathStartTime) / config.DEATH_FADE_DURATION, 0, 1)
                        local currentAngle = lerp(startAngle, 15000, progress)
                        local currentRadius = lerp(config.BASE_RADIUS, 0, progress)

                        for _, child in ipairs(activeVisualizer.effectFolder:GetChildren()) do
                            if child:IsA('BasePart') then
                                child.Transparency = lerp(child.Transparency, 1, progress)
                                child.Size = child.Size:Lerp(Vector3.new(), progress)

                                local childAngle = tonumber(child.Name)

                                if childAngle then
                                    local rotation = CFrame.Angles(0, math.rad(currentAngle * elapsed + childAngle), 0)
                                    local position = (CFrame.new(humanoidRootPart.Position) * rotation * CFrame.new(child:GetAttribute('RelativePos') * currentRadius)).Position

                                    child.CFrame = CFrame.new(position)
                                end
                            end
                        end

                        if config.DEATH_FADE_DURATION <= elapsed then
                            deathConnection:Disconnect()
                            visualizerModule:Stop()
                        end
                    else
                        deathConnection:Disconnect()
                        visualizerModule:Stop()
                    end
                end)

                table.insert(activeVisualizer.connections, deathConnection)
            end
        end

        table.insert(activeVisualizer.connections, humanoid.Died:Connect(startDeathSequence))

        activeVisualizer.mainRenderConnection = runService.RenderStepped:Connect(function(deltaTime)
            if activeVisualizer.isDying or not character.Parent then
                visualizerModule:Stop()
                return
            elseif activeVisualizer.effectFolder and activeVisualizer.effectFolder.Parent then
                local spawnProgress = (function(start, target, t)
                    return (target - start) * (-math.pow(2, -10 * t) + 1) + start
                end)(0, 1, math.clamp((tick() - activeVisualizer.startTime) / config.SPAWN_DURATION, 0, 1))

                local loudness = activeVisualizer.sound.PlaybackLoudness / 1000
                local intensity = math.clamp(loudness, 0, 1)
                local shakeAmount = intensity * config.CAMERA_SHAKE_INTENSITY

                currentCamera.CFrame = currentCamera.CFrame * CFrame.Angles(
                    math.random() * shakeAmount - shakeAmount / 2,
                    math.random() * shakeAmount - shakeAmount / 2,
                    math.random() * shakeAmount - shakeAmount / 2
                )
                currentCamera.FieldOfView = lerp(currentCamera.FieldOfView, activeVisualizer.defaultFieldOfView + config.CAMERA_FOV_INCREASE * intensity, deltaTime * 10)
                activeVisualizer.bloomEffect.Intensity = intensity * config.BLOOM_INTENSITY_MULTIPLIER
                activeVisualizer.sunRaysEffect.Intensity = intensity * config.SUNRAYS_INTENSITY_MULTIPLIER

                local waveAngle = math.abs(math.sin(math.rad(loudness * 360))) * 360
                local rotationAngle = activeVisualizer.sound.TimePosition / activeVisualizer.sound.TimeLength * 3600
                local waveFactor = math.sin(math.rad(waveAngle)) + 1
                local currentRadius = lerp(config.BASE_RADIUS, loudness * config.RADIUS_MULTIPLIER + config.BASE_RADIUS, deltaTime * 4) * spawnProgress

                for _, child in ipairs(activeVisualizer.effectFolder:GetChildren()) do
                    if child:IsA('BasePart') then
                        local childAngle = tonumber(child.Name)
                        local relativeWave = math.sin(math.rad(childAngle)) + 1 - waveFactor
                        local sizeFactor = (2 - math.abs(relativeWave) * 3) / 2
                        local clampedFactor = math.clamp(sizeFactor, 0, 1) * math.clamp(loudness * 10, 0, 1)
                        local height = (function(start, target, t)
                            return (target - start) * math.pow(2, 10 * (t - 1)) + start
                        end)(0, 1, clampedFactor) * config.SIZE_MULTIPLIER * spawnProgress

                        child.Size = child.Size:Lerp(Vector3.new(config.SIZE_X, height, config.SIZE_Z), math.clamp(clampedFactor, deltaTime * 4, 1))

                        local colorAtTime = evaluateColorSequence(config.COLOR_SEQUENCE, clampedFactor)

                        child.Color = child.Color:Lerp(colorAtTime, math.clamp(clampedFactor, deltaTime * 4, 1))

                        local yOffset = math.sin(tick() * config.WAVE_SPEED + childAngle / 30) * config.WAVE_HEIGHT
                        local position = (CFrame.new(humanoidRootPart.Position + Vector3.new(0, child.Size.Y / 2 + yOffset, 0)) *
                                         CFrame.Angles(0, math.rad(rotationAngle * config.ROTATION_SPEED_MULT), 0) *
                                         CFrame.new(child:GetAttribute('RelativePos') * currentRadius)).Position

                        child.CFrame = CFrame.new(position, humanoidRootPart.Position + Vector3.new(0, child.Size.Y / 2, 0))

                        local pointLight = child:FindFirstChildOfClass('PointLight')
                        if pointLight then
                            pointLight:Destroy()
                        end

                        local newLight = Instance.new('PointLight')
                        newLight.Color = child.Color
                        newLight.Brightness = clampedFactor * config.LIGHT_BRIGHTNESS_MULT * spawnProgress
                        newLight.Range = config.LIGHT_RANGE
                        newLight.Shadows = false
                        newLight.Parent = child
                    end
                end
            else
                visualizerModule:Stop()
            end
        end)
    end

    end

    function visualizerModule.LoadConfig(_, loadedConfig)
        for key, value in pairs(loadedConfig) do
            if key == 'COLOR_SEQUENCE' then
                local keypoints = {}
                for _, kp in ipairs(value.Keypoints) do
                    table.insert(keypoints, ColorSequenceKeypoint.new(kp.Time, Color3.new(kp.Value.R, kp.Value.G, kp.Value.B)))
                end
                config[key] = ColorSequence.new(keypoints)
            else
                config[key] = value
            end
        end

        if activeVisualizer.IsActive then
            visualizerModule:Stop()
            visualizerModule:Start()
        end
    end

        local sliderHandles = {}

        local function createSlider(parent, configKey, label, minVal, maxVal, displayFunc, callback)
            local container = Instance.new('Frame', parent)

            container.Size = UDim2.new(1, 0, 0, 50)
            container.BackgroundTransparency = 1

            local labelText = Instance.new('TextLabel', container)

            labelText.Size = UDim2.new(1, 0, 0, 20)
            labelText.Text = label
            labelText.Font = Enum.Font.Gotham
            labelText.TextColor3 = Color3.fromRGB(220, 222, 230)
            labelText.TextXAlignment = Enum.TextXAlignment.Left
            labelText.TextSize = 14
            labelText.BackgroundTransparency = 1

            local valueText = Instance.new('TextLabel', labelText)

            valueText.Size = UDim2.new(0.5, 0, 1, 0)
            valueText.AnchorPoint = Vector2.new(1, 0)
            valueText.Position = UDim2.new(1, 0, 0, 0)
            valueText.Font = Enum.Font.GothamBold
            valueText.TextColor3 = Color3.fromRGB(255, 255, 255)
            valueText.TextXAlignment = Enum.TextXAlignment.Right
            valueText.TextSize = 14
            valueText.BackgroundTransparency = 1

            local track = Instance.new('Frame', container)

            track.Size = UDim2.new(1, 0, 0, 6)
            track.Position = UDim2.new(0, 0, 0, 25)
            track.BackgroundColor3 = Color3.fromRGB(25, 26, 33)
            Instance.new('UICorner', track).CornerRadius = UDim.new(0, 3)

            local fill = Instance.new('Frame', track)

            fill.BackgroundColor3 = Color3.fromRGB(115, 134, 242)
            Instance.new('UICorner', fill).CornerRadius = UDim.new(0, 3)

            local thumb = Instance.new('TextButton', track)

            thumb.Size = UDim2.new(0, 14, 0, 14)
            thumb.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
            thumb.Text = ''
            Instance.new('UICorner', thumb).CornerRadius = UDim.new(1, 0)

            local dragging = false
            local dragConnection = nil

            local function updateSlider(value)
                local norm = (value - minVal) / (maxVal - minVal)
                norm = math.clamp(norm, 0, 1)

                fill.Size = UDim2.new(norm, 0, 1, 0)
                thumb.Position = UDim2.new(norm, -7, 0.5, -7)
                valueText.Text = displayFunc(value)
            end

            thumb.MouseButton1Down:Connect(function()
                dragging = true
                dragConnection = userInputService.InputChanged:Connect(function(input)
                    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                        local relativeX = (input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X
                        local newValue = minVal + (maxVal - minVal) * math.clamp(relativeX, 0, 1)

                        updateSlider(newValue)

                        if callback then
                            callback(newValue)
                        end
                    end
                end)
            end)
            thumb.MouseButton1Up:Connect(function()
                dragging = false
                if dragConnection then
                    dragConnection:Disconnect()
                end
            end)
            updateSlider(visualizerModule:GetCurrentConfig()[configKey])

            if configKey then
                sliderHandles[configKey] = {Update = updateSlider}
            end

            configList:ClearAllChildren()

            Instance.new('UIListLayout', configList).Padding = UDim.new(0, 2)

            local success, files = pcall(listfiles, configFolder)

            if success then
                for _, filePath in ipairs(files) do
                    if filePath:sub(-5) == '.json' then
                        local configName = filePath:sub(#configFolder + 2, -6)
                        local row = Instance.new('Frame', configList)

                        row.Size = UDim2.new(1, 0, 0, 30)

                        local rowLayout = Instance.new('UIListLayout', row)

                        local nameLabel = Instance.new('TextLabel', row)

                        nameLabel.Size = UDim2.new(0.4, -10, 1, 0)
                        nameLabel.BackgroundColor3 = Color3.fromRGB(45, 47, 59)
                        nameLabel.TextColor3 = Color3.fromRGB(220, 222, 230)
                        nameLabel.Font = Enum.Font.Gotham
                        nameLabel.TextSize = 11
                        nameLabel.Text = ' ' .. configName
                        nameLabel.TextXAlignment = Enum.TextXAlignment.Left
                        Instance.new('UICorner', nameLabel).CornerRadius = UDim.new(0, 4)

                        local loadBtn = Instance.new('TextButton', row)

                        loadBtn.Size = UDim2.new(0.2, 0, 1, 0)
                        loadBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 75)
                        loadBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
                        loadBtn.Font = Enum.Font.GothamBold
                        loadBtn.TextSize = 11
                        loadBtn.Text = 'Load'
                        Instance.new('UICorner', loadBtn).CornerRadius = UDim.new(0, 4)

                        local autoBtn = Instance.new('TextButton', row)

                        autoBtn.Size = UDim2.new(0.2, 0, 1, 0)
                        autoBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 75)
                        autoBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
                        autoBtn.Font = Enum.Font.GothamBold
                        autoBtn.TextSize = 11
                        autoBtn.Text = 'Auto'
                        Instance.new('UICorner', autoBtn).CornerRadius = UDim.new(0, 4)

                        local deleteBtn = Instance.new('TextButton', row)

                        deleteBtn.Size = UDim2.new(0.2, 0, 1, 0)
                        deleteBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
                        deleteBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
                        deleteBtn.Font = Enum.Font.GothamBold
                        deleteBtn.TextSize = 11
                        deleteBtn.Text = 'Del'
                        Instance.new('UICorner', deleteBtn).CornerRadius = UDim.new(0, 4)

                            local ok, data = pcall(readfile, filePath)

                            if ok and data then
                                local decodeOk, decoded = pcall(function()
                                    return httpService:JSONDecode(data)
                                end)

                                if decodeOk then
                                    visualizerModule:LoadConfig(decoded)

                                    for key, handle in pairs(sliderHandles) do
                                        handle.Update(decoded[key] or 0)
                                    end
                                end
                            end
                        end)

                        autoBtn.MouseButton1Click:Connect(function()
                            pcall(writefile, configFolder .. '/autoload.txt', configName)
                        end)

                        deleteBtn.MouseButton1Click:Connect(function()
                            pcall(delfile, filePath)
                            refreshConfigList()
                        end)
                    end
                end
            end
        end
