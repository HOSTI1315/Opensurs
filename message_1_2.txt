# Initialize SQLite database
def init_db():
    conn = sqlite3.connect('keys.db')
    c = conn.cursor()
    
    # Create keys table
    c.execute('''CREATE TABLE IF NOT EXISTS keys 
                 (key TEXT PRIMARY KEY, 
                  hwid TEXT, 
                  discord_id TEXT, 
                  roblox_username TEXT, 
                  redeemed BOOLEAN, 
                  redeem_date TEXT)''')
    
    # Create whitelist table
    c.execute('''CREATE TABLE IF NOT EXISTS whitelist 
                 (discord_id TEXT PRIMARY KEY, 
                  hwid TEXT, 
                  roblox_username TEXT)''')
    
    conn.commit()
    conn.close()

init_db()

# Helper function to generate HWID hash
def generate_hwid(discord_id: str, roblox_username: str) -> str:
    data = f"{discord_id}{roblox_username}".encode('utf-8')
    return hashlib.sha256(data).hexdigest()

# Modal for "Redeem Key"
class RedeemKeyModal(Modal, title="🔑 Redeem Key"):
    key_input = TextInput(
        label="Enter Key",
        placeholder="Enter your key here...",
        required=True,
        style=discord.TextStyle.short,
    )
    
    roblox_username = TextInput(
        label="Roblox Username",
        placeholder="Enter your Roblox username...",
        required=True,
        style=discord.TextStyle.short,
    )

    async def on_submit(self, interaction: discord.Interaction):
        entered_key = self.key_input.value
        roblox_user = self.roblox_username.value
        
        conn = sqlite3.connect('keys.db')
        c = conn.cursor()
        
        c.execute("SELECT * FROM keys WHERE key = ? AND redeemed = 0", (entered_key,))
        key_data = c.fetchone()
        
        if not key_data:
            await interaction.response.send_message(
                "❌ Invalid or already redeemed key!", ephemeral=True
            )
            conn.close()
            return
        
        hwid = generate_hwid(str(interaction.user.id), roblox_user)
        current_time = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        c.execute("UPDATE keys SET redeemed = 1, hwid = ?, discord_id = ?, roblox_username = ?, redeem_date = ? WHERE key = ?",
                  (hwid, str(interaction.user.id), roblox_user, current_time, entered_key))
        
        c.execute("INSERT OR REPLACE INTO whitelist (discord_id, hwid, roblox_username) VALUES (?, ?, ?)",
                  (str(interaction.user.id), hwid, roblox_user))
        
        conn.commit()
        conn.close()
        
        await interaction.response.send_message(
            f"✅ Key `{entered_key}` has been successfully redeemed!\n"
            f"🔒 HWID Locked to your Roblox account: `{roblox_user}`\n"
            f"📅 Redeemed on: {current_time}\n\n"
            f"Now click the 'Get Role' button to receive your {BUYER_ROLE_NAME} role!",
            ephemeral=True
        )

# Modal for "Reset HWID"
class ResetHWIDModal(Modal, title="⚙️ Reset HWID"):
    roblox_username = TextInput(
        label="New Roblox Username",
        placeholder="Enter your new Roblox username...",
        required=True,
        style=discord.TextStyle.short,
    )

    async def on_submit(self, interaction: discord.Interaction):
        new_roblox_username = self.roblox_username.value
        discord_id = str(interaction.user.id)
        
        conn = sqlite3.connect('keys.db')
        c = conn.cursor()
        
        c.execute("SELECT hwid, roblox_username FROM whitelist WHERE discord_id = ?", (discord_id,))
        user_data = c.fetchone()
        
        if not user_data:
            await interaction.response.send_message(
                "❌ You need to redeem a key first to reset your HWID!", ephemeral=True
            )
            conn.close()
            return
        
        new_hwid = generate_hwid(discord_id, new_roblox_username)
        current_time = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        
        c.execute("UPDATE whitelist SET hwid = ?, roblox_username = ? WHERE discord_id = ?",
                  (new_hwid, new_roblox_username, discord_id))
        
        c.execute("UPDATE keys SET hwid = ?, roblox_username = ? WHERE discord_id = ? AND redeemed = 1",
                  (new_hwid, new_roblox_username, discord_id))
        
        conn.commit()
        conn.close()
        
        await interaction.response.send_message(
            f"✅ HWID successfully reset!\n"
            f"🔒 New HWID locked to Roblox account: `{new_roblox_username}`\n"
            f"📅 Reset on: {current_time}",
            ephemeral=True
        )

@client.tree.command(name="panel", description="Display the control panel.")

    **{project_name} Control Panel**

    get_role_button = Button(label="🥷 Get Role", style=discord.ButtonStyle.blurple)

                f"❌ You need the {BUYER_ROLE_NAME} role to get the script!", 
                ephemeral=True
            )
            return

        c = conn.cursor()
        c.execute("SELECT * FROM whitelist WHERE discord_id = ?", (str(interaction.user.id),))
        user_data = c.fetchone()
        conn.close()
        
        if user_data:
            await interaction.response.send_message(
                f"📜 Here's your script:\n```lua\n{script_content}\n```\n"
                f"🔒 HWID Locked to: `{user_data[2]}`",
                ephemeral=True
            )
        else:

                "❌ You need to redeem a key first!", ephemeral=True
            )

    async def reset_hwid_callback(interaction: discord.Interaction):
        await interaction.response.send_modal(ResetHWIDModal())

    async def get_role_callback(interaction: discord.Interaction):
        conn = sqlite3.connect('keys.db')
        c = conn.cursor()
        c.execute("SELECT * FROM whitelist WHERE discord_id = ?", (str(interaction.user.id),))
        user_data = c.fetchone()
        conn.close()
        
        if not user_data:
            await interaction.response.send_message(
                "❌ You need to redeem a key first before getting the role!", 
                ephemeral=True
            )
            return

            try:

                    name=BUYER_ROLE_NAME,
                    color=discord.Color.green(),

                )
            except discord.Forbidden:
                await interaction.response.send_message(
                    "❌ I don't have permission to create the Buyer role!",
                    ephemeral=True
                )
                return
        
        try:
            await interaction.user.add_roles(buyer_role)
            await interaction.response.send_message(
                f"✅ You have been given the {BUYER_ROLE_NAME} role!",
                ephemeral=True
            )
        except discord.Forbidden:
            await interaction.response.send_message(
                "❌ I don't have permission to add roles!",
                ephemeral=True
            )

    async def stats_callback(interaction: discord.Interaction):
        discord_id = str(interaction.user.id)
        conn = sqlite3.connect('keys.db')
        c = conn.cursor()
        
        c.execute("SELECT hwid, roblox_username FROM whitelist WHERE discord_id = ?", (discord_id,))
        whitelist_data = c.fetchone()
        
        if not whitelist_data:
            await interaction.response.send_message(
                "❌ You need to redeem a key first to view your stats!", ephemeral=True
            )
            conn.close()
            return
        
        c.execute("SELECT key, redeem_date FROM keys WHERE discord_id = ? AND redeemed = 1", (discord_id,))
        key_data = c.fetchone()

            )
        else:
            stats_message += "**Redeemed Key:** None\n"

        await interaction.response.send_message(stats_message, ephemeral=True)

    script_button.callback = script_callback
    reset_hwid_button.callback = reset_hwid_callback
    get_role_button.callback = get_role_callback

# Unwhitelist command group
unwhitelist_group = app_commands.Group(name="unwhitelist", description="Unwhitelist management commands.")

@unwhitelist_group.command(name="remove", description="Remove a user from the whitelist.")
async def unwhitelist_remove(interaction: discord.Interaction, user: discord.User):
    conn = sqlite3.connect('keys.db')
    c = conn.cursor()
    
    # Check if user is in whitelist
    c.execute("SELECT * FROM whitelist WHERE discord_id = ?", (str(user.id),))
    if not c.fetchone():
        await interaction.response.send_message(
            f"❌ User {user.name} is not in the whitelist!", ephemeral=True
        )
        conn.close()
        return
    
    # Remove from whitelist
    c.execute("DELETE FROM whitelist WHERE discord_id = ?", (str(user.id),))
    conn.commit()
    conn.close()
    
    # Remove Buyer role if they have it
    buyer_role = discord.utils.get(interaction.guild.roles, name=BUYER_ROLE_NAME)
    if buyer_role and buyer_role in user.roles:
        try:
            await user.remove_roles(buyer_role)
            await interaction.response.send_message(
                f"✅ User {user.name} has been removed from the whitelist and their {BUYER_ROLE_NAME} role has been revoked.",
                ephemeral=True
            )
        except discord.Forbidden:
            await interaction.response.send_message(
                f"✅ User {user.name} has been removed from the whitelist, but I don't have permission to remove their {BUYER_ROLE_NAME} role!",
                ephemeral=True
            )
        except discord.HTTPException:
            await interaction.response.send_message(
                f"✅ User {user.name} has been removed from the whitelist, but an error occurred while trying to remove their {BUYER_ROLE_NAME} role.",
                ephemeral=True
            )
    else:
        await interaction.response.send_message(
            f"✅ User {user.name} has been removed from the whitelist. They did not have the {BUYER_ROLE_NAME} role.",
            ephemeral=True
        )

# Key management command group
key_group = app_commands.Group(name="key", description="Key management commands.")

@key_group.command(name="generate", description="Generate a new key (Buy from role only).")
async def generate_key(interaction: discord.Interaction):
    buy_from_role = discord.utils.get(interaction.guild.roles, name=BUY_FROM_ROLE_NAME)
    if not buy_from_role or buy_from_role not in interaction.user.roles:
        await interaction.response.send_message(
            f"❌ You need the {BUY_FROM_ROLE_NAME} role to generate keys!", 
            ephemeral=True
        )
        return
    
    new_key = ''.join(random.choices(string.ascii_uppercase + string.digits, k=16))
    
    conn = sqlite3.connect('keys.db')
    c = conn.cursor()
    
    c.execute("INSERT INTO keys (key, redeemed) VALUES (?, 0)", (new_key,))
    
    conn.commit()
    conn.close()
    
    await interaction.response.send_message(
        f"🔑 New key generated: `{new_key}`\n"
        "This key can be redeemed once and will be locked to the user's HWID.",
        ephemeral=True
    )
