-- --- Logic Implementation ---

local function executeTeleport()
    local root = getRoot()
    if root and pos1 and pos2 then
        root.CFrame = pos1
        task.wait(0.1) -- Small delay to prevent physics glitches
        root.CFrame = pos2
    end
end

pos1Btn.MouseButton1Click:Connect(function()
    local root = getRoot()
    if root then
        pos1 = root.CFrame
        pos1Btn.Text = "✅ Position 1 Saved"
        task.wait(1)
        pos1Btn.Text = "Set Position 1"
    end
end)

pos2Btn.MouseButton1Click:Connect(function()
    local root = getRoot()
    if root then
        pos2 = root.CFrame
        pos2Btn.Text = "✅ Position 2 Saved"
        task.wait(1)
        pos2Btn.Text = "Set Position 2"
    end
end)

tpBtn.MouseButton1Click:Connect(executeTeleport)

bindBtn.MouseButton1Click:Connect(function()
    isBinding = true
    bindBtn.Text = "Press any key..."
end)

UserInputService.InputBegan:Connect(function(input, processed)
    if processed then return end
    
    if isBinding and input.UserInputType == Enum.UserInputType.Keyboard then
        currentKeybind = input.KeyCode
        bindBtn.Text = "⌨️ Bind TP: " .. input.KeyCode.Name
        isBinding = false
    elseif currentKeybind and input.KeyCode == currentKeybind then
        executeTeleport()
    end
end)

closeBtn.MouseButton1Click:Connect(function()
    mainFrame.Visible = false
    openBtn.Visible = true
end)

openBtn.MouseButton1Click:Connect(function()
    mainFrame.Visible = true
    openBtn.Visible = false
end)

RunService.RenderStepped:Connect(function()
    local hue = tick() % 5 / 5
    local color = Color3.fromHSV(hue, 0.8, 1)
    stroke1.Color = color
    stroke2.Color = color
    openBtn.BackgroundColor3 = color
end)
