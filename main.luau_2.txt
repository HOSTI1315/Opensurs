local function computeAcceleration(currentVel, currentPos, currentTime)
    if state.lastBallPos and state.lastBallTime then
        local dt = currentTime - state.lastBallTime
        if dt > 0.001 then
            local prevVel = state.velocitySamples[#state.velocitySamples] or currentVel
            state.acceleration = (currentVel - prevVel) / dt
        end
    end
    state.lastBallPos = currentPos
    state.lastBallTime = currentTime
end

local function getSmoothedVelocity(currentVel)
    table.insert(state.velocitySamples, currentVel)
    if #state.velocitySamples > state.maxSamples then
        table.remove(state.velocitySamples, 1)
    end
    
    local sum = Vector3.zero
    local weight = 0
    for i, v in ipairs(state.velocitySamples) do
        local w = i * i
        sum = sum + v * w
        weight = weight + w
    end
    
    return sum / math.max(weight, 1)
end

local function predictPosition(ballPos, vel, accel, time)
    return ballPos + vel * time + 0.5 * accel * time * time
end

local function calculateTimeToReach(ballPos, playerPos, vel, accel)
    local displacement = playerPos - ballPos
    local speed = vel.Magnitude
    
    if speed < CONFIG.MinSpeed then
        return math.huge
    end
    
    local direction = vel.Unit
    local projectedDist = displacement:Dot(direction)
    
    if projectedDist <= 0 then
        return math.huge
    end
    
    local accelComponent = accel:Dot(direction)
    
    if math.abs(accelComponent) > 0.1 then
        local a = 0.5 * accelComponent
        local b = speed
        local c = -projectedDist
        local discriminant = b * b - 4 * a * c
        
        if discriminant >= 0 then
            local t1 = (-b + math.sqrt(discriminant)) / (2 * a)
            local t2 = (-b - math.sqrt(discriminant)) / (2 * a)
            
            if t1 > 0 and t2 > 0 then
                return math.min(t1, t2)
            elseif t1 > 0 then
                return t1
            elseif t2 > 0 then
                return t2
            end
        end
    end
    
    return projectedDist / speed
end

local function isTargetingPlayer(ball, playerPos, vel, dist)
    if vel.Magnitude < CONFIG.MinSpeed then
        return false, 0
    end
    
    local targetAttr = ball:GetAttribute("target")
    if targetAttr == plr.Name then
        return true, 1
    end
    
    local targetValue = ball:FindFirstChild("Target")
    if targetValue then
        if targetValue:IsA("ObjectValue") and targetValue.Value == plr.Character then
            return true, 1
        elseif targetValue:IsA("StringValue") and targetValue.Value == plr.Name then
            return true, 1
        end
    end
    
    local direction = (playerPos - ball.Position).Unit
    local moveDir = vel.Unit
    local dotProduct = direction:Dot(moveDir)
    
    if dist < 12 then
        return true, 0.95
    end
    
    if dotProduct > CONFIG.AngleTolerance then
        local confidence = (dotProduct - CONFIG.AngleTolerance) / (1 - CONFIG.AngleTolerance)
        return true, math.clamp(confidence, 0.5, 1)
    end
    
    return false, 0
end

local function detectCurve(vel, playerPos, ballPos)
    if vel.Magnitude < CONFIG.MinSpeed then
        return false, 0
    end
    
    local direction = (playerPos - ballPos).Unit
    local moveDir = vel.Unit
    local angle = direction:Dot(moveDir)
    
    if angle < 0.92 and angle > 0.2 then
        local curveAmount = 1 - angle
        return true, curveAmount
    end
    
    return false, 0
end

local function calculateOptimalParryTime(dist, speed, accel, isCurved, curveAmount)
    local baseTime = dist / math.max(speed, CONFIG.MinSpeed)
    
    local accelMag = accel.Magnitude
    if accelMag > 1 then
        baseTime = baseTime * (1 - math.clamp(accelMag / 100, 0, 0.3))
    end
    
    if isCurved then
        baseTime = baseTime * (1 + curveAmount * CONFIG.CurveCompensation * 0.5)
    end
    
    local reactionOffset = 0.08 + (dist / 500)
    
    return math.max(baseTime - reactionOffset, 0)
end

local function shouldParry(ball, playerPos, vel, accel, dist)
    local targeted, confidence = isTargetingPlayer(ball, playerPos, vel, dist)
    
    if not targeted then
        return false
    end
    
    local speed = vel.Magnitude
    if speed < CONFIG.MinSpeed then
        return false
    end
    
    if dist > CONFIG.ParryDistance or dist < CONFIG.MinParryDist then
        return false
    end
    
    local isCurved, curveAmount = detectCurve(vel, playerPos, ball.Position)
    local timeToHit = calculateOptimalParryTime(dist, speed, accel, isCurved, curveAmount)
    
    local predictedPos = predictPosition(ball.Position, vel, accel, timeToHit)
    local predictedDist = (predictedPos - playerPos).Magnitude
    
    local parryWindow = 0.35
    if isCurved then
        parryWindow = parryWindow * (1 + curveAmount * 0.5)
    end
    
    local effectiveRange = CONFIG.ParryDistance * (0.8 + confidence * 0.2)
    
    if predictedDist <= effectiveRange and timeToHit <= CONFIG.MaxPredictionTime then
        local urgency = 1 - (timeToHit / CONFIG.MaxPredictionTime)
        
        if dist <= CONFIG.ParryDistance * 0.6 or urgency > 0.7 then
            return true
        end
    end
    
    if dist <= CONFIG.ParryDistance * 0.4 and speed > CONFIG.MinSpeed * 1.5 then
        return true
    end
    
    return false
end

local function executeParry()
    local currentTime = tick()
    
    if state.debounce then
        return false
    end
    
    if (currentTime - CONFIG.LastParryTime) < CONFIG.ParryCooldown then
        return false
    end
    
    state.debounce = true
    CONFIG.LastParryTime = currentTime
    
    local success = pcall(function()
        local remotes = ReplicatedStorage:FindFirstChild("RemoteEvents")
        if remotes then
            local parryRemote = remotes:FindFirstChild("ParryRequest")
            if parryRemote then
                parryRemote:FireServer(camera.CFrame, nil, 1/0)
            end
        end
    end)
    
    if success then
        state.parryCount = state.parryCount + 1
        parryCountVal.Text = tostring(state.parryCount)
        
        TweenService:Create(indicator, TweenInfo.new(0.1), {
            Size = UDim2.new(0, 18, 0, 18),
            Position = UDim2.new(0.5, -9, 0.5, -9)
        }):Play()
        
        task.delay(0.15, function()
            TweenService:Create(indicator, TweenInfo.new(0.2), {
                Size = UDim2.new(0, 12, 0, 12),
                Position = UDim2.new(0.5, -6, 0.5, -6)
            }):Play()
        end)
    end
    
    task.wait(CONFIG.DebounceTime)
    state.debounce = false
    
    return success
end

local function updateVisualizer(pos, range, targeted, confidence)
    if not CONFIG.VisualizerEnabled then
        visualizer.Parent = nil
        return
    end
    
    if not visualizer.Parent then
        visualizer.Parent = Workspace
    end
    
    local size = range * 2
    TweenService:Create(visualizer, TweenInfo.new(0.08, Enum.EasingStyle.Quad), {
        Size = Vector3.new(size, size, size),
        Position = pos
    }):Play()
    
    local targetColor = targeted and Color3.fromRGB(255, 80, 80) or Color3.fromRGB(80, 160, 255)
    local targetTransparency = targeted and (0.5 + (1 - confidence) * 0.3) or 0.75
    
    TweenService:Create(visualizer, TweenInfo.new(0.12), {
        Color = targetColor,
        Transparency = targetTransparency
    }):Play()
    
    TweenService:Create(visLight, TweenInfo.new(0.12), {
        Color = targetColor,
        Brightness = targeted and 2.5 or 1.5
    }):Play()
end

local function updateUserInterface(speed, dist, status, confidence)
    local t = tick()
    if t - state.lastUpdate < CONFIG.UpdateRate then
        return
    end
    state.lastUpdate = t
    
    speedVal.Text = string.format("%.0f st/s", speed)
    distVal.Text = string.format("%.1f studs", dist)
    rangeVal.Text = string.format("%d studs", CONFIG.ParryDistance)
    
    local statusColors = {
        TARGETED = {
            indicator = Color3.fromRGB(255, 70, 70),
            stroke = Color3.fromRGB(255, 70, 70),
            strokeTrans = 0.2
        },
        TRACKING = {
            indicator = Color3.fromRGB(255, 180, 70),
            stroke = Color3.fromRGB(255, 180, 70),
            strokeTrans = 0.35
        },
        IDLE = {
            indicator = Color3.fromRGB(80, 255, 120),
            stroke = Color3.fromRGB(60, 60, 75),
            strokeTrans = 0.4
        }
    }
    
    local colors = statusColors[status] or statusColors.IDLE
    
    TweenService:Create(indicator, TweenInfo.new(0.15), {
        BackgroundColor3 = colors.indicator
    }):Play()
    
    TweenService:Create(indGlow, TweenInfo.new(0.15), {
        ImageColor3 = colors.indicator
    }):Play()
    
    TweenService:Create(stroke, TweenInfo.new(0.15), {
        Color = colors.stroke,
        Transparency = colors.strokeTrans
    }):Play()
    
    TweenService:Create(innerGlow, TweenInfo.new(0.2), {
        BackgroundColor3 = colors.indicator,
        BackgroundTransparency = status == "TARGETED" and 0.92 or 0.96
    }):Play()
end

local function resetState()
    speedVal.Text = "0 st/s"
    distVal.Text = "---"
    rangeVal.Text = string.format("%d studs", CONFIG.ParryDistance)
    
    TweenService:Create(indicator, TweenInfo.new(0.2), {
        BackgroundColor3 = Color3.fromRGB(80, 255, 120)
    }):Play()
    
    TweenService:Create(indGlow, TweenInfo.new(0.2), {
        ImageColor3 = Color3.fromRGB(80, 255, 120)
    }):Play()
    
    TweenService:Create(stroke, TweenInfo.new(0.2), {
        Color = Color3.fromRGB(60, 60, 75),
        Transparency = 0.4
    }):Play()
    
    visualizer.Parent = nil
    state.velocitySamples = {}
    state.positionSamples = {}
    state.lastBallPos = nil
    state.lastBallTime = nil
    state.acceleration = Vector3.zero
end

RunService.RenderStepped:Connect(function()
    local ball = getBall()
    local char = plr.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    
    if not ball or not root then
        resetState()
        return
    end
    
    local ballPos = ball.Position
    local playerPos = root.Position
    local rawVel = ball.AssemblyLinearVelocity
    local currentTime = tick()
    
    computeAcceleration(rawVel, ballPos, currentTime)
    local smoothedVel = getSmoothedVelocity(rawVel)
    
    local speed = smoothedVel.Magnitude
    local dist = (ballPos - playerPos).Magnitude
    
    local targeted, confidence = isTargetingPlayer(ball, playerPos, smoothedVel, dist)
    
    local status = "IDLE"
    if speed > CONFIG.MinSpeed then
        status = targeted and "TARGETED" or "TRACKING"
    end
    
    updateVisualizer(playerPos, CONFIG.ParryDistance, targeted, confidence)
    
    if shouldParry(ball, playerPos, smoothedVel, state.acceleration, dist) then
        executeParry()
    end
    
    updateUserInterface(speed, dist, status, confidence)
end)
