local Library = {}

function Library.validate(defaults, options)
	for key, value in pairs(defaults) do
		if options[key] == nil then
			options[key] = value
		end
	end
	return options
end

function Library.tween(object, properties, callback)
	local tween = TweenService:Create(object, TweenInfo_Default, properties)
	tween.Completed:Connect(callback or function() end)
	tween:Play()
end

				end
				Tab.Active = true
				Library:tween(Tab["17"], {
					BackgroundTransparency = 0.8,
				})
				Library:tween(Tab["17"], {
					TextColor3 = Color3.fromRGB(255, 255, 255),
				})

		AutoTrainEnabled = false

			coroutine.yield(AutoTrainThread)
			AutoTrainThread = nil
		end
	end
end

local PlayersService2 = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local LocalPlayer2 = PlayersService2.LocalPlayer
local RunService2 = game:GetService("RunService")
local TweenService2 = game:GetService("TweenService")

local function GetNearestPlayer()
	local minDistance = math.huge
	local nearestPlayer = nil
	for _, player in pairs(PlayersService2:GetPlayers()) do
		if player ~= LocalPlayer2 then
			local character = player.Character
			if character and character:FindFirstChild("HumanoidRootPart") then
				local distance = (character.HumanoidRootPart.Position - LocalPlayer2.Character.HumanoidRootPart.Position).Magnitude
				if distance < minDistance then
					minDistance = distance
					nearestPlayer = player
				end
			end
		end
	end
	return nearestPlayer, minDistance
end

local function AttackPlayer(target)
	if target and target.Character and target.Character:FindFirstChild("Humanoid") then
		ReplicatedStorage.common.packages._Index:FindFirstChild("sleitnick_knit@1.5.1").knit.Services.OlympicBoxingService.RF.Attack:InvokeServer(unpack({
			[1] = target.Character.Humanoid,
		}))
	end
end

local function TweenToPlayer(target)
	if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
		local targetPos = target.Character.HumanoidRootPart.Position + Vector3.new(0, 20, 0)
		local root = LocalPlayer2.Character and LocalPlayer2.Character:FindFirstChild("HumanoidRootPart")
		if root then
			TweenService2:Create(root, TweenInfo.new(1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				CFrame = CFrame.new(targetPos),
			}):Play()
		end
	end
end

local VIM3 = game:GetService("VirtualInputManager")
local AutoDeadliftEnabled = false
local AutoDeadliftThread = nil

local function PressSpace()
	VIM3:SendKeyEvent(true, Enum.KeyCode.Space, false, game)
	VIM3:SendKeyEvent(false, Enum.KeyCode.Space, false, game)
end

local function AutoDeadliftLoop()
	while AutoDeadliftEnabled do
		PressSpace()
		wait(0.00001)
	end
end

local function StartAutoDeadlift()
	if not AutoDeadliftEnabled then
		AutoDeadliftEnabled = true
		AutoDeadliftThread = coroutine.create(AutoDeadliftLoop)
		coroutine.resume(AutoDeadliftThread)
	end
end

local function StopAutoDeadlift()
	if AutoDeadliftEnabled then
		AutoDeadliftEnabled = false
		if AutoDeadliftThread then
			coroutine.yield(AutoDeadliftThread)
			AutoDeadliftThread = nil
		end
	end
end

	title = "Auto Deadlift",
	callback = function(state)
		if state then
			StartAutoDeadlift()
		else
			StopAutoDeadlift()
		end
	end,
})
