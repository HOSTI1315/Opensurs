local function makeDrag(obj, handle)
	local dragging, dragStart, startPos
	handle = handle or obj
	
	handle.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = obj.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)
	
	UIS.InputChanged:Connect(function(input)
		if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			local d = input.Position - dragStart
			obj.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + d.X, startPos.Y.Scale, startPos.Y.Offset + d.Y)
		end
	end)
end

local function setupHover(btn, c1, c2)
	btn.MouseEnter:Connect(function()
		TweenService:Create(btn, TweenInfo.new(0.12), {BackgroundColor3 = c2}):Play()
	end)
	btn.MouseLeave:Connect(function()
		local c = c1
		if btn == saveBtn then
			c = tpMode and Color3.fromRGB(45, 100, 170) or Color3.fromRGB(35, 75, 140)
		end
		TweenService:Create(btn, TweenInfo.new(0.12), {BackgroundColor3 = c}):Play()
	end)
end

local function toggleHide()
	hidden = not hidden
	if hidden then
		TweenService:Create(main, TweenInfo.new(0.2), {Size = UDim2.new(0, 0, 0, 0), Position = main.Position + UDim2.new(0, 97, 0, 107)}):Play()
		task.wait(0.2)
		main.Visible = false
		miniBtn.Visible = true
		TweenService:Create(miniBtn, TweenInfo.new(0.15), {Size = UDim2.new(0, 42, 0, 42)}):Play()
	else
		miniBtn.Visible = false
		main.Visible = true
		main.Size = UDim2.new(0, 0, 0, 0)
		TweenService:Create(main, TweenInfo.new(0.2), {Size = UDim2.new(0, 195, 0, 215), Position = main.Position - UDim2.new(0, 97, 0, 107)}):Play()
	end
end

local function updateSlots()
	for i = 1, 3 do
		local sl = slotData[i]
		if i <= #saved then
			sl.f.BackgroundColor3 = Color3.fromRGB(40, 110, 160)
			sl.s.Color = Color3.fromRGB(70, 150, 200)
			sl.n.TextColor3 = Color3.fromRGB(220, 240, 255)
		else
			sl.f.BackgroundColor3 = Color3.fromRGB(28, 38, 55)
			sl.s.Color = Color3.fromRGB(45, 65, 95)
			sl.n.TextColor3 = Color3.fromRGB(70, 100, 140)
		end
	end
end

local function updateBtn()
	if #saved >= 3 then
		tpMode = true
		saveBtn.Text = "TELEPORT"

		saveBtnStroke.Color = Color3.fromRGB(80, 140, 210)

		saveBtn.BackgroundColor3 = Color3.fromRGB(35, 75, 140)
		saveBtnStroke.Color = Color3.fromRGB(60, 110, 180)

local function equipCarpet()
	if not check() then return end
	local bp = plr:FindFirstChild("Backpack")
	local char = plr.Character
	if not bp or not char then return end
	
	local hum = char:FindFirstChild("Humanoid")
	if not hum then return end
	
	for _, t in pairs(bp:GetChildren()) do
		if t:IsA("Tool") then
			local nm = t.Name:lower()
			if nm:find("carpet") or nm:find("flying") then
				hum:EquipTool(t)
				task.wait(0.2)
				return
			end
		end
	end
end

local function save()
	if not check() then return end
	if tping or #saved >= 3 then return end
	local char = plr.Character
	local hrp = char and char:FindFirstChild("HumanoidRootPart")
	if not hrp then return end
	
	table.insert(saved, hrp.Position)
	local m = makeMarker(hrp.Position, #saved)
	if m then table.insert(markers, m) end
	updateBtn()
end

local function tp(pos)
	if not check() then return end
	local char = plr.Character
	local hrp = char and char:FindFirstChild("HumanoidRootPart")
	if not hrp then return end
	hrp.CFrame = CFrame.new(pos + Vector3.new(0, 4, 0))
end

local function autoTp()
	if not check() then return end
	if tping or #saved == 0 then return end
	
	tping = true
	saveBtn.Text = "..."
	saveBtn.BackgroundColor3 = Color3.fromRGB(140, 120, 50)
	
	equipCarpet()
	task.wait(0.25)
	
	for i = 1, #saved do
		if not tping or not check() then break end
		saveBtn.Text = i .. "/" .. #saved
		tp(saved[i])
		if i < #saved then task.wait(DELAY) end
	end
	
	saveBtn.Text = "DONE"
	task.wait(0.5)
	
	tping = false
	updateBtn()
end

local function reset()
	if not check() then return end
	tping = false
	saved = {}
	for _, m in pairs(markers) do
		if m then m:Destroy() end
	end
	markers = {}
	updateBtn()
end

saveBtn.MouseButton1Click:Connect(function()
	if not check() then return end
	if tping then return end
	if tpMode then autoTp() else save() end
end)

resetBtn.MouseButton1Click:Connect(reset)

updateBtn()
