local p=game.Players.LocalPlayer
local rs=game:GetService("RunService")
local fts={}
local base

rs.RenderStepped:Connect(function(d)
    table.insert(fts,d)
    if #fts>60 then table.remove(fts,1) end
    local sum=0 for _,v in ipairs(fts) do sum=sum+v end
    local avg=#fts/sum
    if not base and #fts>=60 then base=avg return end
    if base and avg<base*0.7 then p:Kick("Lagger detected") end
end)
