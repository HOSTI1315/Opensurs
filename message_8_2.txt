local function FlingNearbyPlayers()
    local Character = LocalPlayer.Character
    if not Character then return end
    local RootPart = Character:FindFirstChild("HumanoidRootPart")
    if not RootPart then return end
    
    for _, Player in pairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer and Player.Character then
            local TargetRoot = Player.Character:FindFirstChild("HumanoidRootPart")
            if TargetRoot and (RootPart.Position - TargetRoot.Position).Magnitude < 20 then
                TargetRoot.Velocity = Vector3.new(math.random(-100, 100), math.random(100, 200), math.random(-100, 100))
            end
        end
    end
end

local function PerformImprovedAttack()
    VirtualInputManager:SendMouseButtonEvent(0, 0, 0, true, game, 1)
    task.wait(0.05)
    VirtualInputManager:SendMouseButtonEvent(0, 0, 0, false, game, 1)
    
    for _, Key in pairs({Enum.KeyCode.E, Enum.KeyCode.F, Enum.KeyCode.Q, Enum.KeyCode.R}) do
        VirtualInputManager:SendKeyEvent(true, Key, false, game)
        task.wait(0.02)
        VirtualInputManager:SendKeyEvent(false, Key, false, game)
    end
end

local function ImprovedKillauraRoutine()
    local Character = LocalPlayer.Character
    if not Character then return end
    local RootPart = Character:FindFirstChild("HumanoidRootPart")
    if not RootPart then return end

    for _, Player in pairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer and Player.Character then
            local TargetChar = Player.Character
            local TargetRoot = TargetChar:FindFirstChild("HumanoidRootPart")
            local TargetHead = TargetChar:FindFirstChild("Head")

            if TargetRoot and TargetHead then
                if (RootPart.Position - TargetRoot.Position).Magnitude <= 25 * HitboxMultiplier then
                    if TargetHead:FindFirstChild("OriginalSize") == nil then
                        local SizeValue = Instance.new("Vector3Value")
                        SizeValue.Name = "OriginalSize"
                        SizeValue.Value = TargetHead.Size
                        SizeValue.Parent = TargetHead
                    end
                    TargetHead.Size = TargetHead.OriginalSize.Value * HeadSizeMultiplier
                    PerformImprovedAttack()
                elseif TargetHead:FindFirstChild("OriginalSize") then
                    TargetHead.Size = TargetHead.OriginalSize.Value
                end
            end
        end
    end
end

local function ResetHeadSizes()
    for _, Player in pairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer and Player.Character then
            local Head = Player.Character:FindFirstChild("Head")
            if Head and Head:FindFirstChild("OriginalSize") then
                Head.Size = Head.OriginalSize.Value
            end
        end
    end
end

local function ApplyImprovedInvisibility()
    local Character = LocalPlayer.Character
    if not Character then return end
    for _, Part in pairs(Character:GetDescendants()) do
        if Part:IsA("BasePart") then
            Part.Transparency = 1
            if Part:FindFirstChildOfClass("SurfaceAppearance") then
                Part:FindFirstChildOfClass("SurfaceAppearance"):Destroy()
            end
        elseif Part:IsA("Decal") then
            Part.Transparency = 1
        end
    end
    local Humanoid = Character:FindFirstChild("Humanoid")
    if Humanoid then
        Humanoid.DisplayDistanceType = Enum.HumanoidDisplayDistanceType.None
    end
end

local function RemoveImprovedInvisibility()
    local Character = LocalPlayer.Character
    if not Character then return end
    for _, Part in pairs(Character:GetDescendants()) do
        if Part:IsA("BasePart") then
            Part.Transparency = 0
        elseif Part:IsA("Decal") then
            Part.Transparency = 0
        end
    end
    local Humanoid = Character:FindFirstChild("Humanoid")
    if Humanoid then
        Humanoid.DisplayDistanceType = Enum.HumanoidDisplayDistanceType.Viewer
    end
end

local function CreateKillauraHitboxes()
    for _, Player in pairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer and Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") then
            local Box = Instance.new("SelectionBox")
            Box.Adornee = Player.Character.HumanoidRootPart
            Box.LineThickness = 0.1
            Box.Color3 = Color3.new(1, 0, 0)
            Box.Parent = Player.Character.HumanoidRootPart
            table.insert(KillauraHitboxes, Box)
            Player.Character.HumanoidRootPart.Size = Vector3.new(KillauraRange, KillauraRange, KillauraRange)
            Player.Character.HumanoidRootPart.Transparency = 0.5
            Player.Character.HumanoidRootPart.CanCollide = false
        end
    end
end

local function RemoveKillauraHitboxes()
    for _, Box in pairs(KillauraHitboxes) do
        if Box.Adornee then
            Box.Adornee.Size = Vector3.new(2, 2, 1)
            Box.Adornee.Transparency = 1
            Box:Destroy()
        end
    end
    KillauraHitboxes = {}
end

CombatGroup:AddToggle("AimAssist", {
    Text = "🎯 Aim Assist",
    Tooltip = "Auto aim at nearest player",
    Default = false,
    Callback = function(Value)
        AimAssistEnabled = Value
        if Value then
            if AimAssistConnection then AimAssistConnection:Disconnect() end
            AimAssistConnection = RunService.RenderStepped:Connect(function()
                if not AimAssistEnabled then return end
                local Character = LocalPlayer.Character
                if not Character then return end
                local RootPart = Character:FindFirstChild("HumanoidRootPart")
                local Humanoid = Character:FindFirstChild("Humanoid")
                
                if not RootPart or not Humanoid or Humanoid.Health <= 0 then
                    AimTarget = nil
                    return
                end

                local ClosestTarget = nil
                local ClosestDist = AimRange
                
                for _, Player in pairs(Players:GetPlayers()) do
                    if Player ~= LocalPlayer and Player.Character then
                        local TargetRoot = Player.Character:FindFirstChild("HumanoidRootPart")
                        local TargetHumanoid = Player.Character:FindFirstChild("Humanoid")
                        if TargetRoot and TargetHumanoid and TargetHumanoid.Health > 0 then
                            local Dist = (RootPart.Position - TargetRoot.Position).Magnitude
                            if Dist < ClosestDist then
                                ClosestDist = Dist
                                ClosestTarget = Player
                            end
                        end
                    end
                end
                
                AimTarget = ClosestTarget
                if AimTarget and AimTarget.Character then
                    local TargetRoot = AimTarget.Character:FindFirstChild("HumanoidRootPart")
                    if TargetRoot then
                        local Direction = TargetRoot.Position - RootPart.Position
                        Direction = Vector3.new(Direction.X, 0, Direction.Z)
                        if Direction.Magnitude > 0 then
                            RootPart.CFrame = RootPart.CFrame:Lerp(CFrame.new(RootPart.Position, RootPart.Position + Direction.Unit), RotationSpeed)
                        end
                    end
                end
            end)
        else
            if AimAssistConnection then
                AimAssistConnection:Disconnect()
                AimAssistConnection = nil
            end
            AimTarget = nil
        end
    end,
})

CombatGroup:AddToggle("AutoDodge", {
    Text = "AUTO DODGE",
    Tooltip = "Auto dodge attacks",
    Default = false,
    Callback = function(Value)
        AutoDodgeEnabled = Value
        local DodgePart = nil
        local DodgeHumanoid = nil
        local DodgeDistance = 4
        local DodgeHeight = 75
        local DangerousTools = {"BOTTLE", "KNIFE", "FORK"}

        local function GetDodgeTool()
            local Backpack = LocalPlayer:FindFirstChild("Backpack")
            if Backpack and Backpack:FindFirstChild("DODGE!") then
                return Backpack:FindFirstChild("DODGE!")
            end
            local Character = LocalPlayer.Character
            if Character and Character:FindFirstChild("DODGE!") then
                return Character:FindFirstChild("DODGE!")
            end
            return nil
        end

        local function ActivateDodge()
            local Tool = GetDodgeTool()
            if Tool and Tool:FindFirstChild("RemoteEvent") then
                pcall(function()
                    Tool.RemoteEvent:FireServer()
                end)
                return true
            end
            return false
        end

        local function MoveUp()
            if DodgePart then
                DodgePart.CFrame = DodgePart.CFrame + Vector3.new(0, DodgeHeight, 0)
            end
        end

        local function IsInDanger()
            for _, Player in ipairs(Players:GetPlayers()) do
                if Player ~= LocalPlayer and Player.Character then
                    local Character = Player.Character
                    local RootPart = Character:FindFirstChild("HumanoidRootPart")
                    if RootPart and (RootPart.Position - DodgePart.Position).Magnitude <= DodgeDistance then
                        for _, Item in ipairs(Character:GetChildren()) do
                            if Item:IsA("Tool") and table.find(DangerousTools, Item.Name:upper()) then
                                return true
                            end
                        end
                    end
                end
            end
            return false
        end

        if AutoDodgeEnabled then
            task.spawn(function()
                while AutoDodgeEnabled do
                    task.wait(0.15)
                    local Character = LocalPlayer.Character
                    DodgePart = Character and Character:FindFirstChild("HumanoidRootPart")
                    DodgeHumanoid = Character and Character:FindFirstChildOfClass("Humanoid")
                    
                    if DodgePart and DodgeHumanoid and DodgeHumanoid.Health > 0 and IsInDanger() and not ActivateDodge() then
                        MoveUp()
                    end
                end
            end)
        end
    end,
})

local function GetPlayerNames()
    local Names = {}
    for _, Player in pairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer then
            table.insert(Names, Player.DisplayName)
        end
    end
    if #Names == 0 then Names = {"No Players"} end
    return Names
end

local function GetPlayerByName(Name)
    for _, Player in pairs(Players:GetPlayers()) do
        if Player.DisplayName == Name then
            return Player
        end
    end
    return nil
end

local function TeleportToPlayer(Target)
    if not Target or not Target.Character then return end
    local TargetRoot = Target.Character:FindFirstChild("HumanoidRootPart")
    local MyRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if TargetRoot and MyRoot then
        MyRoot.CFrame = TargetRoot.CFrame * CFrame.new(0, 0, 3)
    end
end

local function TeleportToRandomPlayer()
    local AlivePlayers = {}
    for _, Player in pairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer and Player.Character then
            table.insert(AlivePlayers, Player)
        end
    end
    if #AlivePlayers > 0 then
        TeleportToPlayer(AlivePlayers[math.random(1, #AlivePlayers)])
    end
end

local function FindNearestAlivePlayer()
    local Nearest = nil
    local MinDist = math.huge
    local MyRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not MyRoot then return nil end

    for _, Player in pairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer and Player.Character then
            local Root = Player.Character:FindFirstChild("HumanoidRootPart")
            local Hum = Player.Character:FindFirstChild("Humanoid")
            if Root and Hum and Hum.Health > 0 then
                local Dist = (MyRoot.Position - Root.Position).Magnitude
                if Dist < MinDist then
                    MinDist = Dist
                    Nearest = Player
                end
            end
        end
    end
    return Nearest
end

local function ImprovedAntiFallRoutine()
    local Character = LocalPlayer.Character
    if not Character then return end
    local Root = Character:FindFirstChild("HumanoidRootPart")
    local Hum = Character:FindFirstChild("Humanoid")
    if not Root or not Hum or Hum.Health <= 0 then return end
    
    if Root.Position.Y < 10 then
        local Target = FindNearestAlivePlayer()
        if Target and Target.Character then
            local TargetRoot = Target.Character:FindFirstChild("HumanoidRootPart")
            if TargetRoot then
                Root.CFrame = TargetRoot.CFrame + Vector3.new(0, 3, 0)
                task.spawn(function()
                    local Part = Instance.new("Part")
                    Part.Size = Vector3.new(8, 0.2, 8)
                    Part.Position = Root.Position - Vector3.new(0, 3, 0)
                    Part.BrickColor = BrickColor.new("Bright green")
                    Part.Material = Enum.Material.Neon
                    Part.Anchored = true
                    Part.CanCollide = false
                    Part.Transparency = 0.5
                    Part.Parent = Workspace
                    Debris:AddItem(Part, 0.5)
                end)
            end
        end
    end
end

local function CreateGlassBridgeCover()
    local Bridge = Workspace:FindFirstChild("GlassBridge")
    if not Bridge then return end
    local Holder = Bridge:FindFirstChild("GlassHolder")
    if not Holder then return end
    
    local Children = Holder:GetChildren()
    if #Children == 0 then return end
    
    local MinX, MinY, MinZ = math.huge, math.huge, math.huge
    local MaxX, MaxY, MaxZ = -math.huge, -math.huge, -math.huge
    
    for _, Child in ipairs(Children) do
        if Child:IsA("Model") or Child:IsA("BasePart") then
            local CF, Size
            if Child:IsA("Model") then
                CF, Size = Child:GetBoundingBox()
            else
                CF = Child.CFrame
                Size = Child.Size
            end
            local Half = Size / 2
            for _, Corner in ipairs({
                CF * CFrame.new(-Half.X, -Half.Y, -Half.Z), CF * CFrame.new(Half.X, -Half.Y, -Half.Z),
                CF * CFrame.new(-Half.X, Half.Y, -Half.Z), CF * CFrame.new(Half.X, Half.Y, -Half.Z),
                CF * CFrame.new(-Half.X, -Half.Y, Half.Z), CF * CFrame.new(Half.X, -Half.Y, Half.Z),
                CF * CFrame.new(-Half.X, Half.Y, Half.Z), CF * CFrame.new(Half.X, Half.Y, Half.Z)
            }) do
                local Pos = Corner.Position
                MinX, MinY, MinZ = math.min(MinX, Pos.X), math.min(MinY, Pos.Y), math.min(MinZ, Pos.Z)
                MaxX, MaxY, MaxZ = math.max(MaxX, Pos.X), math.max(MaxY, Pos.Y), math.max(MaxZ, Pos.Z)
            end
        end
    end
    
    local Part = Instance.new("Part")
    Part.Name = "GlassBridgeCover"
    Part.Anchored = true
    Part.CanCollide = true
    Part.Material = Enum.Material.SmoothPlastic
    Part.Color = Color3.fromRGB(100, 100, 255)
    Part.Transparency = 0.3
    Part.Size = Vector3.new(MaxX - MinX + 2, MaxY - MinY + 2, MaxZ - MinZ + 2)
    Part.CFrame = CFrame.new((MinX + MaxX) / 2, (MinY + MaxY) / 2, (MinZ + MaxZ) / 2)
    Part.Parent = Workspace
    return Part
end

local function AutoSkySquidQTERoutine()
    local PGui = LocalPlayer.PlayerGui
    if not PGui then return end
    local Gui = PGui:FindFirstChild("ScreenGui") or PGui:FindFirstChild("QTE")
    if Gui then
        for _, Descendant in pairs(Gui:GetDescendants()) do
            if Descendant:IsA("TextButton") or Descendant:IsA("ImageButton") then
                local Text = Descendant.Text or Descendant.Name
                if Text:match("[FEQR]") or Descendant:FindFirstChild("QTE") then
                    local YPos = Descendant.AbsolutePosition.Y + Descendant.AbsoluteSize.Y / 2
                    if YPos > 350 and YPos < 450 then
                        local Key = nil
                        if Text:find("F") then Key = Enum.KeyCode.F
                        elseif Text:find("E") then Key = Enum.KeyCode.E
                        elseif Text:find("Q") then Key = Enum.KeyCode.Q
                        elseif Text:find("R") then Key = Enum.KeyCode.R
                        else Key = Enum.KeyCode.F end
                        
                        if Key then
                            VirtualInputManager:SendKeyEvent(true, Key, false, game)
                            task.wait(0.05)
                            VirtualInputManager:SendKeyEvent(false, Key, false, game)
                        end
                    end
                end
            end
        end
    end
end

            for _, Player in pairs(Players:GetChildren()) do
                if Player ~= LocalPlayer and LocalPlayer.Character and Player.Character then
                    local MyRoot = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")

                    if MyRoot and TRoot then

                            MinDist = Dist
                            TargetChar = Player.Character

                    end
                end
            end
            
            if TargetChar then
                while true do
                    task.wait()
                    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                        local TRoot = TargetChar:FindFirstChild("HumanoidRootPart")
                        if TRoot then
                            if _G.TeleportPlayerAuto then
                                local MoveDir = TargetChar.Humanoid.MoveDirection.Magnitude
                                if MoveDir > 0 then
                                    LocalPlayer.Character.HumanoidRootPart.CFrame = TRoot.CFrame * CFrame.new(0, 0, -7)
                                else
                                    LocalPlayer.Character.HumanoidRootPart.CFrame = TRoot.CFrame
                                end
                            else
                                if not _G.TeleportPlayerAuto then
                                    Workspace.CurrentCamera.CFrame = CFrame.lookAt(Workspace.CurrentCamera.CFrame.Position, Workspace.CurrentCamera.CFrame.Position + (TRoot.Position - Workspace.CurrentCamera.CFrame.Position).Unit)
                                end
                            end
                        end
                    end
                    
                    if not _G.CamlockPlayer or (TargetChar:FindFirstChild("Humanoid") and TargetChar.Humanoid.Health <= 0) then
                        break
                    end
                end
            end
            task.wait()
        end
    end,
})

local function HasTool(Name)
    if LocalPlayer.Character then
        for _, Item in pairs(LocalPlayer.Character:GetChildren()) do
            if Item:IsA("Tool") and Item.Name == Name then return true end
        end
    end
    if LocalPlayer.Backpack then
        for _, Item in pairs(LocalPlayer.Backpack:GetChildren()) do
            if Item:IsA("Tool") and Item.Name == Name then return true end
        end
    end
    return false
end

end
