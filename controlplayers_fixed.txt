local UseItemRemote = nil
local function GetUseItemRemote()
    if UseItemRemote then return UseItemRemote end
    
    local success, remote = pcall(function()
        local packages = ReplicatedStorage:FindFirstChild("Packages")
        if packages then
            local net = packages:FindFirstChild("Net")
            if net then
                local reUseItem = net:FindFirstChild("RE/UseItem")
                if reUseItem and reUseItem:IsA("RemoteEvent") then
                    return reUseItem
                end
                local reFolder = net:FindFirstChild("RE")
                if reFolder then
                    local useItem = reFolder:FindFirstChild("UseItem")
                    if useItem and useItem:IsA("RemoteEvent") then
                        return useItem
                    end
                end
            end
        end
        return nil
    end)
    
    if success and remote then
        UseItemRemote = remote
        return UseItemRemote
    end
    
    local fallbackSuccess, fallbackRemote = pcall(function()
        return ReplicatedStorage:FindFirstChild("UseItem", true)
    end)
    
    if fallbackSuccess and fallbackRemote and fallbackRemote:IsA("RemoteEvent") then
        UseItemRemote = fallbackRemote
        return UseItemRemote
    end
    
    return nil
end

local PlayerMouseModule = nil

            end
        end
        return nil
    end)
    if success and result then
        PlayerMouseModule = result
        return PlayerMouseModule
    end
    return nil
end

end

local function EnsureTool(toolName)
    if not toolName then return false end
    if not Character then return false end
    
    local hasTool = false
    if LocalPlayer.Backpack then
        hasTool = LocalPlayer.Backpack:FindFirstChild(toolName) ~= nil
    end
    if not hasTool then
        hasTool = Character:FindFirstChild(toolName) ~= nil
    end
    
    if not hasTool and Net and Config.AutoEquipTools then
        pcall(function()
            Net:RemoteFunction("CoinsShopService/RequestBuy"):InvokeServer(toolName)
        end)
    end
    
    local tool = LocalPlayer.Backpack and LocalPlayer.Backpack:FindFirstChild(toolName)
    local humanoid = tool and Character and Character:FindFirstChildOfClass("Humanoid")
    if humanoid and tool then
        pcall(function()
            humanoid:EquipTool(tool)
        end)
        if Character:FindFirstChild(toolName) then
            return true
        end
    elseif Character:FindFirstChild(toolName) then
        return true
    end
    
    return false
end

local function IsPlayerAllowed(player)
    if not player then return false end
    
    if Config.BlacklistEnabled then
        for _, blocked in ipairs(Config.Blacklist) do
            if type(blocked) == "number" then
                if player.UserId == blocked then
                    return false, "Player is blacklisted"
                end
            elseif type(blocked) == "string" then
                if player.Name:lower():find(blocked:lower()) then
                    return false, "Player is blacklisted"
                end
            end
        end
    end
    
    if Config.WhitelistEnabled then
        local isWhitelisted = false
        for _, allowed in ipairs(Config.Whitelist) do
            if type(allowed) == "number" then
                if player.UserId == allowed then
                    isWhitelisted = true
                    break
                end
            elseif type(allowed) == "string" then
                if player.Name:lower():find(allowed:lower()) then
                    isWhitelisted = true
                    break
                end
            end
        end
        if not isWhitelisted then
            return false, "Player is not whitelisted"
        end
    end
    
    return true
end

local function GetPlayerList()
    local playerList = {}
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local hrp = player.Character:FindFirstChild("HumanoidRootPart")
            if hrp then
                local allowed, reason = IsPlayerAllowed(player)
                if allowed then
                    table.insert(playerList, {
                        Player = player,
                        Name = player.Name,
                        DisplayName = player.DisplayName,
                        UserId = player.UserId,
                        Distance = Character and HumanoidRootPart and (hrp.Position - HumanoidRootPart.Position).Magnitude or math.huge
                    })
                end
            end
        end
    end
    
    table.sort(playerList, function(a, b)
        return a.Distance < b.Distance
    end)
    
    return playerList
end

local function FindNearestPlayer()
    local playerList = GetPlayerList()
    if #playerList > 0 then
        return playerList[1].Player
    end
    return nil
end

local function ControlPlayer(targetPlayer)
    if IsControlling then
        ShowNotification("Already controlling a player!", 2)
        return false
    end
    
    if not targetPlayer or not targetPlayer.Character then
        ShowNotification("Invalid target player!", 2)
        return false
    end
    
    local allowed, reason = IsPlayerAllowed(targetPlayer)
    if not allowed then
        ShowNotification(reason or "Player not allowed", 2)
        return false
    end
    
    local targetCharacter = targetPlayer.Character
    local targetHRP = targetCharacter:FindFirstChild("HumanoidRootPart")
    if not targetHRP then
        ShowNotification("Target missing HumanoidRootPart!", 2)
        return false
    end
    
    local myCharacter = LocalPlayer.Character
    if not myCharacter then
        ShowNotification("Your character not found!", 2)
        return false
    end
    
    local characterSounds = LocalPlayer.PlayerScripts:FindFirstChild("RbxCharacterSounds")
    if characterSounds then
        characterSounds.Disabled = true
    end
    
    local useItemRemote = GetUseItemRemote()
    local playerMouse = GetPlayerMouse()
    
    if not useItemRemote then
        ShowNotification("Failed to find UseItem remote!", 1)
        return false
    end
    
    EnsureTool(Config.WebSlingerTool)
    local webSlinger = Character:FindFirstChild(Config.WebSlingerTool) or LocalPlayer.Backpack:FindFirstChild(Config.WebSlingerTool)
    if webSlinger and webSlinger.Parent == LocalPlayer.Backpack then
        local humanoid = Character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid:EquipTool(webSlinger)
        end
    end
    
    if not Character:FindFirstChild(Config.WebSlingerTool) then
        ShowNotification("Web Slinger not equipped!", 1)
        if characterSounds then
            characterSounds.Disabled = false
        end
        return false
    end
    
    local function FireWebSlingAtCurrentPosition()
        if not targetHRP or not targetHRP.Parent or not targetCharacter or not targetCharacter.Parent then
            return false
        end
        
        local currentHRP = targetCharacter:FindFirstChild("HumanoidRootPart")
        if not currentHRP then
            return false
        end
        
        local currentPosition = currentHRP.Position
        local currentTarget = currentHRP
        
        if playerMouse then
            playerMouse.Hit = CFrame.new(currentPosition)
            playerMouse.Target = currentTarget
            currentPosition = playerMouse.Hit.Position
            currentTarget = playerMouse.Target
        end
        
        local webSlingArgs = {currentPosition, currentTarget}
        
        useItemRemote:FireServer(unpack(webSlingArgs))
        return true
    end
    
    if not targetCharacter or not targetCharacter.Parent then
        ShowNotification("Target lost!", 1)
        if characterSounds then
            characterSounds.Disabled = false
        end
        return false
    end
    
    targetHRP = targetCharacter:FindFirstChild("HumanoidRootPart")
    if not targetHRP then
        ShowNotification("Target missing HumanoidRootPart!", 1)
        if characterSounds then
            characterSounds.Disabled = false
        end
        return false
    end
    
    FireWebSlingAtCurrentPosition()

    local webTarget = targetHRP:WaitForChild("WebTargetAttch", 2)
    if not webTarget then
        ShowNotification("Failed to websling!", 1)
        if characterSounds then
            characterSounds.Disabled = false
        end
        return false
    end

    IsControlling = true
    ControlledPlayer = targetPlayer

    local webRope = Workspace:FindFirstChild("WebRope")
    if webRope and webRope:IsA("RopeConstraint") then
        webRope.Length = 1000
    end

    LocalPlayer.Character = targetCharacter
    local targetHumanoid = targetCharacter:FindFirstChildOfClass("Humanoid")
    if targetHumanoid then
        Workspace.CurrentCamera.CameraSubject = targetHumanoid
    end

    local startTime = tick()
    local controlDuration = Config.ControlDuration
    
    task.spawn(function()
        while IsControlling and ControlledPlayer == targetPlayer and (tick() - startTime) < controlDuration do
            local elapsed = tick() - startTime
            local remaining = math.max(0, controlDuration - elapsed)
            if TimerLabel then
                TimerLabel.Text = string.format("%.1fs", remaining)
            end
            task.wait(0.1)
        end
        if TimerLabel then
            TimerLabel.Text = ""
        end
    end)

    task.delay(controlDuration, function()
        if myCharacter and myCharacter.Parent then
            LocalPlayer.Character = myCharacter
            local myHumanoid = myCharacter:FindFirstChildOfClass("Humanoid")
            if myHumanoid then
                Workspace.CurrentCamera.CameraSubject = myHumanoid
            end
        end
        if characterSounds then
            characterSounds.Disabled = false
        end
        IsControlling = false
        ControlledPlayer = nil
        if TimerLabel then
            TimerLabel.Text = ""
        end
        ShowNotification("Control released automatically", 1)
    end)
    
    ShowNotification("Successfully controlling " .. targetPlayer.Name, 2)
    return true
end

local function ReleaseControl()
    if not IsControlling then return end
    
    local myCharacter = LocalPlayer.Character
    if myCharacter and myCharacter.Parent then
        LocalPlayer.Character = myCharacter
        local myHumanoid = myCharacter:FindFirstChildOfClass("Humanoid")
        if myHumanoid then
            Workspace.CurrentCamera.CameraSubject = myHumanoid
        end
    end
    
    local characterSounds = LocalPlayer.PlayerScripts:FindFirstChild("RbxCharacterSounds")
    if characterSounds then
        characterSounds.Disabled = false
    end
    
    IsControlling = false
    ControlledPlayer = nil
    if TimerLabel then
        TimerLabel.Text = ""
    end
    ShowNotification("Control released manually", 1)
end

if Config.AutoControl then
    task.spawn(function()
        while Config.AutoControl do
            if not IsControlling then
                local target = FindNearestPlayer()
                if target then
                    ControlPlayer(target)
                end
            end
            task.wait(1)
        end
    end)
end
