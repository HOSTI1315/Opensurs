local tabs = {}
local currentTab = nil
local minimized = false

minimizeBtn.MouseButton1Click:Connect(function()
	minimized = not minimized
	for _, v in pairs(f:GetChildren()) do
		if v:IsA("Frame") or v:IsA("TextButton") then
			if v ~= minimizeBtn and v ~= closeBtn and v ~= title then
				v.Visible = not minimized
			end
		end
	end
end)

uis.InputBegan:Connect(function(i, p)
	if not p and i.KeyCode == Enum.KeyCode.RightShift then
		f.Visible = not f.Visible
	end
end)

local function createTab(name)
	local button = Instance.new("TextButton", tabBar)
	button.Size = UDim2.new(0, 90, 1, 0)
	button.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
	button.Text = name
	button.Font = Enum.Font.GothamBold
	button.TextSize = 13
	button.TextColor3 = Color3.new(1, 1, 1)
	Instance.new("UICorner", button).CornerRadius = UDim.new(0, 6)

	local content = Instance.new("Frame", tabContent)
	content.Size = UDim2.new(1, 0, 1, 0)
	content.BackgroundTransparency = 1
	content.Visible = false

	local layout = Instance.new("UIListLayout", content)
	layout.Padding = UDim.new(0, 5)
	layout.FillDirection = Enum.FillDirection.Vertical
	layout.HorizontalAlignment = Enum.HorizontalAlignment.Center
	layout.SortOrder = Enum.SortOrder.LayoutOrder

	button.MouseButton1Click:Connect(function()
		if currentTab then currentTab.Visible = false end
		currentTab = content
		content.Visible = true
	end)

	tabs[name] = content
	return content
end

local function addBtn(tab, txt, callback)
	local b = Instance.new("TextButton", tab)
	b.Size = UDim2.new(0, 260, 0, 30)
	b.BackgroundColor3 = Color3.fromRGB(255, 80, 80)
	b.Text = txt
	b.TextColor3 = Color3.new(1,1,1)
	b.Font = Enum.Font.GothamSemibold
	b.TextSize = 14
	b.AutoButtonColor = true
	Instance.new("UICorner", b).CornerRadius = UDim.new(0, 6)
	b.MouseButton1Click:Connect(callback)
end

	noclipOn = not noclipOn
end)

rs.Stepped:Connect(function()
	if noclipOn and l.Character then
		for _, v in pairs(l.Character:GetDescendants()) do
			if v:IsA("BasePart") then v.CanCollide = false end
		end
	end
end)

addBtn(visualsTab, "Reveal Glass Tiles", function()
	local ts = game:GetService("TweenService")
	local h = workspace:FindFirstChild("GlassBridge") and workspace.GlassBridge:FindFirstChild("GlassHolder")
	if not h then return end
	for _, pair in pairs(h:GetChildren()) do
		for _, m in pairs(pair:GetChildren()) do
			if m:IsA("Model") and m.PrimaryPart then
				local p = m.PrimaryPart
				for _, c in ipairs(m:GetChildren()) do
					if c:IsA("Highlight") then c:Destroy() end
				end
				local b = p:GetAttribute("exploitingisevil") == true
				local color = b and Color3.fromRGB(255,0,0) or Color3.fromRGB(0,255,0)
				for _, part in pairs(m:GetDescendants()) do
					if part:IsA("BasePart") then
						ts:Create(part, TweenInfo.new(0.5, Enum.EasingStyle.Linear), { Transparency = 0.5, Color = color }):Play()
					end
				end
				local h = Instance.new("Highlight")
				h.FillColor = color
				h.FillTransparency = 0.7
				h.OutlineTransparency = 0.5
				h.Parent = m
			end
		end
	end
end)
