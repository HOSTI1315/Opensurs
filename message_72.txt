local function fetchKeys(url)
    local success, response = pcall(function()
        return game:HttpGet(url)
    end)
    if not success or not response then
        return {}
    end

    local successDecode, decodedResponse = pcall(function()
        return HttpService:JSONDecode(response)
    end)

    if successDecode and type(decodedResponse) == "table" then
        local keys = {}
        for _, key in ipairs(decodedResponse) do
            if type(key) == "string" and #key > 0 then
                table.insert(keys, key)
            end
        end
        return keys
    end

    local cleanedResponse = response:gsub("\r", "\n")
    local keys = {}
    for line in cleanedResponse:gmatch("[^\n,]+") do
        line = line:match("^%s*(.-)%s*$")
        if line ~= "" then
            table.insert(keys, line)
        end
    end
    return keys
end

local function findBallShadow()
    for _, descendant in ipairs(workspace:GetDescendants()) do

local function updateCourtAndTeam()
    local character = workspace:FindFirstChild(localPlayer.Name)
    if not character then return end
    local gameId = character:GetAttribute("GameId")
    local team = character:GetAttribute("Team")
    if gameId and team then
        local court = workspace.Courts:FindFirstChild(tostring(gameId))
        if court and court:FindFirstChild("Map") then
            currentCourt = court
            currentNet = currentCourt.Map.Colliders:FindFirstChild("Net")
        end
        currentTeam = team
        warn("[AutoPlay] Updated to Court", gameId, "Team", team)
    end
end

local function isBallInServeSpawnArea(characterPart)
    if not currentCourt or not currentCourt:FindFirstChild("Map") then
        return false
    end
    local serveSpawns = currentCourt.Map:FindFirstChild("ServeSpawns")
    if not serveSpawns then return false end
    local spawnArea = serveSpawns:FindFirstChild(tostring(currentTeam))
    if not spawnArea then return false end
    for _, part in ipairs(spawnArea:GetChildren()) do
        if part:IsA("BasePart") and (characterPart.Position - part.Position).Magnitude < 15 then
            return true
        end
    end
    return false
end

local function autoPlayBall()
    if not currentCourt or not currentCourt:FindFirstChild("Map") then return end
    local serveSpawns = currentCourt.Map:FindFirstChild("ServeSpawns")
    if not serveSpawns then return end
    local spawnArea = serveSpawns:FindFirstChild(tostring(currentTeam))
    if not spawnArea then return end

    return function(characterPart)
        for _, spawnPart in ipairs(spawnArea:GetChildren()) do
            if spawnPart:IsA("BasePart") and (characterPart.Position - spawnPart.Position).Magnitude < 15 then
                return true
            end
        end
        return false
    end
end

local function aimAndStrike()
    local connection
    if connection then connection:Disconnect() end
    connection = RunService.Heartbeat:Connect(function()
        if not STATE.AutoPlayBall then return end
        local character = getCharacter()
        local humanoidRootPart = character:FindFirstChild("HumanoidRootPart")
        local ballShadow = findBallShadow()
        if not humanoidRootPart or not currentNet or not currentTeam then return end
        local currentTime = tick()

        local isInServeSpawn = isBallInServeSpawnArea(humanoidRootPart)
        if isInServeSpawn and (currentTime - lastBallTouchTime) >= 0.5 then
            VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.F, false, game)
            task.wait(0.05)
            VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.F, false, game)
            lastBallTouchTime = currentTime
        end

        if currentTeam then
            local netPosZ = currentNet.Position.Z
            local targetPosZ = humanoidRootPart.Position.Z
            if netPosZ < targetPosZ then
                targetPosZ = 1
            end
            if targetPosZ == currentTeam then
                humanoidRootPart.CFrame = CFrame.new(humanoidRootPart.CFrame.Position, currentNet.CFrame.Position + Vector3.new(0, 8, 0))
                if (currentTime - lastBallTouchTime) >= 0.5 then
                    VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.F, false, game)
                    task.wait(0.04)
                    VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.F, false, game)
                    lastBallTouchTime = currentTime
                end
            end
        end
    end)
end

local aimConnection = nil
local aimTargetIndex = 1
local aimTargetRandom = 1.2

local function startAiming()
    if aimConnection then aimConnection:Disconnect() end
    aimConnection = RunService.RenderStepped:Connect(function()
        if not STATE.AutoAimSpawns or not currentCourt or not currentTeam then return end
        local spawns = currentCourt.Map:FindFirstChild("Spawns")
        if not spawns then return end

        local targetSpawnIndex = currentTeam == 1 and 2 or 1
        local spawnPoint = spawns:FindFirstChild(tostring(targetSpawnIndex))
        if not spawnPoint then return end

        local spawnTarget = spawnPoint:FindFirstChild(tostring(aimTargetIndex))
        if not spawnTarget then return end

        local position = spawnTarget.Position
        if aimTargetIndex == 2 then
            position = position + Vector3.new(10, 0, 0)
        elseif aimTargetIndex == 3 then
            position = position + Vector3.new(-20, 0, 0)
        end

        local camera = workspace.CurrentCamera
        if camera then
            camera.CFrame = CFrame.new(camera.CFrame.Position, position)
        end
    end)

    task.spawn(function()
        while STATE.AutoAimSpawns do
            aimTargetIndex = math.random(1, 3)
            task.wait(aimTargetRandom)
        end
    end)
end

local function stopAiming()
    if aimConnection then
        aimConnection:Disconnect()
        aimConnection = nil
    end
end

local function parseKeyCode(input)
    local keyStr = tostring(input or ""):upper():gsub("%s+", "")
    if #keyStr == 1 and keyStr:match("%a") then
        return Enum.KeyCode[keyStr]
    end
    if keyMap[keyStr] then
        return Enum.KeyCode[keyMap[keyStr]]
    end
    if keyNameMap[keyStr] then
        return Enum.KeyCode[keyNameMap[keyStr]]
    end
    local matchKey = keyStr:match("^%[?([A-Z0-9])")
    if matchKey then
        if keyMap[matchKey] then
            return Enum.KeyCode[keyMap[matchKey]]
        end
        if matchKey:match("%a") then
            return Enum.KeyCode[matchKey]
        end
    end
    return nil
end

local function getNumberFromString(str)
    return tonumber(str:match("(%d+)")) or 0
end
