-- Utility Functions
local function getCharacterParts(char)
    if not char then return nil, nil, nil end
    return char:FindFirstChild("HumanoidRootPart"),
           char:FindFirstChild("Humanoid"),
           char:FindFirstChild("Communicate")
end

local function isAttacking(char)
    local hum = char:FindFirstChild("Humanoid")
    if not hum then return false, false end
    
    local anim = hum:FindFirstChildOfClass("Animator")
    if not anim then return false, false end
    
    for _, track in ipairs(anim:GetPlayingAnimationTracks()) do
        local animId = tostring(track.Animation.AnimationId):match("%d+")
        if animId then
            if dashAnimSet[animId] then return false, true end
            if attackAnimSet[animId] then return true, false end
        end
    end
    
    return false, false
end

local function isMovingTowards(enemyHRP, myPos)
    local vel = enemyHRP.Velocity
    if vel.Magnitude < 10 then return false end
    
    local dirToPlayer = (myPos - enemyHRP.Position).Unit
    local velNorm = vel.Unit
    return velNorm:Dot(dirToPlayer) > 0.7
end

local function getTool()
    return LocalPlayer.Backpack:FindFirstChild("Prey's Peril") 
        or (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Prey's Peril"))
end

-- Counter Logic
local function fireCounter()
    if isCurrentlyCountering then return end
    isCurrentlyCountering = true
    
    task.spawn(function()
        if CounterReactionTime > 0 then task.wait(CounterReactionTime) end
        
        local tool = getTool()
        local _, _, communicate = getCharacterParts(LocalPlayer.Character)
        
        if tool and communicate then
            local success = pcall(function()
                communicate:FireServer({ 
                    Tool = tool, 
                    Goal = "Console Move", 
                    ToolName = "Prey's Peril" 
                })
                
                task.wait(0.2)
                communicate:FireServer({ 
                    Goal = "Auto Use End", 
                    Tool = tool 
                })
            end)
        end
        
        task.wait(0.3)
        isCurrentlyCountering = false
    end)
end

-- Block Logic
local function pressBlock()
    if isCurrentlyBlocking then return end
    isCurrentlyBlocking = true
    
    task.spawn(function()
        task.wait(BlockReactionTime)
        
        local _, _, communicate = getCharacterParts(LocalPlayer.Character)
        if not communicate then 
            isCurrentlyBlocking = false
            return 
        end
        
        pcall(function()
            communicate:FireServer({ Goal = "KeyPress", Key = Enum.KeyCode.F })
        end)
        
        local blockStartTime = tick()
        while tick() - blockStartTime < BlockHoldTime do
            if isLeftClickHeld then break end
            task.wait(0.05)
        end
        
        pcall(function()
            communicate:FireServer({ Goal = "KeyRelease", Key = Enum.KeyCode.F })
        end)
        
        isCurrentlyBlocking = false
    end)
end

-- Shared Detection Logic
local function checkThreats(myPos, distAttack, distDash, trackedTable, actionCallback)
    for _, enemy in ipairs(Players:GetPlayers()) do
        if enemy == LocalPlayer then continue end
        
        local enemyChar = enemy.Character
        if not enemyChar then continue end
        
        local eHRP, eHum = getCharacterParts(enemyChar)
        if not eHRP or not eHum or eHum.Health <= 0 then continue end
        
        local dist = (eHRP.Position - myPos).Magnitude
        local isAttack, isDash = isAttacking(enemyChar)
        
        local maxDist = math.max(distAttack, distDash)
        
        -- Clear tracking if out of range
        if dist >= maxDist or (not isAttack and not isDash) then
            trackedTable[enemy.UserId] = nil
            continue
        end
        
        -- Check if should trigger action
        local shouldAct = false
        if isAttack and dist < distAttack and isMovingTowards(eHRP, myPos) then
            shouldAct = true
        elseif isDash and dist < distDash and isMovingTowards(eHRP, myPos) then
            shouldAct = true
        end
        
        if shouldAct and not trackedTable[enemy.UserId] then
            trackedTable[enemy.UserId] = true
            actionCallback()
            return -- Only handle one threat per frame
        end
    end
end

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer

-- Animation IDs cần detect
local DASH_ANIM_IDS = {"13380255751", "10479335397"}

-- Cấu hình
local MAX_TARGET_DISTANCE = 50
local FOV_ANGLE = 120

-- State management
local lastDamagedPlayers = {}
local damageTimeout = 5
local currentTarget = nil
local isEnabled = false

-- Connections để cleanup
local connections = {}

-- Kiểm tra target ở trước mặt
local function isInFrontOfPlayer(rootPart, targetPosition)
    local lookVector = rootPart.CFrame.LookVector
    local directionToTarget = (targetPosition - rootPart.Position).Unit
    local dotProduct = lookVector:Dot(directionToTarget)
    local angle = math.deg(math.acos(math.clamp(dotProduct, -1, 1)))
    return angle <= (FOV_ANGLE / 2)
end

-- Track damage
local function trackDamage(targetPlayer)
    lastDamagedPlayers[targetPlayer] = tick()
end

-- Tìm target
local function findTargetInFront(rootPart)
    local validTargets = {}
    local currentTime = tick()
    
    -- Cleanup damaged players timeout
    for targetPlayer, damageTime in pairs(lastDamagedPlayers) do
        if currentTime - damageTime > damageTimeout then
            lastDamagedPlayers[targetPlayer] = nil
        end
    end
    
    -- Ưu tiên người vừa đánh
    for targetPlayer, damageTime in pairs(lastDamagedPlayers) do
        if currentTime - damageTime <= damageTimeout then
            if targetPlayer and targetPlayer.Character then
                local targetRoot = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                local targetHumanoid = targetPlayer.Character:FindFirstChild("Humanoid")
                
                if targetRoot and targetHumanoid and targetHumanoid.Health > 0 then
                    local distance = (rootPart.Position - targetRoot.Position).Magnitude
                    
                    if distance <= MAX_TARGET_DISTANCE and isInFrontOfPlayer(rootPart, targetRoot.Position) then
                        table.insert(validTargets, {
                            player = targetPlayer,
                            distance = distance,
                            root = targetRoot
                        })
                    end
                end
            end
        end
    end
    
    -- Nếu không có người vừa đánh, tìm người chơi gần nhất
    if #validTargets == 0 then
        for _, otherPlayer in pairs(Players:GetPlayers()) do
            if otherPlayer ~= player and otherPlayer.Character then
                local otherRoot = otherPlayer.Character:FindFirstChild("HumanoidRootPart")
                local otherHumanoid = otherPlayer.Character:FindFirstChild("Humanoid")
                
                if otherRoot and otherHumanoid and otherHumanoid.Health > 0 then
                    local distance = (rootPart.Position - otherRoot.Position).Magnitude
                    
                    if distance <= MAX_TARGET_DISTANCE and isInFrontOfPlayer(rootPart, otherRoot.Position) then
                        table.insert(validTargets, {
                            player = otherPlayer,
                            distance = distance,
                            root = otherRoot
                        })
                    end
                end
            end
        end
    end
    
    if #validTargets > 0 then
        table.sort(validTargets, function(a, b) return a.distance < b.distance end)
        return validTargets[1].player, validTargets[1].root
    end
    
    return nil, nil
end

-- Setup damage detection cho character
local function setupDamageDetection(character)
    local function onToolAdded(tool)
        if tool:IsA("Tool") then
            local handle = tool:FindFirstChild("Handle")
            if handle then
                local conn = handle.Touched:Connect(function(hit)
                    if not isEnabled then return end
                    
                    local targetChar = hit.Parent
                    if targetChar and targetChar:FindFirstChild("Humanoid") then
                        local targetPlayer = Players:GetPlayerFromCharacter(targetChar)
                        if targetPlayer and targetPlayer ~= player then
                            trackDamage(targetPlayer)
                        end
                    end
                end)
                table.insert(connections, conn)
            end
        end
    end
    
    local conn1 = character.ChildAdded:Connect(onToolAdded)
    table.insert(connections, conn1)
    
    for _, tool in pairs(character:GetChildren()) do
        onToolAdded(tool)
    end
end

-- Monitor animation để tự động set target
local function monitorDashAnimation(character)
    local humanoid = character:WaitForChild("Humanoid")
    local rootPart = character:WaitForChild("HumanoidRootPart")
    
    local conn = humanoid.AnimationPlayed:Connect(function(animTrack)
        -- CHECK: Không chạy nếu không enabled HOẶC S+Dash đang active
        if not isEnabled then return end
        if _G.isSDashActive == true then return end
        
        local animId = animTrack.Animation.AnimationId
        
        -- Kiểm tra có phải dash animation không
        for _, dashId in pairs(DASH_ANIM_IDS) do
            if animId:find(dashId) then
                -- Tìm và set target ngay khi animation bắt đầu
                local target, targetRoot = findTargetInFront(rootPart)
                
                if target and targetRoot then
                    currentTarget = targetRoot
                    
                    -- Bắt đầu auto-aim
                    local originalAutoRotate = humanoid.AutoRotate
                    humanoid.AutoRotate = false
                    
                    local aimConnection
                    aimConnection = RunService.Heartbeat:Connect(function()
                        -- Kiểm tra các điều kiện dừng
                        if not isEnabled then
                            if aimConnection then aimConnection:Disconnect() end
                            pcall(function()
                                if humanoid and humanoid.Parent then
                                    humanoid.AutoRotate = originalAutoRotate
                                end
                            end)
                            currentTarget = nil
                            return
                        end
                        
                        -- Dừng nếu S+Dash active
                        if _G.isSDashActive == true then
                            if aimConnection then aimConnection:Disconnect() end
                            pcall(function()
                                if humanoid and humanoid.Parent then
                                    humanoid.AutoRotate = originalAutoRotate
                                end
                            end)
                            currentTarget = nil
                            return
                        end
                        
                        -- Dừng nếu animation không còn play hoặc target mất
                        if not animTrack.IsPlaying or not currentTarget or not currentTarget.Parent then
                            if aimConnection then aimConnection:Disconnect() end
                            pcall(function()
                                if humanoid and humanoid.Parent then
                                    humanoid.AutoRotate = originalAutoRotate
                                end
                            end)
                            currentTarget = nil
                            return
                        end
                        
                        -- Xoay nhân vật về target
                        pcall(function()
                            if rootPart and rootPart.Parent and currentTarget and currentTarget.Parent then
                                local targetPos = currentTarget.Position
                                local lookAtTarget = CFrame.new(
                                    rootPart.Position, 
                                    Vector3.new(targetPos.X, rootPart.Position.Y, targetPos.Z)
                                )
                                rootPart.CFrame = lookAtTarget
                            end
                        end)
                    end)
                    
                    table.insert(connections, aimConnection)
                end
                
                break
            end
        end
    end)
    
    table.insert(connections, conn)
end

-- Cleanup tất cả connections
local function cleanup()
    for _, conn in pairs(connections) do
        if conn and conn.Connected then
            conn:Disconnect()
        end
    end
    connections = {}
    
    currentTarget = nil
    lastDamagedPlayers = {}
    
    -- Restore AutoRotate nếu có character
    pcall(function()
        if player.Character then
            local humanoid = player.Character:FindFirstChild("Humanoid")
            if humanoid then
                humanoid.AutoRotate = true
            end
        end
    end)
end

    local rootPart = character:WaitForChild("HumanoidRootPart")
    
    if not humanoid or not rootPart then return end
    
    setupDamageDetection(character)
    monitorDashAnimation(character)
end

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local char, root, humanoid

-- Animation IDs
local DASH_ANIM_IDS = {["13380255751"] = true, ["10479335397"] = true}
local BLOCK_ANIM_ID = "10470389827", "13935548552"

-- Constants
local TAU = math.pi * 2
local HALF_PI = math.pi * 0.5

-- Config
local config = {
    -- Target Dash
    targetMaxDistance = 50,
    targetFOV = 120,
    damageTimeout = 5,
    
    -- Teleport Behind
    behindDistance = 7,
    teleportTriggerDistance = 10,
    aimDelay = 0.5
}

-- State
local state = {
    enabled = false,
    target = nil,
    targetPlayer = nil,
    mode = "IDLE", -- "IDLE", "TARGET_DASH", "DASH_TO_TELEPORT", "TELEPORT_BEHIND"
    
    -- Teleport Behind state
    lockedHeight = nil,
    phase = "IDLE",
    behindCheckTime = 0,
    lastCleanup = 0
}

local damagedPlayers = {}
local connections = {}
local targetBlockAnimTrack = nil

-- ============================================================================
-- UTILITY FUNCTIONS
-- ============================================================================

local function isInFOV(fromPos, lookVec, toPos, fovAngle)
    local dir = (toPos - fromPos).Unit
    local dot = lookVec:Dot(dir)
    return math.acos(math.clamp(dot, -1, 1)) <= math.rad(fovAngle * 0.5)
end

local function cleanupDamagedPlayers()
    local now = tick()
    if now - state.lastCleanup < 1 then return end
    
    state.lastCleanup = now
    for p, t in pairs(damagedPlayers) do
        if now - t > config.damageTimeout or not p.Parent then
            damagedPlayers[p] = nil
        end
    end
end

-- ============================================================================
-- BLOCK DETECTION
-- ============================================================================

local function isTargetBlockingPlayer()
    if not state.target or not state.targetPlayer then return false end
    if not root then return false end
    
    local targetChar = state.targetPlayer.Character
    if not targetChar then return false end
    
    local targetHumanoid = targetChar:FindFirstChild("Humanoid")
    if not targetHumanoid then return false end
    
    -- Check khoảng cách
    local distance = (root.Position - state.target.Position).Magnitude
    if distance > config.targetMaxDistance then return false end
    
    -- Check target có đang play block animation không
    local isBlocking = false
    for _, track in pairs(targetHumanoid:GetPlayingAnimationTracks()) do
        local animId = track.Animation.AnimationId:match("%d+")
        if animId == BLOCK_ANIM_ID then
            isBlocking = true
            targetBlockAnimTrack = track
            break
        end
    end
    
    if not isBlocking then 
        targetBlockAnimTrack = nil
        return false 
    end
    
    -- Check player có trong FOV của target không
    local targetPos = state.target.Position
    local targetLookVec = state.target.CFrame.LookVector
    local playerPos = root.Position
    
    return isInFOV(targetPos, targetLookVec, playerPos, config.targetFOV)
end

-- ============================================================================
-- TARGET FINDING
-- ============================================================================

local function findTarget()
    if not root then return nil, nil end
    
    cleanupDamagedPlayers()
    
    local rootPos = root.Position
    local lookVec = root.CFrame.LookVector
    local bestTarget, bestPlayer, bestDist = nil, nil, math.huge
    
    -- Priority 1: Recently damaged players
    for p in pairs(damagedPlayers) do
        if p.Character then
            local r = p.Character:FindFirstChild("HumanoidRootPart")
            local h = p.Character:FindFirstChild("Humanoid")
            
            if r and h and h.Health > 0 then
                local dist = (rootPos - r.Position).Magnitude
                if dist <= config.targetMaxDistance and dist < bestDist and isInFOV(rootPos, lookVec, r.Position, config.targetFOV) then
                    bestTarget = r
                    bestPlayer = p
                    bestDist = dist
                end
            end
        end
    end
    
    if bestTarget then return bestTarget, bestPlayer end
    
    -- Priority 2: Any valid player in FOV
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character then
            local r = p.Character:FindFirstChild("HumanoidRootPart")
            local h = p.Character:FindFirstChild("Humanoid")
            
            if r and h and h.Health > 0 then
                local dist = (rootPos - r.Position).Magnitude
                if dist <= config.targetMaxDistance and dist < bestDist and isInFOV(rootPos, lookVec, r.Position, config.targetFOV) then
                    bestTarget = r
                    bestPlayer = p
                    bestDist = dist
                end
            end
        end
    end
    
    return bestTarget, bestPlayer
end

-- ============================================================================
-- TARGET DASH LOGIC
-- ============================================================================

local function updateTargetDash()
    if not state.enabled or not root or not state.target or not state.target.Parent then
        return false
    end
    
    -- Xoay về target
    pcall(function()
        if root and root.Parent and state.target and state.target.Parent then
            local targetPos = state.target.Position
            local lookAtTarget = CFrame.new(
                root.Position, 
                Vector3.new(targetPos.X, root.Position.Y, targetPos.Z)
            )
            root.CFrame = lookAtTarget
        end
    end)
    
    return true
end

-- ============================================================================
-- DASH TO TELEPORT LOGIC
-- ============================================================================

local function updateDashToTeleport()
    if not state.enabled or not root or not state.target or not state.target.Parent then
        return false
    end
    
    -- Tính khoảng cách đến target
    local distance = (root.Position - state.target.Position).Magnitude
    
    -- Nếu đã đến gần target đủ (9 studs), chuyển sang teleport
    if distance <= config.teleportTriggerDistance then
        print("[HYBRID DASH] Distance reached, teleporting behind...")
        state.mode = "TELEPORT_BEHIND"
        state.lockedHeight = root.Position.Y
        state.phase = "TELEPORT"
        return true
    end
    
    -- Vẫn còn xa, tiếp tục dash về phía target
    pcall(function()
        if root and root.Parent and state.target and state.target.Parent then
            local targetPos = state.target.Position
            local lookAtTarget = CFrame.new(
                root.Position, 
                Vector3.new(targetPos.X, root.Position.Y, targetPos.Z)
            )
            root.CFrame = lookAtTarget
        end
    end)
    
    return true
end

-- ============================================================================
-- TELEPORT BEHIND LOGIC
-- ============================================================================

local function updateTeleportBehind(dt)
    if not state.enabled or not root or not state.target or not state.target.Parent then
        return false
    end
    
    local targetPos = state.target.Position
    
    -- Phase 1: Teleport behind target
    if state.phase == "TELEPORT" then
        -- Lấy hướng nhìn của target
        local targetLook = Vector3.new(
            state.target.CFrame.LookVector.X,
            0,
            state.target.CFrame.LookVector.Z
        ).Unit
        
        -- Tính vị trí phía sau target (distance = 7 studs)
        local behindPos = Vector3.new(
            targetPos.X - targetLook.X * config.behindDistance,
            state.lockedHeight,
            targetPos.Z - targetLook.Z * config.behindDistance
        )
        
        -- Teleport thẳng về phía sau
        root.CFrame = CFrame.new(behindPos, Vector3.new(targetPos.X, state.lockedHeight, targetPos.Z))
        
        state.phase = "BEHIND_AIM"
        state.behindCheckTime = tick()
        
    -- Phase 2: Aim at target from behind
    elseif state.phase == "BEHIND_AIM" then
        local targetLook = Vector3.new(
            state.target.CFrame.LookVector.X,
            0,
            state.target.CFrame.LookVector.Z
        ).Unit
        
        local behindPos = Vector3.new(
            targetPos.X - targetLook.X * config.behindDistance,
            state.lockedHeight,
            targetPos.Z - targetLook.Z * config.behindDistance
        )
        
        -- Giữ vị trí và aim vào target
        root.CFrame = CFrame.new(behindPos, Vector3.new(targetPos.X, state.lockedHeight, targetPos.Z))
        
        if tick() - state.behindCheckTime >= config.aimDelay then
            return false
        end
    end
    
    return true
end

local function setupDamageDetection()
    local function onToolAdded(tool)
        if not tool:IsA("Tool") then return end
        
        local handle = tool:FindFirstChild("Handle")
        if not handle then return end
        
        local conn = handle.Touched:Connect(function(hit)
            if not state.enabled then return end
            
            local targetChar = hit.Parent
            if not targetChar then return end
            
            local targetHum = targetChar:FindFirstChild("Humanoid")
            if not targetHum then return end
            
            local targetPlayer = Players:GetPlayerFromCharacter(targetChar)
            if targetPlayer and targetPlayer ~= player then
                damagedPlayers[targetPlayer] = tick()
            end
        end)
        
        table.insert(connections, conn)
    end
    
    table.insert(connections, char.ChildAdded:Connect(onToolAdded))
    
    for _, tool in ipairs(char:GetChildren()) do
        onToolAdded(tool)
    end
end

local function startDashMode(dashAnimTrack, targetRoot, targetPlr)
    state.target = targetRoot
    state.targetPlayer = targetPlr
    
    -- Check nếu target đang block và nhìn player
    local shouldTeleportBehind = isTargetBlockingPlayer()
    
    if shouldTeleportBehind then
        -- DASH TO TELEPORT MODE
        state.mode = "DASH_TO_TELEPORT"
        
        print("[HYBRID DASH] Dashing to target (will teleport at 9 studs)")
    else
        -- TARGET DASH MODE
        state.mode = "TARGET_DASH"
        print("[HYBRID DASH] Target Dash activated")
    end
    
    -- Save and disable auto-rotate
    local originalAutoRotate = humanoid.AutoRotate
    humanoid.AutoRotate = false
    
    -- Main dash loop
    local dashConn
    dashConn = RunService.Heartbeat:Connect(function(dt)
        -- Stop conditions
        if not state.enabled or not dashAnimTrack.IsPlaying then
            if dashConn then dashConn:Disconnect() end
            pcall(function()
                if humanoid and humanoid.Parent then
                    humanoid.AutoRotate = originalAutoRotate
                end
            end)
            state.target = nil
            state.targetPlayer = nil
            state.mode = "IDLE"
            state.lockedHeight = nil
            state.phase = "IDLE"
            targetBlockAnimTrack = nil
            return
        end
        
        -- Check nếu target bắt đầu block giữa chừng -> chuyển sang Dash to Teleport
        if state.mode == "TARGET_DASH" and isTargetBlockingPlayer() then
            print("[HYBRID DASH] Target started blocking, switching to dash-teleport mode")
            state.mode = "DASH_TO_TELEPORT"
        end
        
        -- Update based on mode
        local success, shouldContinue = true, true
        
        if state.mode == "TARGET_DASH" then
            success, shouldContinue = pcall(updateTargetDash)
        elseif state.mode == "DASH_TO_TELEPORT" then
            success, shouldContinue = pcall(updateDashToTeleport)
        elseif state.mode == "TELEPORT_BEHIND" then
            success, shouldContinue = pcall(updateTeleportBehind, dt)
        end
        
        if not success or not shouldContinue then
            if dashConn then dashConn:Disconnect() end
            pcall(function()
                if humanoid and humanoid.Parent then
                    humanoid.AutoRotate = originalAutoRotate
                end
            end)
            state.target = nil
            state.targetPlayer = nil
            state.mode = "IDLE"
            state.lockedHeight = nil
            state.phase = "IDLE"
            targetBlockAnimTrack = nil
        end
    end)
    
    table.insert(connections, dashConn)
end

local function monitorAnimations()
    local conn = humanoid.AnimationPlayed:Connect(function(track)
        if not state.enabled then return end
        if _G.isSDashActive == true then return end
        
        local animId = track.Animation.AnimationId:match("%d+")
        if not animId or not DASH_ANIM_IDS[animId] then return end
        
        local targetRoot, targetPlr = findTarget()
        if not targetRoot then return end
        
        startDashMode(track, targetRoot, targetPlr)
    end)
    
    table.insert(connections, conn)
end

local function cleanup()
    for _, conn in ipairs(connections) do
        if conn.Connected then
            conn:Disconnect()
        end
    end
    table.clear(connections)
    
    state.target = nil
    state.targetPlayer = nil
    state.mode = "IDLE"
    state.lockedHeight = nil
    state.phase = "IDLE"
    table.clear(damagedPlayers)
    targetBlockAnimTrack = nil
    
    if humanoid and humanoid.Parent then
        humanoid.AutoRotate = true
    end
end

    cleanup()
    
    char = character
    root = character:WaitForChild("HumanoidRootPart")
    humanoid = character:WaitForChild("Humanoid")
    
    if not state.enabled then return end
    if not root or not humanoid then return end
    
    setupDamageDetection()
    monitorAnimations()
end

end

local function onSDashAnimation(animationTrack)
    if not isSDashEnabled or isSDashProcessing then return end
    
    local animId = animationTrack.Animation.AnimationId
    animId = animId:gsub("http://www.roblox.com/asset/%?id=", "rbxassetid://")
    animId = animId:gsub("https://www.roblox.com/asset/%?id=", "rbxassetid://")
    
    if animId == "rbxassetid://13294471966" then
        isSDashProcessing = true
        _G.isSDashActive = true -- Đánh dấu S+Dash đang active
        
        task.wait(0.4)
        pressKey(Enum.KeyCode.S)
        task.wait(0.05)
        pressKey(Enum.KeyCode.Q)
        task.wait(0.1)
        
        _G.isSDashActive = false -- Đánh dấu S+Dash đã xong
        isSDashProcessing = false
    end
end

end
