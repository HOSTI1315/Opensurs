--// get sub prefix or smth lmfao
function GetSubPrefix(str)
    local str = tostring(str):gsub(" ", "")
    local var = ""

    if #str == 2 then
        local sec = string.sub(str, #str, #str + 1)
        var = sec == "1" and "st" or sec == "2" and "nd" or sec == "3" and "rd" or "th"
    end

    return var
end

local function getEnumMaterial(materialString)
    local materialMap = {
        ["Neon"] = Enum.Material.Neon,
        ["ForceField"] = Enum.Material.ForceField,
        ["Plastic"] = Enum.Material.Plastic,
    }
    return materialMap[materialString] or Enum.Material.Neon
end

task.spawn(function()
    while true do
        wait()
        if Settings.Visuals.Local.CloneChams.Enabled then
            repeat
                lplr.Character.Archivable = true
                local Clone = lplr.Character:Clone()
                for _, Obj in next, Clone:GetDescendants() do
                    if Obj.Name == "HumanoidRootPart" or Obj:IsA("Humanoid") or Obj:IsA("LocalScript") or Obj:IsA("Script") or Obj:IsA("Decal") then
                        Obj:Destroy()
                    elseif Obj:IsA("BasePart") or Obj:IsA("MeshPart") or Obj:IsA("Part") then
                        if Obj.Transparency == 1 then
                            Obj:Destroy()
                        else
                            Obj.CanCollide = false
                            Obj.Anchored = true
                            Obj.Material = getEnumMaterial(Settings.Visuals.Local.CloneChams.Material)

                            Obj.Color = Settings.Visuals.Local.CloneChams.Color
                            Obj.Transparency = 0
                            Obj.Size = Obj.Size + Vector3.new(0.03, 0.03, 0.03)
                        end
                    end
                end

                Clone.Parent = workspace
                wait(Settings.Visuals.Local.CloneChams.Duration)
                Clone:Destroy()
            until not Settings.Visuals.Local.CloneChams.Enabled
        end
    end
end)

local a = false -- Variable to track the toggle state

-- Adding the "AutoFarm (Nearest Players)" toggle inside the group
Main:AddToggle('MyToggle', {
    Text = 'AutoFarm (Nearest Players)',
    Default = false, -- Default value (true / false)
    Tooltip = 'AutoFarm Nearest Players (Turn Off If Finishing)', -- Information shown when you hover over the toggle

    Callback = function(Value)
        a = Value -- Set 'a' to true or false depending on the toggle state
        if a then
            -- Start the autofarm loop when the toggle is enabled
            startAutoFarm()
        else
            -- Stop the autofarm loop when the toggle is disabled
            stopAutoFarm()
        end
    end
})

local autofarmTask -- This will hold the task reference to stop the loop later

-- Function to start the autofarm task
function startAutoFarm()
    autofarmTask = task.spawn(function()
        while a do
            task.wait(0.02)
            pcall(function()
                local player = game.Players.LocalPlayer
                local char = player.Character
                local char_root = char and char:FindFirstChild("HumanoidRootPart")

                if not char_root then
                    -- Character not loaded yet, wait for it
                    char = player.CharacterAdded:Wait()
                    char_root = char:WaitForChild("HumanoidRootPart")
                end

                local function findNearestPlayer()
                    local nearest_dist = math.huge
                    local target_root

                    for _, targetPlayer in pairs(game.Players:GetPlayers()) do
                        if targetPlayer ~= player and targetPlayer.Character then
                            local targetChar = targetPlayer.Character
                            local targetHumanoidRoot = targetChar:FindFirstChild("HumanoidRootPart")
                            if targetHumanoidRoot then
                                local dist = (char_root.Position - targetHumanoidRoot.Position).magnitude

                                if dist < nearest_dist and dist < 8 then -- Target players within 10 studs
                                    local obstructed = false

                                    -- Check for obstacles between the player and target
                                    local ray = Ray.new(char_root.Position, (targetHumanoidRoot.Position - char_root.Position).unit * 20)
                                    local part, _ = workspace:FindPartOnRayWithIgnoreList(ray, { char }, false, true)

                                    if part and part.Parent ~= targetChar then
                                        obstructed = true
                                    end

                                    if not obstructed then
                                        nearest_dist = dist
                                        target_root = targetHumanoidRoot
                                    end
                                end
                            end
                        end
                    end

                    return target_root
                end

                local function farm()
                    local target_root = findNearestPlayer()

                    if target_root then
                        -- Move towards the nearest player (teleporting fast)
                        local behind = (target_root.CFrame * CFrame.new(0, 0, 3)).p
                        local lookAt = (target_root.Position - char_root.Position).unit

                        char_root.CFrame = CFrame.new(behind)
                        char_root.CFrame = CFrame.new(behind, behind + Vector3.new(lookAt.X, 0, lookAt.Z))
                    end
                end

                -- Throttle the script to run at most once every 0.5 seconds
                local lastRunTime = 0
                local throttleInterval = 0.05

                while a do
                    local currentTime = tick()
                    if currentTime - lastRunTime >= throttleInterval then
                        farm()
                        lastRunTime = currentTime
                    end
                    wait() -- Adjust the wait time to balance responsiveness and performance
                end
            end)
        end
    end)
end

-- Function to stop the autofarm task
function stopAutoFarm()
    if autofarmTask then
        task.cancel(autofarmTask) -- Stop the task when the toggle is turned off
        autofarmTask = nil
    end
end

    return qLockValue == true

-- Closest player finder (within MAX_DISTANCE)
function getClosestPlayer()
    local closest, minDist = nil, math.huge
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character and player.Character:FindFirstChild("Head") then
            local headPart = player.Character.Head
            local distance = (headPart.Position - Camera.CFrame.Position).Magnitude

            if distance <= MAX_DISTANCE then -- only if nearby
                local screenPos, onScreen = Camera:WorldToViewportPoint(headPart.Position)
                if onScreen then
                    local screenDist = (Vector2.new(screenPos.X, screenPos.Y) - Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)).Magnitude
                    if screenDist < minDist then
                        minDist = screenDist
                        closest = player
                    end
                end
            end
        end
    end
    return closest
end

-- Services
local run_service = game:GetService("RunService");
local players = game:GetService("Players");
local replicated_storage = game:GetService("ReplicatedStorage");
local virtual_input_manager = game:GetService("VirtualInputManager");
local tween_service = game:GetService("TweenService");

-- Variables
local local_player = players.LocalPlayer;
local player_gui = local_player.PlayerGui;
local camera = workspace.CurrentCamera;

local map = workspace.Map;
local tiles = map.Tiles;

local steak_house = tiles.ShoppingTile.SteakHouse;
local steak_interior = steak_house.Interior;

local updated = tick();

-- Main
do
    local get_proximity_prompt = function(ancestor)
        for _, descendant in ancestor:GetDescendants() do
            if (descendant.ClassName == "ProximityPrompt") then
                return descendant;
            end
        end
    end

    local visualize_click = function(x, y, lifetime)
        local circle = Drawing.new("Circle");
        circle.Visible = true;
        circle.Transparency = 1;
        circle.Filled = true;
        circle.Radius = 10;
        circle.Color = Color3.fromRGB(255, 255, 255);

        circle.Position = Vector2.new(x, y);
        task.wait(lifetime);
        circle:Destroy();
    end

    local click_button = function(button, offset)
        local absolute_position = button.AbsolutePosition;
        local absolute_size = button.AbsoluteSize;
    
        local center_x = absolute_position.X + (absolute_size.X / 2);
        local center_y = absolute_position.Y + (absolute_size.Y * 2);
    
        if (offset) then
            center_x = center_x + offset.X;
            center_y = center_y + offset.Y;
        end
    
        local screen_size = camera.ViewportSize;
        center_x, center_y = math.clamp(center_x, 0, screen_size.X), math.clamp(center_y, 0, screen_size.Y);
        visualize_click(center_x, center_y, 0.3);
    
        virtual_input_manager:SendMouseButtonEvent(center_x, center_y, 0, true, game, 1);
        task.wait();
        virtual_input_manager:SendMouseButtonEvent(center_x, center_y, 0, false, game, 1);
    end

    local teleport = function(position, time)
        time = time or 3.5;

        local character = local_player.Character;
        local root = character:FindFirstChild("HumanoidRootPart");

        if (not root) then
            return;
        end

        local completed = false;
        local tween = tween_service:Create(root, TweenInfo.new(time, Enum.EasingStyle.Linear, Enum.EasingDirection.Out), {CFrame = typeof(position) == "CFrame" and position or CFrame.new(position)});
        tween:Play();
        tween.Completed:Connect(function()
            completed = true;
        end)

        repeat
            task.wait();
        until completed; -- plz yield until completed, bro thx!
    end

    run_service.RenderStepped:Connect(function()
        if (Toggles.autofarm_enabled.Value and tick() - updated >= .3) then
            local character = local_player.Character;
            local root = (character and character:FindFirstChild("HumanoidRootPart"));
            
            if (root) then
                local proximity_prompt = get_proximity_prompt(steak_interior.Fridge);
                local job = local_player:GetAttribute("Job");
                
                if (job ~= "steakhouse_cook") then
                    local touch = steak_interior.SteakHouseBeacon.TouchPart;

                    if (touch) then
                        teleport(touch.CFrame.Position);
                        task.spawn(click_button, player_gui.JobApplication.JobApplicationFrame.ApplyJob);
                    end
                else
                    --[[
                    if (Toggles.safe_mode.Value) then
                        pcall(function()
                            require(replicated_storage.Modules.Core.Net).send("request_respawn");
                        end)
                    end
                    ]]

                    local progress_bar_frame = player_gui.ProgressBar.ProgressBarFrame;
                    
                    local main_frame = progress_bar_frame.MainFrame;
                    local input_frame = progress_bar_frame.InputFrame;

                    if (proximity_prompt and not progress_bar_frame.Visible) then
                        if (Toggles.get_steak.Value) then
                            local end_cf = proximity_prompt.Parent.Parent.CFrame;
                            teleport((end_cf.Position + (end_cf.LookVector * -2)) + Vector3.new(0, 2.5, 0));

                            proximity_prompt:InputHoldBegin();
                            proximity_prompt:InputHoldEnd();
                        end
                    end
                    
                    local beacon = workspace:FindFirstChild("Beacon");
                    local base = (beacon and beacon.Base);
    
                    if (base) then
                        teleport(base.CFrame + vector.create(0, 1, 0));
                    end
    
                    local bar = main_frame.BarAmount;
                    local cooked_perfectly = main_frame.CookedPerfectly;
    
                    if (progress_bar_frame.Visible and bar and cooked_perfectly) then
                        if (bar.Size.X.Scale >= cooked_perfectly.Position.X.Scale) then
                            task.spawn(click_button, input_frame.FinishBar);
                        end
                    end
                end

                updated = tick();
            end
        end
    end)
end
