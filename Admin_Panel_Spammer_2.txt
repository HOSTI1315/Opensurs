
local env = getgenv and getgenv() or _G
if env.__XD_SPAM_PANEL__ then return end
env.__XD_SPAM_PANEL__ = true


repeat task.wait() until game:IsLoaded()


local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local StarterGui = game:GetService("StarterGui")
local CoreGui = game:GetService("CoreGui")

local LocalPlayer = Players.LocalPlayer


local SHOW_NOTIFICATIONS = false
local COMMANDS = { "rocket", "balloon", "inverse", "tiny" }


local function notify(title, text, duration)
	if not SHOW_NOTIFICATIONS then return end
	pcall(function()
		StarterGui:SetCore("SendNotification", {
			Title = title,
			Text = text,
			Duration = duration or 3
		})
	end)
end


local function findRemote(timeout)
	timeout = timeout or 6
	local start = os.clock()

	while os.clock() - start < timeout do
		local packages = ReplicatedStorage:FindFirstChild("Packages")
		if packages then
			local net = packages:FindFirstChild("Net")
			if net then
				local remote = net:FindFirstChild("RE/AdminPanelService/ExecuteCommand")
				if remote and remote:IsA("RemoteEvent") then
					return remote
				end
			end
		end
		task.wait(0.1)
	end

	return nil
end

local ExecuteCommand = findRemote(6)
if not ExecuteCommand then
	notify("Missing Remote", "ExecuteCommand not found", 5)
	return
end


local function formatPlayer(player)
	if player.DisplayName and player.DisplayName ~= player.Name then
		return string.format("%s (@%s)", player.DisplayName, player.Name)
	end
	return "@" .. player.Name
end

local function runCommands(player)
	if not player or player == LocalPlayer then return end

	task.spawn(function()
		for _, cmd in ipairs(COMMANDS) do
			pcall(function()
				ExecuteCommand:FireServer(player, cmd)
			end)
		end
		notify("Sent", "→ " .. player.Name, 2)
	end)
end

local function createPlayerButton(player)
	local btn = Instance.new("TextButton")
	btn.Size = UDim2.new(1, -6, 0, 34)
	btn.BackgroundColor3 = Color3.fromRGB(38,38,38)
	btn.TextColor3 = Color3.fromRGB(235,235,235)
	btn.Font = Enum.Font.Gotham
	btn.TextSize = 14
	btn.TextXAlignment = Left
	btn.Text = "  " .. formatPlayer(player)

	Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 8)

	btn.Activated:Connect(function()
		runCommands(player)
	end)

	return btn
end

local function refreshList()
	ListFrame:ClearAllChildren()
	Layout.Parent = ListFrame

	for _, player in ipairs(Players:GetPlayers()) do
		if player ~= LocalPlayer then
			createPlayerButton(player).Parent = ListFrame
		end
	end
end



Refresh.MouseButton1Click:Connect(refreshList)
Players.PlayerAdded:Connect(refreshList)
Players.PlayerRemoving:Connect(refreshList)

refreshList()
notify("xd spam ready", "Click a player to run commands", 4)
