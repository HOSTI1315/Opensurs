-- Get environment and services
local Env = getfenv()  -- Get current environment
local RunService_2 = game:GetService("RunService")  -- Game loop service
local LocalPlayer = game:GetService("Players").LocalPlayer  -- Your player

-- Function to start velocity movement
Env.StartVelocityMove = function(speedValue)
    local speed = speedValue[1]  -- Get speed value
    if not speed then return end  -- Stop if no speed
    
    -- Create movement loop
    local Connection_3
    Connection_3 = RunService_2.Heartbeat:Connect(function(DeltaTime)
        local Character_2 = LocalPlayer.Character  -- Your character
        if not Character_2 then return end  -- Stop if no character
        
        local HumanoidRootPart = Character_2:FindFirstChild("HumanoidRootPart")  -- Root part
        local Humanoid_3 = Character_2:FindFirstChildOfClass("Humanoid")  -- Humanoid
        
        if not HumanoidRootPart then return end  -- Stop if no root part
        if not Humanoid_3 then return end  -- Stop if no humanoid
        
        -- Calculate movement direction
        local MoveDirection = Humanoid_3.MoveDirection  -- Current movement
        local horizontalMove = Vector3.new(MoveDirection.X, 0, MoveDirection.Z) * speed  -- Apply speed
        
        -- Apply velocity smoothing
        local currentVelocity = HumanoidRootPart.Velocity  -- Current velocity
        HumanoidRootPart.Velocity = currentVelocity:Lerp(
            Vector3.new(horizontalMove.X, HumanoidRootPart.Velocity.Y, horizontalMove.Z),  -- Target velocity
            0.45  -- Smoothing factor
        )
    end)
end

-- Function to stop velocity movement
Env.StopVelocityMove = function()
    if Connection_3 then
        Connection_3:Disconnect()  -- Stop movement loop
    end
end
