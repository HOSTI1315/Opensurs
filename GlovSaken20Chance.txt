-- Variables
local chanceActive = false
local chanceAimDuration = 1.6
local chanceAimTargets = {"Slasher", "c00lkidd", "JohnDoe", "1x1x1x1", "Noli", "Sixer", "Nosferatu"}
local chancePredictionValue = 4
local chanceMovementThreshold = 0.5
local chancePlayerHumanoid, chancePlayerHRP = nil, nil
local chanceLastTriggerTime = 0
local chanceAiming = false
local chanceOriginalWS, chanceOriginalAutoRotate = nil, nil
local chanceActiveTrack = nil

-- Animation IDs that trigger aim
local chanceAimAnimationIds = {
    "133491532453922",
    "133607163653602",
    "76649505662612",
    "86371356500204"
}

-- Setup character references
local function chanceConfigureCharacter(char)
    chancePlayerHumanoid = char:FindFirstChild("Humanoid")
    chancePlayerHRP = char:FindFirstChild("HumanoidRootPart")
end

if playersService.LocalPlayer.Character then
    chanceConfigureCharacter(playersService.LocalPlayer.Character)
end
playersService.LocalPlayer.CharacterAdded:Connect(chanceConfigureCharacter)

-- Find valid killer target
local function chanceFindValidTarget()
    local killersFolder = workspaceService:FindFirstChild("Players") and workspaceService.Players:FindFirstChild("Killers")
    if not killersFolder then return nil end
    for _, name in ipairs(chanceAimTargets) do
        local target = killersFolder:FindFirstChild(name)
        if target and target:FindFirstChild("HumanoidRootPart") then
            return target.HumanoidRootPart
        end
    end
    return nil
end

-- Simple studs prediction
local function chancePredictPosition(targetHRP, studs)
    if not targetHRP then return nil end
    if targetHRP.Velocity.Magnitude > chanceMovementThreshold then
        return targetHRP.Position + (targetHRP.CFrame.LookVector * studs)
    end
    return targetHRP.Position
end

-- Detect aim animation start
local function chanceCheckAimAnimation()
    if not chancePlayerHumanoid or chanceActiveTrack then return false end
    for _, track in ipairs(chancePlayerHumanoid:GetPlayingAnimationTracks()) do
        local id = track.Animation.AnimationId:match("%d+")
        if id and table.find(chanceAimAnimationIds, id) then
            chanceActiveTrack = track
            track.Stopped:Once(function()
                chanceActiveTrack = nil
            end)
            return true
        end
    end
    return false
end

-- Main aim loop
runService.RenderStepped:Connect(function()
    if not chanceActive or not chancePlayerHumanoid or not chancePlayerHRP then
        if chanceAiming then
            chanceAiming = false
            if chanceOriginalWS and chanceOriginalAutoRotate ~= nil then
                chancePlayerHumanoid.WalkSpeed = chanceOriginalWS
                chancePlayerHumanoid.AutoRotate = chanceOriginalAutoRotate
                chanceOriginalWS, chanceOriginalAutoRotate = nil, nil
            end
        end
        return
    end

    if not chanceAiming and chanceCheckAimAnimation() then
        chanceLastTriggerTime = tick()
        chanceAiming = true
        chanceOriginalWS = chancePlayerHumanoid.WalkSpeed
        chanceOriginalAutoRotate = chancePlayerHumanoid.AutoRotate
    end

    if chanceAiming then
        local elapsed = tick() - chanceLastTriggerTime
        local targetHRP = chanceFindValidTarget()

        if not targetHRP or (targetHRP.Position - chancePlayerHRP.Position).Magnitude < 6 or elapsed > chanceAimDuration then
            chanceAiming = false
            if chanceOriginalWS and chanceOriginalAutoRotate ~= nil then
                chancePlayerHumanoid.WalkSpeed = chanceOriginalWS
                chancePlayerHumanoid.AutoRotate = chanceOriginalAutoRotate
                chanceOriginalWS, chanceOriginalAutoRotate = nil, nil
            end
            return
        end

        chancePlayerHumanoid.AutoRotate = false
        chancePlayerHRP.AssemblyAngularVelocity = Vector3.zero

        local predicted = chancePredictPosition(targetHRP, chancePredictionValue)
        if predicted then
            local dir = (predicted - chancePlayerHRP.Position).Unit
            local yaw = math.atan2(-dir.X, -dir.Z)
            chancePlayerHRP.CFrame = CFrame.new(chancePlayerHRP.Position) * CFrame.Angles(0, yaw, 0)
        end
    end
end)
