-- Função Draggable
local function makeDraggable(obj)
    local dragging, dragInput, dragStart, startPos
    local function update(input)
        local delta = input.Position - dragStart
        obj.Position = UDim2.new(
            startPos.X.Scale, startPos.X.Offset + delta.X,
            startPos.Y.Scale, startPos.Y.Offset + delta.Y
        )
    end

    obj.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = obj.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    obj.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input == dragInput then
            update(input)
        end
    end)
end

local function makeToggleButton(name, parent, callback)
    local btn = Instance.new("TextButton")
    btn.Parent = parent
    btn.Size = UDim2.new(1,-20,0,50)
    btn.Text = name
    btn.TextColor3 = Color3.fromRGB(0,0,0)
    btn.Font = Enum.Font.GothamBold
    btn.TextSize = 18
    btn.BackgroundColor3 = Color3.fromRGB(255,40,255)
    local uic = Instance.new("UICorner")
    uic.CornerRadius = UDim.new(0,8)
    uic.Parent = btn

    local active = false

    btn.MouseButton1Click:Connect(function()
        active = not active
        local goal = {BackgroundColor3 = active and Color3.fromRGB(0,255,0) or Color3.fromRGB(255,40,255)}
        TweenService:Create(btn,TweenInfo.new(0.5,Enum.EasingStyle.Quad), goal):Play()

        -- Splash efeito
        local splash = Instance.new("Frame")
        splash.Size = UDim2.new(0,0,0,0)
        splash.Position = UDim2.new(0.5,0,0.5,0)
        splash.BackgroundColor3 = Color3.fromRGB(255,255,255)
        splash.AnchorPoint = Vector2.new(0.5,0.5)
        splash.BackgroundTransparency = 0.5
        splash.ZIndex = btn.ZIndex+1
        splash.Parent = btn
        TweenService:Create(splash, TweenInfo.new(0.3,Enum.EasingStyle.Quad), {Size=UDim2.new(2,0,2,0), BackgroundTransparency = 1}):Play()
        task.delay(0.3,function() splash:Destroy() end)

        -- Som do botão
        local btnSound = Instance.new("Sound")
        btnSound.SoundId = "rbxassetid://9083627113"
        btnSound.Volume = 1
        btnSound.Parent = btn
        btnSound:Play()
        task.delay(1,function() btnSound:Destroy() end)

        callback(active)
    end)

    return btn
end

-- Botão Plataforma V2
----------------------------------------------------
makeToggleButton("Plataforma V2", ScrollFrame, function(active)
    platformActive = active
    if platformActive then
        character = player.Character or player.CharacterAdded:Wait()

        local function createPlatform()
            if platform and platform.Parent then return platform end
            local hrp = character:FindFirstChild("HumanoidRootPart")
            if not hrp then return end

            local p = Instance.new("Part")
            p.Size = PLATFORM_SIZE
            p.Anchored = true
            p.CanCollide = true
            p.Material = Enum.Material.Neon
            p.Color = Color3.fromRGB(0,140,255)
            p.Transparency = 0.15
            p.Name = "Hyper_Platform"
            p.CFrame = hrp.CFrame * CFrame.new(0, -OFFSET_BELOW - (PLATFORM_SIZE.Y/2),0)
            p.Parent = Workspace

            platform = p
            return p
        end

        local function startMoving()
            if conn then return end
            conn = RunService.Heartbeat:Connect(function()
                if not platform or not platform.Parent then return end
                local hrp = character:FindFirstChild("HumanoidRootPart")
                if not hrp then return end
                platform.CFrame = CFrame.new(hrp.Position - Vector3.new(0, OFFSET_BELOW,0))
            end)
        end

        createPlatform()
        startMoving()
    else
        if platform then platform:Destroy() platform=nil end
        if conn then conn:Disconnect() conn=nil end
    end
end)

-- Botão FPS KILLER
----------------------------------------------------
local FPSDevourer = {}
do
    FPSDevourer.running = false
    local TOOL_NAME = "Bat"
    local function equipBat()
        local character = player.Character
        local backpack = player:FindFirstChild("Backpack")
        if not character or not backpack then return false end
        local tool = backpack:FindFirstChild(TOOL_NAME)
        if tool then tool.Parent = character return true end
        return false
    end
    local function unequipBat()
        local character = player.Character
        local backpack = player:FindFirstChild("Backpack")
        if not character or not backpack then return false end
        local tool = character:FindFirstChild(TOOL_NAME)
        if tool then tool.Parent = backpack return true end
        return false
    end
    local function removeAllAccessoriesFromCharacter()
        local character = player.Character
        if not character then return end
        for _, item in ipairs(character:GetChildren()) do
            if item:IsA("Accessory")
                or item:IsA("LayeredClothing")
                or item:IsA("Shirt")
                or item:IsA("ShirtGraphic")
                or item:IsA("Pants")
                or item:IsA("BodyColors")
                or item:IsA("CharacterMesh") then
                pcall(function() item:Destroy() end)
            end
        end
    end

    makeToggleButton("FPS KILLER", ScrollFrame, function(active)
        if active then
            FPSDevourer.running = true
            FPSDevourer._stop = false
            removeAllAccessoriesFromCharacter()
            task.spawn(function()
                while FPSDevourer.running and not FPSDevourer._stop do
                    equipBat()
                    task.wait(0.035)
                    unequipBat()
                    task.wait(0.035)
                end
            end)
        else
            FPSDevourer.running = false
            FPSDevourer._stop = true
            unequipBat()
        end
    end)

    player.CharacterAdded:Connect(function()
        FPSDevourer.running = false
        FPSDevourer._stop = true
        task.wait(0.2)
        removeAllAccessoriesFromCharacter()
    end)
end

-- Botão Float Walk suave e lenta
----------------------------------------------------
makeToggleButton("Float Walk", ScrollFrame, function(active)
    character = player.Character or player.CharacterAdded:Wait()
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not hrp then return end

    floatWalkActive = active

    if floatWalkActive then
        floatWalkForce = Instance.new("BodyVelocity")
        floatWalkForce.MaxForce = Vector3.new(1e5,1e5,1e5)
        floatWalkForce.Velocity = Vector3.new(0,0,0)
        floatWalkForce.Parent = hrp

        floatConn = RunService.Heartbeat:Connect(function()
            if floatWalkActive and hrp.Parent then
                local cam = workspace.CurrentCamera
                local moveDir = Vector3.new(cam.CFrame.LookVector.X, 0, cam.CFrame.LookVector.Z)
                if moveDir.Magnitude > 0 then
                    moveDir = moveDir.Unit
                else
                    moveDir = Vector3.new(0,0,0)
                end
                floatWalkForce.Velocity = moveDir * floatSpeed
            end
        end)
    else
        if floatWalkForce then floatWalkForce:Destroy() floatWalkForce=nil end
        if floatConn then floatConn:Disconnect() floatConn=nil end
    end
end)
