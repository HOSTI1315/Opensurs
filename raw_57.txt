-- Movement variables
local movementConnection
local centerPosition
local elapsedTime = 0
local radius = 15 -- small circle radius
local angularSpeed = math.pi / 4 -- radians per second (adjust speed)

-- Recent spawn positions to avoid repeat spawns
local recentSpawns = {}
local MAX_RECENT = 5
local MIN_DISTANCE = 3 -- studs

local function distance(a, b)
	return (a - b).Magnitude
end

-- Start moving player in a small circle around initial position
function startMovement()
	local player = game.Players.LocalPlayer
	local character = player.Character or player.CharacterAdded:Wait()
	local root = character:WaitForChild("HumanoidRootPart")
	centerPosition = root.Position
	elapsedTime = 0

	movementConnection = RunService.Heartbeat:Connect(function(dt)
		elapsedTime = elapsedTime + dt
		local angle = elapsedTime * angularSpeed

		local offsetX = math.cos(angle) * radius
		local offsetZ = math.sin(angle) * radius
		local newPos = Vector3.new(centerPosition.X + offsetX, root.Position.Y, centerPosition.Z + offsetZ)

		root.CFrame = CFrame.new(newPos, newPos + root.CFrame.LookVector)
	end)
end

-- Stop player movement
function stopMovement()
	if movementConnection then
		movementConnection:Disconnect()
		movementConnection = nil
	end
end

-- Main auto place coroutine
task.spawn(function()
	local lastPlayerPos = nil

	while true do
		if autoPlace then
			local player = game.Players.LocalPlayer
			local character = player.Character or player.CharacterAdded:Wait()
			local root = character:WaitForChild("HumanoidRootPart")
			local currentPos = root.Position

			-- Reset recent spawns if player moved >1 stud
			if lastPlayerPos and (currentPos - lastPlayerPos).Magnitude > 1 then
				recentSpawns = {}
			end
			lastPlayerPos = currentPos

			local unitFolder = workspace:FindFirstChild("UnitFolder")
			if unitFolder then
				local foundModel = false
				for _, unit in ipairs(unitFolder:GetChildren()) do
					if unit:IsA("Model") then
						foundModel = true
						local unitName = unit.Name

						local playerCFrame = root.CFrame
						local groundY = 3.75 -- fixed working ground height

						local spawnPosition
						local tries = 0
						repeat
							local offsetX = (math.random(0,1) == 0 and -1 or 1) * math.random(3,5)
							local candidatePos = Vector3.new(playerCFrame.Position.X + offsetX, groundY, playerCFrame.Position.Z)

							-- Check if candidatePos is too close to recent spawns
							local tooClose = false
							for _, recentPos in ipairs(recentSpawns) do
								if distance(candidatePos, recentPos) < MIN_DISTANCE then
									tooClose = true
									break
								end
							end

							if not tooClose then
								spawnPosition = candidatePos
							end

							tries = tries + 1
						until spawnPosition or tries >= 10

						-- If no good spot found, just use last candidate
						if not spawnPosition then
							spawnPosition = Vector3.new(playerCFrame.Position.X + ((math.random(0,1) == 0 and -1 or 1) * math.random(3,5)), groundY, playerCFrame.Position.Z)
						end

						table.insert(recentSpawns, spawnPosition)
						if #recentSpawns > MAX_RECENT then
							table.remove(recentSpawns, 1)
						end

						local spawnCFrame = CFrame.new(spawnPosition)

						local args = {
							"GameStuff",
							{
								"Summon",
								unitName,
								spawnCFrame
							}
						}

						pcall(function()
							game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("SetEvent"):FireServer(unpack(args))
						end)

						task.wait(0.5)
					end
				end

				if not foundModel then
					StarterGui:SetCore("SendNotification", {
						Title = "Auto Place";
						Text = "Place the troops you want to auto place first!";
						Duration = 5;
					})
				end
			else
				StarterGui:SetCore("SendNotification", {
					Title = "Auto Place";
					Text = "UnitFolder not found in Workspace!";
					Duration = 5;
				})
			end
		end
		task.wait(1)
	end
end)

local movementConnection = nil  -- Used to store the movement loop connection

-- Function to start CFrame movement
function startCFrameSpeed()
    if movementConnection == nil then
        movementConnection = RunService.Stepped:Connect(function()
            LocalPlayer.Character.HumanoidRootPart.CFrame = LocalPlayer.Character.HumanoidRootPart.CFrame + LocalPlayer.Character.Humanoid.MoveDirection * Multiplier
        end)
    end
end

-- Function to stop CFrame movement
function stopCFrameSpeed()
    if movementConnection then
        movementConnection:Disconnect()
        movementConnection = nil
    end
end

-- Key bindings for increasing and decreasing speed multiplier
UserInputService.InputBegan:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.LeftBracket then
        Multiplier = Multiplier + 0.01
        print(Multiplier)
        wait(0.2)
        while UserInputService:IsKeyDown(Enum.KeyCode.LeftBracket) do
            wait()
            Multiplier = Multiplier + 0.01
            print(Multiplier)
        end
    elseif input.KeyCode == Enum.KeyCode.RightBracket then
        Multiplier = Multiplier - 0.01
        print(Multiplier)
        wait(0.2)
        while UserInputService:IsKeyDown(Enum.KeyCode.RightBracket) do
            wait()
            Multiplier = Multiplier - 0.01
            print(Multiplier)
        end
    end
end)

local function createTutorialGui()
    -- Create the GUI
    local Turtorial = Instance.new("ScreenGui")
    local main = Instance.new("Frame")
    local JoinButton = Instance.new("TextButton")
    local UICorner_Join = Instance.new("UICorner")
    local shadow = Instance.new("Frame")

    Turtorial.Name = "Turtorial"
    Turtorial.Parent = player:WaitForChild("PlayerGui")
    Turtorial.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

    main.Name = "main"
    main.Parent = Turtorial
    main.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    main.AnchorPoint = Vector2.new(0, 0)

    -- Function to position the GUI just below the top-left corner of the screen
    local function setPositionBelowTopLeft()
        local screenSize = game:GetService("Workspace").CurrentCamera.ViewportSize
        main.Position = UDim2.new(0, 10, 0, 10)  -- Just below the top-left with a slight offset
        main.Size = UDim2.new(0.15, 0, 0.10, 0)  -- Standard size for both devices
    end

    setPositionBelowTopLeft()

    main.Active = true
    main.Draggable = true

    -- Shadow effect for main frame
    shadow.Name = "shadow"
    shadow.Parent = main
    shadow.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    shadow.BackgroundTransparency = 0.6
    shadow.Position = UDim2.new(0, 5, 0, 5)
    shadow.Size = UDim2.new(1, 0, 1, 0)
    shadow.ZIndex = 0

    JoinButton.Name = "JoinButton"
    JoinButton.Parent = main
    JoinButton.BackgroundColor3 = Color3.fromRGB(40, 120, 255)

    -- Adjust button size based on device type
    if game:GetService("UserInputService").TouchEnabled then
        JoinButton.Size = UDim2.new(0.9, 0, 0.4, 0)
    else
        JoinButton.Size = UDim2.new(0.9, 0, 0.4, 0)
    end

    JoinButton.Position = UDim2.new(0.05, 0, 0.5, 0)
    JoinButton.Font = Enum.Font.Highway
    JoinButton.Text = "Toggle Gui"
    JoinButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    JoinButton.TextSize = 25
    JoinButton.TextStrokeTransparency = 0.7
    JoinButton.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)

    UICorner_Join.CornerRadius = UDim.new(0.15, 0)
    UICorner_Join.Parent = JoinButton

    -- Toggle the visibility of ScreenGuis
    JoinButton.MouseButton1Down:Connect(function()
        local CoreGui = game:GetService("CoreGui")
        local function containsNumber(name)
            return name:match("%d") ~= nil
        end

        local function toggleScreenGuisWithNumbers()
            for _, gui in ipairs(CoreGui:GetChildren()) do
                if gui:IsA("ScreenGui") and containsNumber(gui.Name) and gui.Name ~= "Turtorial" then
                    gui.Enabled = not gui.Enabled
                    print((gui.Enabled and "Enabled" or "Disabled") .. " ScreenGui:", gui.Name)
                end
            end
        end

        toggleScreenGuisWithNumbers()

        CoreGui.ChildAdded:Connect(function(child)
            if child:IsA("ScreenGui") then
                task.wait(0.1)
                if containsNumber(child.Name) and child.Name ~= "Turtorial" then
                    child.Enabled = false
                    print("Disabled new ScreenGui:", child.Name)
                end
            end
        end)
    end)
end
