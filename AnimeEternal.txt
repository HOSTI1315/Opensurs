local function updateToggle()
    local tweenInfo = TweenInfo.new(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)
    
    if AutoFarm then
        StatusLabel.Text = "ON"
        local colorTween = TweenService:Create(StatusLabel, tweenInfo, {TextColor3 = Color3.fromRGB(80, 255, 120)})
        local strokeTween = TweenService:Create(ToggleStroke, tweenInfo, {Color = Color3.fromRGB(80, 255, 120)})
        colorTween:Play()
        strokeTween:Play()
    else
        StatusLabel.Text = "OFF"
        local colorTween = TweenService:Create(StatusLabel, tweenInfo, {TextColor3 = Color3.fromRGB(255, 80, 80)})
        local strokeTween = TweenService:Create(ToggleStroke, tweenInfo, {Color = Color3.fromRGB(70, 70, 90)})
        colorTween:Play()
        strokeTween:Play()
    end
end

ToggleButton.MouseButton1Click:Connect(function()
    AutoFarm = not AutoFarm
    updateToggle()
end)

ToggleButton.MouseEnter:Connect(function()
    local hoverTween = TweenService:Create(ToggleFrame, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(50, 50, 65)})
    hoverTween:Play()
end)

ToggleButton.MouseLeave:Connect(function()
    local leaveTween = TweenService:Create(ToggleFrame, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(40, 40, 55)})
    leaveTween:Play()
end)

RunService.RenderStepped:Connect(function()
    local Character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    if AutoFarm and Character and Character:FindFirstChild("HumanoidRootPart") then
        local Closest, ClosestMag = nil, 30
        for _,Enemy in workspace.Debris.Monsters:GetChildren() do
            if Enemy:FindFirstChild("HumanoidRootPart")
            and Enemy:GetAttribute("Health")
            and Enemy:FindFirstChild("Humanoid") then
                local mag = (Character.HumanoidRootPart.Position - Enemy.HumanoidRootPart.Position).Magnitude
                if mag < ClosestMag then
                    ClosestMag = mag
                    Closest = Enemy
                end
            end
        end
        if Closest then
            ReplicatedStorage.Events.To_Server:FireServer({Id = Closest.Name, Action = "_Mouse_Click"})
        else
            ReplicatedStorage.Events.To_Server:FireServer({Action = "_Mouse_Click"})
        end
    end
end)
