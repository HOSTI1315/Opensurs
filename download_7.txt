--// Dupe Logic

local dupeRunning = false

local dupeConnection

local function startDupe()

	if dupeRunning then return end

	dupeRunning = true

	DupeBtn.Text = "Stop CANDY Dupe"

	DupeBtn.BackgroundColor3 = Color3.fromRGB(255, 60, 60)

	

	dupeConnection = RunService.Heartbeat:Connect(function()

		if CandyEvent then

			CandyEvent:FireServer()

		end

	end)

end

local function stopDupe()

	if not dupeRunning then return end

	dupeRunning = false

	DupeBtn.Text = "Start CANDY Dupe"

	DupeBtn.BackgroundColor3 = Color3.fromRGB(40, 120, 255)

	

	if dupeConnection then

		dupeConnection:Disconnect()

		dupeConnection = nil

	end

end

--// Button Control

DupeBtn.MouseButton1Click:Connect(function()

	if dupeRunning then

		stopDupe()

	else

		startDupe()

	end

end)
